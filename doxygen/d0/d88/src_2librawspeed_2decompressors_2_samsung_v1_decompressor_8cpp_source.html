65 inline int32_t
66 SamsungV1Decompressor::samsungDiff(BitPumpMSB* pump,
67  const std::vector<encTableItem>& tbl) {
68  pump->fill(23); // That is the maximal number of bits we will need here.
69  // We read 10 bits to index into our table
70  uint32_t c = pump->peekBitsNoFill(10);
71  // Skip the bits that were used to encode this case
72  pump->skipBitsNoFill(tbl[c].encLen);
73  // Read the number of bits the table tells me
74  int32_t len = tbl[c].diffLen;
75  if (len == 0)
76  return 0;
77  int32_t diff = pump->getBitsNoFill(len);
78  // If the first bit is 0 we need to turn this into a negative number
79  diff = HuffmanTable::extend(diff, len);
80  return diff;
81 }

83 void SamsungV1Decompressor::decompress() {
84  // This format has a variable length encoding of how many bits are needed
85  // to encode the difference between pixels, we use a table to process it
86  // that has two values, the first the number of bits that were used to
87  // encode, the second the number of bits that come after with the difference
88  // The table has 14 entries because the difference can have between 0 (no
89  // difference) and 13 bits (differences between 12 bits numbers can need 13)
90  static const std::array<std::array<uint8_t, 2>, 14> tab = {{{3, 4},
91  {3, 7},
92  {2, 6},
93  {2, 5},
94  {4, 3},
95  {6, 0},
96  {7, 9},
97  {8, 10},
98  {9, 11},
99  {10, 12},
100  {10, 13},
101  {5, 1},
102  {4, 8},
103  {4, 2}}};
104  std::vector<encTableItem> tbl(1024);
105 
106  // We generate a 1024 entry table (to be addressed by reading 10 bits) by
107  // consecutively filling in 2^(10-N) positions where N is the variable number
108  // of bits of the encoding. So for example 4 is encoded with 3 bits so the
109  // first 2^(10-3)=128 positions are set with 3,4 so that any time we read 000
110  // we know the next 4 bits are the difference. We read 10 bits because that is
111  // the maximum number of bits used in the variable encoding (for the 12 and
112  // 13 cases)
113  uint32_t n = 0;
114  for (auto i : tab) {
115  for (int32_t c = 0; c < (1024 >> i[0]); c++) {
116  tbl[n].encLen = i[0];
117  tbl[n].diffLen = i[1];
118  n++;
119  }
120  }
121 
122  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
123  assert(out.width % 32 == 0 && "Should have even count of pixels per row.");
124  assert(out.height % 2 == 0 && "Should have even row count.");
125  BitPumpMSB pump(*bs);
126  for (int row = 0; row < out.height; row++) {
127  std::array<int, 2> pred = {{}};
128  if (row >= 2)
129  pred = {out(row - 2, 0), out(row - 2, 1)};
130 
131  for (int col = 0; col < out.width; col++) {
132  int32_t diff = samsungDiff(&pump, tbl);
133  pred[col & 1] += diff;
134 
135  int value = pred[col & 1];
136  if (!isIntN(value, bits))
137  ThrowRDE("decoded value out of bounds");
138  out(row, col) = value;
139  }
140  }
141 }
