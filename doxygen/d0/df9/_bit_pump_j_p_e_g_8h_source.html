42 inline BitPumpJPEG::size_type BitPumpJPEG::fillCache(const uint8_t* input) {
43  static_assert(BitStreamCacheBase::MaxGetBits >= 32, "check implementation");
44 
45  // short-cut path for the most common case (no FF marker in the next 4 bytes)
46  // this is slightly faster than the else-case alone.
47  // TODO: investigate applicability of vector intrinsics to speed up if-cascade
48  if (input[0] != 0xFF &&
49  input[1] != 0xFF &&
50  input[2] != 0xFF &&
51  input[3] != 0xFF ) {
52  cache.push(getBE<uint32_t>(input), 32);
53  return 4;
54  }
55 
56  size_type p = 0;
57  for (size_type i = 0; i < 4; ++i) {
58  // Pre-execute most common case, where next byte is 'normal'/non-FF
59  const int c0 = input[p++];
60  cache.push(c0, 8);
61  if (c0 == 0xFF) {
62  // Found FF -> pre-execute case of FF/00, which represents an FF data byte -> ignore the 00
63  const int c1 = input[p++];
64  if (c1 != 0) {
65  // Found FF/xx with xx != 00. This is the end of stream marker.
66  // That means we shouldn't have pushed last 8 bits (0xFF, from c0).
67  // We need to "unpush" them, and fill the vacant cache bits with zeros.
68 
69  // First, recover the cache fill level.
70  cache.fillLevel -= 8;
71  // Now, this code is incredibly underencapsulated, and the
72  // implementation details are leaking into here. Thus, we know that
73  // all the fillLevel bits in cache are all high bits. So to "unpush"
74  // the last 8 bits, and fill the vacant cache bits with zeros, we only
75  // need to keep the high fillLevel bits. So just create a mask with only
76  // high fillLevel bits set, and 'and' the cache with it.
77  // Caution, we know fillLevel won't be 64, but it may be 0,
78  // so pick the mask-creation idiom accordingly.
79  cache.cache &= ~((~0ULL) >> cache.fillLevel);
80  cache.fillLevel = 64;
81 
82  // No further reading from this buffer shall happen. Do signal that by
83  // claiming that we have consumed all the remaining bytes of the buffer.
84  return getRemainingSize();
85  }
86  }
87  }
88  return p;
89 }
