99 static inline void BM_RawSpeed(benchmark::State& state, const char* fileName,
100  int threads) {
101  currThreadCount = threads;
102 
103 #ifdef HAVE_PUGIXML
104  static const CameraMetaData metadata(RAWSPEED_SOURCE_DIR "/data/cameras.xml");
105 #else
106  static const CameraMetaData metadata{};
107 #endif
108 
109  FileReader reader(fileName);
110  const auto map(reader.readFile());
111 
112  Timer<ChooseClockType::type> WT;
113  Timer<CPUClock> TT;
114 
115  unsigned pixels = 0;
116  for (auto _ : state) {
117  RawParser parser(map.get());
118  auto decoder(parser.getDecoder(&metadata));
119 
120  decoder->failOnUnknown = false;
121  decoder->checkSupport(&metadata);
122 
123  decoder->decodeRaw();
124  decoder->decodeMetaData(&metadata);
125  RawImage raw = decoder->mRaw;
126 
127  benchmark::DoNotOptimize(raw);
128 
129  pixels = raw->getUncroppedDim().area();
130  }
131 
132  // These are total over all the `state.iterations()` iterations.
133  const double CPUTime = TT().count();
134  const double WallTime = WT().count();
135 
136  // For each iteration:
137  state.counters.insert({
138  {"CPUTime,s",
139  benchmark::Counter(CPUTime, benchmark::Counter::Flags::kAvgIterations)},
140  {"WallTime,s",
141  benchmark::Counter(WallTime, benchmark::Counter::Flags::kAvgIterations)},
142  {"CPUTime/WallTime", CPUTime / WallTime}, // 'Threading factor'
143  {"Pixels", pixels},
144  {"Pixels/CPUTime",
145  benchmark::Counter(pixels / CPUTime,
146  benchmark::Counter::Flags::kIsIterationInvariant)},
147  {"Pixels/WallTime",
148  benchmark::Counter(pixels / WallTime,
149  benchmark::Counter::Flags::kIsIterationInvariant)},
150  /* {"Raws", 1}, */
151  {"Raws/CPUTime",
152  benchmark::Counter(1.0 / CPUTime,
153  benchmark::Counter::Flags::kIsIterationInvariant)},
154  {"Raws/WallTime",
155  benchmark::Counter(1.0 / WallTime,
156  benchmark::Counter::Flags::kIsIterationInvariant)},
157  });
158  // Could also have counters wrt. the filesize,
159  // but i'm not sure they are interesting.
160 }

172 int main(int argc, char** argv) {
173  benchmark::Initialize(&argc, argv);
174 
175  auto hasFlag = [argc, argv](const std::string& flag) {
176  int found = 0;
177  for (int i = 1; i < argc; ++i) {
178  if (!argv[i] || argv[i] != flag)
179  continue;
180  found = i;
181  argv[i] = nullptr;
182  }
183  return found;
184  };
185 
186  bool threading = hasFlag("-t");
187 
188 #ifdef HAVE_OPENMP
189  const auto threadsMax = omp_get_max_threads();
190 #else
191  const auto threadsMax = 1;
192 #endif
193 
194  const auto threadsMin = threading ? 1 : threadsMax;
195 
196  // Were we told to use the repo (i.e. filelist.sha1 in that directory)?
197  int useChecksumFile = hasFlag("-r");
198  std::vector<rawspeed::ChecksumFileEntry> ChecksumFileEntries;
199  if (useChecksumFile && useChecksumFile + 1 < argc) {
200  char*& checksumFileRepo = argv[useChecksumFile + 1];
201  if (checksumFileRepo)
202  ChecksumFileEntries = rawspeed::ReadChecksumFile(checksumFileRepo);
203  checksumFileRepo = nullptr;
204  }
205 
206  // If there are normal filenames, append them.
207  for (int i = 1; i < argc; i++) {
208  if (!argv[i])
209  continue;
210 
211  rawspeed::ChecksumFileEntry Entry;
212  const char* fName = argv[i];
213  // These are supposed to be either absolute paths, or relative the run dir.
214  // We don't do any beautification.
215  Entry.FullFileName = fName;
216  Entry.RelFileName = fName;
217  ChecksumFileEntries.emplace_back(Entry);
218  }
219 
220  // And finally, actually add the raws to be benchmarked.
221  for (const auto& Entry : ChecksumFileEntries) {
222  const char* fName = Entry.RelFileName.c_str();
223  std::string tName(fName);
224  tName += "/threads:";
225 
226  for (auto threads = threadsMin; threads <= threadsMax; threads++)
227  addBench(Entry.FullFileName.c_str(), tName, threads);
228  }
229 
230  benchmark::RunSpecifiedBenchmarks();
231 }
