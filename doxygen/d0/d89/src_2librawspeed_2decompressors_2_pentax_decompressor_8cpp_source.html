70 HuffmanTable PentaxDecompressor::SetupHuffmanTable_Modern(ByteStream stream) {
71  HuffmanTable ht;
72 
73  const uint32_t depth = stream.getU16() + 12;
74  if (depth > 15)
75  ThrowRDE("Depth of huffman table is too great (%u).", depth);
76 
77  stream.skipBytes(12);
78 
79  std::array<uint32_t, 16> v0;
80  std::array<uint32_t, 16> v1;
81  for (uint32_t i = 0; i < depth; i++)
82  v0[i] = stream.getU16();
83  for (uint32_t i = 0; i < depth; i++) {
84  v1[i] = stream.getByte();
85 
86  if (v1[i] == 0 || v1[i] > 12)
87  ThrowRDE("Data corrupt: v1[%i]=%i, expected [1..12]", depth, v1[i]);
88  }
89 
90  std::vector<uint8_t> nCodesPerLength;
91  nCodesPerLength.resize(17);
92 
93  std::array<uint32_t, 16> v2;
94  /* Calculate codes and store bitcounts */
95  for (uint32_t c = 0; c < depth; c++) {
96  v2[c] = extractHighBits(v0[c], v1[c], /*effectiveBitwidth=*/12);
97  nCodesPerLength.at(v1[c])++;
98  }
99 
100  assert(nCodesPerLength.size() == 17);
101  assert(nCodesPerLength[0] == 0);
102  auto nCodes = ht.setNCodesPerLength(Buffer(&nCodesPerLength[1], 16));
103  assert(nCodes == depth);
104 
105  std::vector<uint8_t> codeValues;
106  codeValues.reserve(nCodes);
107 
108  /* Find smallest */
109  for (uint32_t i = 0; i < depth; i++) {
110  uint32_t sm_val = 0xfffffff;
111  uint32_t sm_num = 0xff;
112  for (uint32_t j = 0; j < depth; j++) {
113  if (v2[j] <= sm_val) {
114  sm_num = j;
115  sm_val = v2[j];
116  }
117  }
118  codeValues.push_back(sm_num);
119  v2[sm_num] = 0xffffffff;
120  }
121 
122  assert(codeValues.size() == nCodes);
123  ht.setCodeValues(Buffer(codeValues.data(), nCodes));
124 
125  return ht;
126 }

141 void PentaxDecompressor::decompress(const ByteStream& data) const {
142  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
143 
144  assert(out.height > 0);
145  assert(out.width > 0);
146  assert(out.width % 2 == 0);
147 
148  BitPumpMSB bs(data);
149  for (int row = 0; row < out.height; row++) {
150  std::array<int, 2> pred = {{}};
151  if (row >= 2)
152  pred = {out(row - 2, 0), out(row - 2, 1)};
153 
154  for (int col = 0; col < out.width; col++) {
155  pred[col & 1] += ht.decodeDifference(bs);
156  int value = pred[col & 1];
157  if (!isIntN(value, 16))
158  ThrowRDE("decoded value out of bounds at %d:%d", col, row);
159  out(row, col) = value;
160  }
161  }
162 }
