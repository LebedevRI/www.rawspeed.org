41 CrwDecompressor::CrwDecompressor(const RawImage& img, uint32_t dec_table,
42  bool lowbits_, ByteStream rawData)
43  : mRaw(img), lowbits(lowbits_) {
44  if (mRaw->getCpp() != 1 || mRaw->getDataType() != TYPE_USHORT16 ||
45  mRaw->getBpp() != sizeof(uint16_t))
46  ThrowRDE("Unexpected component count / data type");
47 
48  const uint32_t width = mRaw->dim.x;
49  const uint32_t height = mRaw->dim.y;
50 
51  if (width == 0 || height == 0 || width % 4 != 0 || width > 4104 ||
52  height > 3048 || (height * width) % 64 != 0)
53  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
54 
55  if (lowbits) {
56  // If there are low bits, the first part (size is calculatable) is low bits
57  // Each block is 4 pairs of 2 bits, so we have 1 block per 4 pixels
58  const unsigned lBlocks = 1 * height * width / 4;
59  assert(lBlocks > 0);
60  lowbitInput = rawData.getStream(lBlocks);
61  }
62 
63  // We always ignore next 514 bytes of 'padding'. No idea what is in there.
64  rawData.skipBytes(514);
65 
66  // Rest is the high bits.
67  rawInput = rawData.getStream(rawData.getRemainSize());
68 
69  mHuff = initHuffTables(dec_table);
70 }

164 inline void CrwDecompressor::decodeBlock(std::array<int16_t, 64>* diffBuf,
165  const crw_hts& mHuff,
166  BitPumpJPEG* bs) {
167  assert(diffBuf);
168  assert(bs);
169 
170  // decode the block
171  for (int i = 0; i < 64; i++) {
172  bs->fill(32);
173 
174  const uint8_t codeValue = mHuff[i > 0].decodeCodeValue(*bs);
175  const int len = codeValue & 0b1111;
176  const int index = codeValue >> 4;
177  assert(len >= 0 && index >= 0);
178 
179  if (len == 0 && index == 0 && i)
180  break;
181 
182  if (len == 0xf && index == 0xf)
183  continue;
184 
185  i += index;
186 
187  if (len == 0)
188  continue;
189 
190  int diff = bs->getBitsNoFill(len);
191 
192  if (i >= 64)
193  break;
194 
195  diff = HuffmanTable::extend(diff, len);
196 
197  (*diffBuf)[i] = diff;
198  }
199 }

202 void CrwDecompressor::decompress() {
203  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
204  assert(out.width > 0);
205  assert(out.width % 4 == 0);
206  assert(out.height > 0);
207 
208  {
209  // Each block encodes 64 pixels
210 
211  assert((out.height * out.width) % 64 == 0);
212  const unsigned hBlocks = out.height * out.width / 64;
213  assert(hBlocks > 0);
214 
215  BitPumpJPEG bs(rawInput);
216 
217  int carry = 0;
218  std::array<int, 2> base = {512, 512}; // starting predictors
219 
220  int row = 0;
221  int col = 0;
222 
223  for (unsigned block = 0; block < hBlocks; block++) {
224  array<int16_t, 64> diffBuf = {{}};
225  decodeBlock(&diffBuf, mHuff, &bs);
226 
227  // predict and output the block
228 
229  diffBuf[0] += carry;
230  carry = diffBuf[0];
231 
232  for (uint32_t k = 0; k < 64; ++k, ++col) {
233  if (col == out.width) {
234  // new line. sadly, does not always happen when k == 0.
235  col = 0;
236  row++;
237  base = {512, 512}; // reinit.
238  }
239 
240  base[k & 1] += diffBuf[k];
241 
242  if (!isIntN(base[k & 1], 10))
243  ThrowRDE("Error decompressing");
244 
245  out(row, col) = base[k & 1];
246  }
247  }
248  assert(row == (out.height - 1));
249  assert(col == out.width);
250  }
251 
252  // Add the uncompressed 2 low bits to the decoded 8 high bits
253  if (lowbits) {
254  for (int row = 0; row < out.height; row++) {
255  for (int col = 0; col < out.width; /* NOTE: col += 4 */) {
256  const uint8_t c = lowbitInput.getByte();
257  // LSB-packed: p3 << 6 | p2 << 4 | p1 << 2 | p0 << 0
258 
259  // We have read 8 bits, which is 4 pairs of 2 bits. So process 4 pixels.
260  for (uint32_t p = 0; p < 4; ++p, ++col) {
261  uint16_t& pixel = out(row, col);
262 
263  uint16_t low = (c >> (2 * p)) & 0b11;
264  uint16_t val = (pixel << 2) | low;
265 
266  if (out.width == 2672 && val < 512)
267  val += 2; // No idea why this is needed
268 
269  pixel = val;
270  }
271  }
272  }
273  }
