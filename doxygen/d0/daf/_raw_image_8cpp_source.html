69 void RawImageData::createData() {
70  static constexpr const auto alignment = 16;
71 
72  if (dim.x > 65535 || dim.y > 65535)
73  ThrowRDE("Dimensions too large for allocation.");
74  if (dim.x <= 0 || dim.y <= 0)
75  ThrowRDE("Dimension of one sides is less than 1 - cannot allocate image.");
76  if (data)
77  ThrowRDE("Duplicate data allocation in createData.");
78 
79  // want each line to start at 16-byte aligned address
80  pitch = roundUp(static_cast<size_t>(dim.x) * bpp, alignment);
81  assert(isAligned(pitch, alignment));
82 
83 #if defined(DEBUG) || __has_feature(address_sanitizer) ||
84  defined(__SANITIZE_ADDRESS__)
85  // want to ensure that we have some padding
86  pitch += alignment * alignment;
87  assert(isAligned(pitch, alignment));
88 #endif
89 
90  padding = pitch - dim.x * bpp;
91 
92 #if defined(DEBUG) || __has_feature(address_sanitizer) ||
93  defined(__SANITIZE_ADDRESS__)
94  assert(padding > 0);
95 #endif
96 
97  data = alignedMallocArray<uint8_t, alignment>(dim.y, pitch);
98 
99  if (!data)
100  ThrowRDE("Memory Allocation failed.");
101 
102  uncropped_dim = dim;
103 
104 #ifndef NDEBUG
105  if (dim.y > 1) {
106  // padding is the size of the area after last pixel of line n
107  // and before the first pixel of line n+1
108  assert(getData(dim.x - 1, 0) + bpp + padding == getData(0, 1));
109  }
110 
111  for (int j = 0; j < dim.y; j++) {
112  const uint8_t* const line = getData(0, j);
113  // each line is indeed 16-byte aligned
114  assert(isAligned(line, alignment));
115  }
116 #endif
117 
118  poisonPadding();
119 }

250 void RawImageData::subFrame(iRectangle2D crop) {
251  if (!crop.dim.isThisInside(dim - crop.pos)) {
252  writeLog(DEBUG_PRIO_WARNING, "WARNING: RawImageData::subFrame - Attempted "
253  "to create new subframe larger than original "
254  "size. Crop skipped.");
255  return;
256  }
257  if (crop.pos.x < 0 || crop.pos.y < 0 || !crop.hasPositiveArea()) {
258  writeLog(DEBUG_PRIO_WARNING, "WARNING: RawImageData::subFrame - Negative "
259  "crop offset. Crop skipped.");
260  return;
261  }
262 
263  // if CFA, and not X-Trans, adjust.
264  if (isCFA && cfa.getDcrawFilter() != 1 && cfa.getDcrawFilter() != 9) {
265  cfa.shiftLeft(crop.pos.x);
266  cfa.shiftDown(crop.pos.y);
267  }
268 
269  mOffset += crop.pos;
270  dim = crop.dim;
271 }

273 void RawImageData::createBadPixelMap()
274 {
275  if (!isAllocated())
276  ThrowRDE("(internal) Bad pixel map cannot be allocated before image.");
277  mBadPixelMapPitch = roundUp(roundUpDivision(uncropped_dim.x, 8), 16);
278  mBadPixelMap =
279  alignedMallocArray<uint8_t, 16>(uncropped_dim.y, mBadPixelMapPitch);
280  memset(mBadPixelMap, 0,

282  if (!mBadPixelMap)
283  ThrowRDE("Memory Allocation failed.");

310 void RawImageData::transferBadPixelsToMap()
311 {
312  MutexLocker guard(&mBadPixelMutex);
313  if (mBadPixelPositions.empty())
314  return;
315 
316  if (!mBadPixelMap)
317  createBadPixelMap();
318 
319  for (unsigned int pos : mBadPixelPositions) {
320  uint16_t pos_x = pos & 0xffff;
321  uint16_t pos_y = pos >> 16;
322 
323  assert(pos_x < static_cast<uint16_t>(uncropped_dim.x));
324  assert(pos_y < static_cast<uint16_t>(uncropped_dim.y));
325 
326  mBadPixelMap[mBadPixelMapPitch * pos_y + (pos_x >> 3)] |= 1 << (pos_x&7);
327  }
328  mBadPixelPositions.clear();
329 }

331 void RawImageData::fixBadPixels()
332 {
333 #if !defined (EMULATE_DCRAW_BAD_PIXELS)
334 
335  /* Transfer if not already done */
336  transferBadPixelsToMap();
337 
338 #if 0 // For testing purposes
339  if (!mBadPixelMap)
340  createBadPixelMap();
341  for (int y = 400; y < 700; y++){
342  for (int x = 1200; x < 1700; x++) {
343  mBadPixelMap[mBadPixelMapPitch * y + (x >> 3)] |= 1 << (x&7);
344  }
345  }
346 #endif
347 
348  /* Process bad pixels, if any */
349  if (mBadPixelMap)
350  startWorker(RawImageWorker::FIX_BAD_PIXELS, false);
351 
352 #else // EMULATE_DCRAW_BAD_PIXELS - not recommended, testing purposes only
353 
354  for (vector<uint32_t>::iterator i = mBadPixelPositions.begin();
355  i != mBadPixelPositions.end(); ++i) {
356  uint32_t pos = *i;
357  uint32_t pos_x = pos & 0xffff;
358  uint32_t pos_y = pos >> 16;
359  uint32_t total = 0;
360  uint32_t div = 0;
361  // 0 side covered by unsignedness.
362  for (uint32_t r = pos_x - 2;
363  r <= pos_x + 2 && r < (uint32_t)uncropped_dim.x; r += 2) {
364  for (uint32_t c = pos_y - 2;
365  c <= pos_y + 2 && c < (uint32_t)uncropped_dim.y; c += 2) {
366  uint16_t* pix = (uint16_t*)getDataUncropped(r, c);
367  if (*pix) {
368  total += *pix;
369  div++;
370  }
371  }
372  }
373  uint16_t* pix = (uint16_t*)getDataUncropped(pos_x, pos_y);
374  if (div) {
375  pix[0] = total / div;
376  }
377  }
378 #endif

382 void RawImageData::startWorker(const RawImageWorker::RawImageWorkerTask task,
383  bool cropped) {
384  const int height = [&]() {
385  int h = (cropped) ? dim.y : uncropped_dim.y;
386  if (task & RawImageWorker::FULL_IMAGE) {
387  h = uncropped_dim.y;
388  }
389  return h;
390  }();
391 
392  const int threads = rawspeed_get_number_of_processor_cores();
393  const int y_per_thread = (height + threads - 1) / threads;
394 
395 #ifdef HAVE_OPENMP
396 #pragma omp parallel for default(none)
397  OMPFIRSTPRIVATECLAUSE(threads, y_per_thread, height, task)
398  num_threads(threads) schedule(static)
399 #endif
400  for (int i = 0; i < threads; i++) {
401  int y_offset = std::min(i * y_per_thread, height);
402  int y_end = std::min((i + 1) * y_per_thread, height);
403 
404  RawImageWorker worker(this, task, y_offset, y_end);
405  }
406 }

408 void RawImageData::fixBadPixelsThread(int start_y, int end_y) {
409  int gw = (uncropped_dim.x + 15) / 32;
410 
411  for (int y = start_y; y < end_y; y++) {
412  const auto* bad_map =
413  reinterpret_cast<const uint32_t*>(&mBadPixelMap[y * mBadPixelMapPitch]);
414  for (int x = 0; x < gw; x++) {
415  // Test if there is a bad pixel within these 32 pixels
416  if (bad_map[x] == 0)
417  continue;
418  const auto* bad = reinterpret_cast<const uint8_t*>(&bad_map[x]);
419  // Go through each pixel
420  for (int i = 0; i < 4; i++) {
421  for (int j = 0; j < 8; j++) {
422  if (1 != ((bad[i] >> j) & 1))
423  continue;
424 
425  fixBadPixel(x * 32 + i * 8 + j, y, 0);
426  }
427  }
428  }
429  }
430 }

432 void RawImageData::blitFrom(const RawImage& src, const iPoint2D& srcPos,
433  const iPoint2D& size, const iPoint2D& destPos) {
434  iRectangle2D src_rect(srcPos, size);
435  iRectangle2D dest_rect(destPos, size);
436  src_rect = src_rect.getOverlap(iRectangle2D(iPoint2D(0,0), src->dim));
437  dest_rect = dest_rect.getOverlap(iRectangle2D(iPoint2D(0,0), dim));
438 
439  iPoint2D blitsize = src_rect.dim.getSmallest(dest_rect.dim);
440  if (blitsize.area() <= 0)
441  return;
442 
443  // TODO: Move offsets after crop.
444  copyPixels(getData(dest_rect.pos.x, dest_rect.pos.y), pitch,
445  src->getData(src_rect.pos.x, src_rect.pos.y), src->pitch,
446  blitsize.x * bpp, blitsize.y);
447 }

449 /* Does not take cfa into consideration */
450 void RawImageData::expandBorder(iRectangle2D validData)
451 {
452  validData = validData.getOverlap(iRectangle2D(0,0,dim.x, dim.y));
453  if (validData.pos.x > 0) {
454  for (int y = 0; y < dim.y; y++ ) {
455  uint8_t* src_pos = getData(validData.pos.x, y);
456  uint8_t* dst_pos = getData(validData.pos.x - 1, y);
457  for (int x = validData.pos.x; x >= 0; x--) {
458  for (int i = 0; i < bpp; i++) {
459  dst_pos[i] = src_pos[i];
460  }
461  dst_pos -= bpp;
462  }
463  }
464  }
465 
466  if (validData.getRight() < dim.x) {
467  int pos = validData.getRight();
468  for (int y = 0; y < dim.y; y++ ) {
469  uint8_t* src_pos = getData(pos - 1, y);
470  uint8_t* dst_pos = getData(pos, y);
471  for (int x = pos; x < dim.x; x++) {
472  for (int i = 0; i < bpp; i++) {
473  dst_pos[i] = src_pos[i];
474  }
475  dst_pos += bpp;
476  }
477  }
478  }
479 
480  if (validData.pos.y > 0) {
481  uint8_t* src_pos = getData(0, validData.pos.y);
482  for (int y = 0; y < validData.pos.y; y++ ) {
483  uint8_t* dst_pos = getData(0, y);
484  memcpy(dst_pos, src_pos, static_cast<size_t>(dim.x) * bpp);
485  }
486  }
487  if (validData.getBottom() < dim.y) {
488  uint8_t* src_pos = getData(0, validData.getBottom() - 1);
489  for (int y = validData.getBottom(); y < dim.y; y++ ) {
490  uint8_t* dst_pos = getData(0, y);
491  memcpy(dst_pos, src_pos, static_cast<size_t>(dim.x) * bpp);
492  }
493  }
494 }

532 void RawImageWorker::performTask() noexcept {
533  try {
534  switch(task)
535  {
536  case SCALE_VALUES:
537  data->scaleValues(start_y, end_y);
538  break;
539  case FIX_BAD_PIXELS:
540  data->fixBadPixelsThread(start_y, end_y);
541  break;
542  case APPLY_LOOKUP:
543  data->doLookup(start_y, end_y);
544  break;
545  default:
546  assert(false);
547  }
548  } catch (RawDecoderException &e) {
549  data->setError(e.what());
550  } catch (TiffParserException &e) {
551  data->setError(e.what());
552  } catch (IOException &e) {
553  data->setError(e.what());
554  }
