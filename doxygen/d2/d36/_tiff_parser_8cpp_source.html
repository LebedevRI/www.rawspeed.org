65 TiffRootIFDOwner TiffParser::parse(TiffIFD* parent, const Buffer& data) {
66  ByteStream bs(DataBuffer(data, Endianness::unknown));
67  bs.setByteOrder(getTiffByteOrder(bs, 0, "TIFF header"));
68  bs.skipBytes(2);
69 
70  uint16_t magic = bs.getU16();
71  if (magic != 42 && magic != 0x4f52 && magic != 0x5352 && magic != 0x55) // ORF has 0x4f52/0x5352, RW2 0x55 - Brilliant!
72  ThrowTPE("Not a TIFF file (magic 42)");
73 
74  TiffRootIFDOwner root = std::make_unique<TiffRootIFD>(
75  parent, nullptr, bs,
76  UINT32_MAX); // tell TiffIFD constructor not to parse bs as IFD
77 
78  NORangesSet<Buffer> ifds;
79 
80  for (uint32_t IFDOffset = bs.getU32(); IFDOffset;
81  IFDOffset = root->getSubIFDs().back()->getNextIFD()) {
82  root->add(std::make_unique<TiffIFD>(root.get(), &ifds, bs, IFDOffset));
83  }
84 
85  return root;
86 }

88 std::unique_ptr<RawDecoder> TiffParser::makeDecoder(TiffRootIFDOwner root,
89  const Buffer& data) {
90  const Buffer* mInput = &data;
91  if (!root)
92  ThrowTPE("TiffIFD is null.");
93 
94  for (const auto& decoder : Map) {
95  checker_t dChecker = nullptr;
96  constructor_t dConstructor = nullptr;
97 
98  std::tie(dChecker, dConstructor) = decoder;
99 
100  assert(dChecker);
101  assert(dConstructor);
102 
103  if (!dChecker(root.get(), mInput))
104  continue;
105 
106  return dConstructor(move(root), mInput);
107  }
108 
109  ThrowTPE("No decoder found. Sorry.");
110 }
