37 CiffEntry::CiffEntry(NORangesSet<Buffer>* valueDatas,
38  const ByteStream* valueData, ByteStream dirEntry) {
39  uint16_t p = dirEntry.getU16();
40 
41  // NOLINTNEXTLINE cppcoreguidelines-prefer-member-initializer
42  tag = static_cast<CiffTag>(p & 0x3fff);
43  uint16_t datalocation = (p & 0xc000);
44  // NOLINTNEXTLINE cppcoreguidelines-prefer-member-initializer
45  type = static_cast<CiffDataType>(p & 0x3800);
46 
47  uint32_t data_offset;
48  uint32_t bytesize;
49 
50  switch (datalocation) {
51  case 0x0000:
52  // Data is offset in value_data
53  bytesize = dirEntry.getU32();
54  data_offset = dirEntry.getU32();
55  data = valueData->getSubStream(data_offset, bytesize);
56  if (!valueDatas->emplace(data).second)
57  ThrowCPE("Two valueData's overlap. Raw corrupt!");
58  break;
59  case 0x4000:
60  // Data is stored directly in entry
61  data_offset = dirEntry.getPosition();
62  // Maximum of 8 bytes of data (the size and offset fields)
63  bytesize = 8;
64  data = dirEntry.getStream(bytesize);
65  break;
66  default:
67  ThrowCPE("Don't understand data location 0x%x", datalocation);
68  }
69 
70  // Set the number of items using the shift
71  count = bytesize >> getElementShift();
72 }

110 uint32_t CiffEntry::getU32(uint32_t num) const {
111  if (!isInt()) {
112  ThrowCPE(

114  type, tag);

149 vector<string> CiffEntry::getStrings() const {
150  if (type != CIFF_ASCII)
151  ThrowCPE("Wrong type 0x%x encountered. Expected Ascii", type);
152 
153  const string str(reinterpret_cast<const char*>(data.peekData(count)), count);
154 
155  vector<string> strs;
156 
157  uint32_t start = 0;
158  for (uint32_t i = 0; i < count; i++) {
159  if (str[i] != 0)
160  continue;
161 
162  strs.emplace_back(reinterpret_cast<const char*>(&str[start]));
163  start = i + 1;
164  }
165 
166  return strs;
167 }
