42 void UncompressedDecompressor::sanityCheck(const uint32_t* h,
43  int bytesPerLine) {
44  assert(h != nullptr);
45  assert(*h > 0);
46  assert(bytesPerLine > 0);
47  assert(input.getSize() > 0);
48 
49  // How many multiples of bpl are there in the input buffer?
50  // The remainder is ignored/discarded.
51  const auto fullRows = input.getRemainSize() / bytesPerLine;
52 
53  // If more than the output height, we are all good.
54  if (fullRows >= *h)
55  return; // all good!
56 
57  if (fullRows == 0)
58  ThrowIOE("Not enough data to decode a single line. Image file truncated.");
59 
60  ThrowIOE("Image truncated, only %u of %u lines found", fullRows, *h);
61 
62  // FIXME: need to come up with some common variable to allow proceeding here
63  // *h = min_h;
64 }

78 int UncompressedDecompressor::bytesPerLine(int w, bool skips) {
79  assert(w > 0);
80 
81  if ((12 * w) % 8 != 0)
82  ThrowIOE("Bad image width");
83 
84  // Calculate expected bytes per line.
85  auto perline = (12 * w) / 8;
86 
87  if (!skips)
88  return perline;
89 
90  // Add skips every 10 pixels
91  perline += ((w + 2) / 10);
92 
93  return perline;
94 }

96 void UncompressedDecompressor::readUncompressedRaw(const iPoint2D& size,
97  const iPoint2D& offset,
98  int inputPitchBytes,
99  int bitPerPixel,
100  BitOrder order) {
101  assert(inputPitchBytes > 0);
102  assert(bitPerPixel > 0);
103 
104  uint8_t* data = mRaw->getData();
105  uint32_t outPitch = mRaw->pitch;
106  uint32_t w = size.x;
107  uint32_t h = size.y;
108  uint32_t cpp = mRaw->getCpp();
109  uint64_t ox = offset.x;
110  uint64_t oy = offset.y;
111 
112  if (bitPerPixel > 16 && mRaw->getDataType() == TYPE_USHORT16)
113  ThrowRDE("Unsupported bit depth");
114 
115  const int outPixelBits = w * cpp * bitPerPixel;
116  assert(outPixelBits > 0);
117 
118  if (outPixelBits % 8 != 0) {
119  ThrowRDE("Bad combination of cpp (%u), bps (%u) and width (%u), the "
120  "pitch is %u bits, which is not a multiple of 8 (1 byte)",
121  cpp, bitPerPixel, w, outPixelBits);
122  }
123 
124  const int outPixelBytes = outPixelBits / 8;
125 
126  // The input pitch might be larger than needed (i.e. have some padding),
127  // but it can *not* be smaller than needed.
128  if (inputPitchBytes < outPixelBytes)
129  ThrowRDE("Specified pitch is smaller than minimally-required pitch");
130 
131  // Check the specified pitch, not the minimally-required pitch.
132  sanityCheck(&h, inputPitchBytes);
133 
134  assert(inputPitchBytes >= outPixelBytes);
135  uint32_t skipBytes = inputPitchBytes - outPixelBytes; // Skip per line
136 
137  if (oy > static_cast<uint64_t>(mRaw->dim.y))
138  ThrowRDE("Invalid y offset");
139  if (ox + size.x > static_cast<uint64_t>(mRaw->dim.x))
140  ThrowRDE("Invalid x offset");
141 
142  uint64_t y = oy;
143  h = min(h + oy, static_cast<uint64_t>(mRaw->dim.y));
144 
145  if (mRaw->getDataType() == TYPE_FLOAT32) {
146  if (bitPerPixel == 32) {
147  copyPixels(&data[offset.x * sizeof(float) * cpp + y * outPitch], outPitch,
148  input.getData(inputPitchBytes * (h - y)), inputPitchBytes,
149  w * mRaw->getBpp(), h - y);
150  return;
151  }
152  if (BitOrder_MSB == order && bitPerPixel == 16) {
153  BitPumpMSB bits(input);
154  w *= cpp;
155  for (; y < h; y++) {
156  auto* dest = reinterpret_cast<uint32_t*>(
157  &data[offset.x * sizeof(uint32_t) * cpp + y * outPitch]);
158  for (uint32_t x = 0; x < w; x++) {
159  uint16_t b = bits.getBits(bitPerPixel);
160  dest[x] = fp16ToFloat(b);
161  }
162  bits.skipBytes(skipBytes);
163  }
164  return;
165  }
166  if (BitOrder_LSB == order && bitPerPixel == 16) {
167  BitPumpLSB bits(input);
168  w *= cpp;
169  for (; y < h; y++) {
170  auto* dest = reinterpret_cast<uint32_t*>(
171  &data[offset.x * sizeof(uint32_t) * cpp + y * outPitch]);
172  for (uint32_t x = 0; x < w; x++) {
173  uint16_t b = bits.getBits(bitPerPixel);
174  dest[x] = fp16ToFloat(b);
175  }
176  bits.skipBytes(skipBytes);
177  }
178  return;
179  }
180  if (BitOrder_MSB == order && bitPerPixel == 24) {
181  BitPumpMSB bits(input);
182  w *= cpp;
183  for (; y < h; y++) {
184  auto* dest = reinterpret_cast<uint32_t*>(
185  &data[offset.x * sizeof(uint32_t) * cpp + y * outPitch]);
186  for (uint32_t x = 0; x < w; x++) {
187  uint32_t b = bits.getBits(bitPerPixel);
188  dest[x] = fp24ToFloat(b);
189  }
190  bits.skipBytes(skipBytes);
191  }
192  return;
193  }
194  if (BitOrder_LSB == order && bitPerPixel == 24) {
195  BitPumpLSB bits(input);
196  w *= cpp;
197  for (; y < h; y++) {
198  auto* dest = reinterpret_cast<uint32_t*>(
199  &data[offset.x * sizeof(uint32_t) * cpp + y * outPitch]);
200  for (uint32_t x = 0; x < w; x++) {
201  uint32_t b = bits.getBits(bitPerPixel);
202  dest[x] = fp24ToFloat(b);
203  }
204  bits.skipBytes(skipBytes);
205  }
206  return;
207  }
208  ThrowRDE("Unsupported floating-point input bitwidth/bit packing: %u / %u",
209  bitPerPixel, order);
210  }
211 
212  if (BitOrder_MSB == order) {
213  BitPumpMSB bits(input);
214  w *= cpp;
215  for (; y < h; y++) {
216  auto* dest = reinterpret_cast<uint16_t*>(
217  &data[offset.x * sizeof(uint16_t) * cpp + y * outPitch]);
218  for (uint32_t x = 0; x < w; x++) {
219  uint32_t b = bits.getBits(bitPerPixel);
220  dest[x] = b;
221  }
222  bits.skipBytes(skipBytes);
223  }
224  } else if (BitOrder_MSB16 == order) {
225  BitPumpMSB16 bits(input);
226  w *= cpp;
227  for (; y < h; y++) {
228  auto* dest = reinterpret_cast<uint16_t*>(
229  &data[offset.x * sizeof(uint16_t) * cpp + y * outPitch]);
230  for (uint32_t x = 0; x < w; x++) {
231  uint32_t b = bits.getBits(bitPerPixel);
232  dest[x] = b;
233  }
234  bits.skipBytes(skipBytes);
235  }
236  } else if (BitOrder_MSB32 == order) {
237  BitPumpMSB32 bits(input);
238  w *= cpp;
239  for (; y < h; y++) {
240  auto* dest = reinterpret_cast<uint16_t*>(
241  &data[offset.x * sizeof(uint16_t) * cpp + y * outPitch]);
242  for (uint32_t x = 0; x < w; x++) {
243  uint32_t b = bits.getBits(bitPerPixel);
244  dest[x] = b;
245  }
246  bits.skipBytes(skipBytes);
247  }
248  } else {
249  if (bitPerPixel == 16 && getHostEndianness() == Endianness::little) {
250  copyPixels(&data[offset.x * sizeof(uint16_t) * cpp + y * outPitch],
251  outPitch, input.getData(inputPitchBytes * (h - y)),
252  inputPitchBytes, w * mRaw->getBpp(), h - y);
253  return;
254  }
255  if (bitPerPixel == 12 && static_cast<int>(w) == inputPitchBytes * 8 / 12 &&
256  getHostEndianness() == Endianness::little) {
257  decode12BitRaw<Endianness::little>(w, h);
258  return;
259  }
260  BitPumpLSB bits(input);
261  w *= cpp;
262  for (; y < h; y++) {
263  auto* dest = reinterpret_cast<uint16_t*>(
264  &data[offset.x * sizeof(uint16_t) + y * outPitch]);
265  for (uint32_t x = 0; x < w; x++) {
266  uint32_t b = bits.getBits(bitPerPixel);
267  dest[x] = b;
268  }
269  bits.skipBytes(skipBytes);
270  }
271  }

276  sanityCheck(w, &h, 1);
277 
278  uint8_t* data = mRaw->getData();
279  uint32_t pitch = mRaw->pitch;
280  const uint8_t* in = input.getData(w * h);
281  uint32_t random = 0;
282  for (uint32_t y = 0; y < h; y++) {
283  auto* dest = reinterpret_cast<uint16_t*>(&data[y * pitch]);
284  for (uint32_t x = 0; x < w; x++) {
285  if (uncorrectedRawValues)
286  dest[x] = *in;
287  else
288  mRaw->setWithLookUp(*in, reinterpret_cast<uint8_t*>(&dest[x]), &random);
289  in++;
290  }
291  }

299 template <Endianness e, bool interlaced, bool skips>
300 void UncompressedDecompressor::decode12BitRaw(uint32_t w, uint32_t h) {
301  static constexpr const auto bits = 12;
302 
303  static_assert(e == Endianness::little || e == Endianness::big,
304  "unknown endiannes");
305 
306  static constexpr const auto shift = 16 - bits;
307  static constexpr const auto pack = 8 - shift;
308  static constexpr const auto mask = (1 << pack) - 1;
309 
310  static_assert(bits == 12 && pack == 4, "wrong pack");
311 
312  static_assert(bits == 12 && mask == 0x0f, "wrong mask");
313 
314  uint32_t perline = bytesPerLine(w, skips);
315 
316  sanityCheck(&h, perline);
317 
318  uint8_t* data = mRaw->getData();
319  uint32_t pitch = mRaw->pitch;
320 
321  // FIXME: maybe check size of interlaced data?
322  const uint8_t* in = input.peekData(perline * h);
323  uint32_t half = (h + 1) >> 1;
324  for (uint32_t row = 0; row < h; row++) {
325  uint32_t y = !interlaced ? row : row % half * 2 + row / half;
326  auto* dest = reinterpret_cast<uint16_t*>(&data[y * pitch]);
327 
328  if (interlaced && y == 1) {
329  // The second field starts at a 2048 byte alignment
330  const uint32_t offset = ((half * w * 3 / 2 >> 11) + 1) << 11;
331  input.skipBytes(offset);
332  in = input.peekData(perline * (h - row));
333  }
334 
335  for (uint32_t x = 0; x < w; x += 2, in += 3) {
336  uint32_t g1 = in[0];
337  uint32_t g2 = in[1];
338 
339  auto process = [dest](uint32_t i, bool invert, uint32_t p1, uint32_t p2) {
340  if (!(invert ^ (e == Endianness::little)))
341  dest[i] = (p1 << pack) | (p2 >> pack);
342  else
343  dest[i] = ((p2 & mask) << 8) | p1;
344  };
345 
346  process(x, false, g1, g2);
347 
348  g1 = in[2];
349 
350  process(x + 1, true, g1, g2);
351 
352  if (skips && ((x % 10) == 8))
353  in++;
354  }
355  }
356  input.skipBytes(input.getRemainSize());
357 }

402 template <int bits, Endianness e>
403 void UncompressedDecompressor::decodeRawUnpacked(uint32_t w, uint32_t h) {
404  static_assert(bits == 12 || bits == 14 || bits == 16, "unhandled bitdepth");
405  static_assert(e == Endianness::little || e == Endianness::big,
406  "unknown endiannes");
407 
408  static constexpr const auto shift = 16 - bits;
409  static constexpr const auto mask = (1 << (8 - shift)) - 1;
410 
411  static_assert((bits == 12 && mask == 0x0f) || bits != 12, "wrong mask");
412  static_assert((bits == 14 && mask == 0x3f) || bits != 14, "wrong mask");
413  static_assert((bits == 16 && mask == 0xff) || bits != 16, "wrong mask");
414 
415  sanityCheck(w, &h, 2);
416 
417  uint8_t* data = mRaw->getData();
418  uint32_t pitch = mRaw->pitch;
419  const uint8_t* in = input.getData(w * h * 2);
420 
421  for (uint32_t y = 0; y < h; y++) {
422  auto* dest = reinterpret_cast<uint16_t*>(&data[y * pitch]);
423  for (uint32_t x = 0; x < w; x += 1, in += 2) {
424  uint32_t g1 = in[0];
425  uint32_t g2 = in[1];
426 
427  if (e == Endianness::little)
428  dest[x] = ((g2 << 8) | g1) >> shift;
429  else
430  dest[x] = ((g1 & mask) << 8) | g2;
431  }
432  }
433 }
