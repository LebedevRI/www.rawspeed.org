239 template <typename T> T lerp(T v0, T v1, T t) {
240  return (1.0 - t) * v0 + t * v1;
241 }
242 
243 template <typename T = int,
244  typename = std::enable_if_t<std::is_arithmetic<T>::value>>
245 std::vector<T> calculateSteps(int numCp) {
246  std::vector<T> steps;
247 
248  const auto ptsTotal = 2U + numCp;
249  steps.reserve(ptsTotal);
250 
251  std::generate_n(std::back_inserter(steps), ptsTotal,
252  [ptsTotal, &steps]() -> T {
253  const double t = double(steps.size()) / (ptsTotal - 1);
254  const double x = lerp(0.0, 65535.0, t);
255  if (std::is_floating_point<T>::value)
256  return x;
257  return std::lround(x);
258  });
259 
260  assert(ptsTotal == steps.size());
261  return steps;
262 }

265  const auto pts = calculateSteps(steps);

267  ASSERT_EQ(pts[x], x);
