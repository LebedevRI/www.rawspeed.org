37 // Normally, we'd just use std::signbit(int) here. But, some (non-conforming?)
38 // compilers do not provide that overload, so the code simply fails to compile.
39 // One could cast the int to the double, but at least right now that results
40 // in a horrible code. So let's just provide our own signbit(). It compiles to
41 // the exact same code as the std::signbit(int).
42 template <typename T, typename = std::enable_if_t<std::is_integral<T>::value>>
43 inline constexpr __attribute__((const)) bool SignBit(T x) {
44  return x < 0;
45 }

70 inline __attribute__((always_inline)) int
71 OlympusDecompressor::parseCarry(BitPumpMSB* bits,
72  std::array<int, 3>* carry) const {
73  bits->fill();
74  int i = 2 * ((*carry)[2] < 3);
75  int nbits;
76  for (nbits = 2 + i; static_cast<uint16_t>((*carry)[0]) >> (nbits + i);
77  nbits++)
78  ;
79 
80  int b = bits->peekBitsNoFill(15);
81  int sign = (b >> 14) * -1;
82  int low = (b >> 12) & 3;
83  int high = bittable[b & 4095];
84 
85  // Skip bytes used above or read bits
86  if (high == 12) {
87  bits->skipBitsNoFill(15);
88  high = bits->getBitsNoFill(16 - nbits) >> 1;
89  } else
90  bits->skipBitsNoFill(high + 1 + 3);
91 
92  (*carry)[0] = (high << nbits) | bits->getBitsNoFill(nbits);
93  int diff = ((*carry)[0] ^ sign) + (*carry)[1];
94  (*carry)[1] = (diff * 3 + (*carry)[1]) >> 5;
95  (*carry)[2] = (*carry)[0] > 16 ? 0 : (*carry)[2] + 1;
96 
97  return (diff * 4) | low;
98 }

100 inline int OlympusDecompressor::getPred(const Array2DRef<uint16_t> out, int row,
101  int col) {
102  auto getLeft = [&]() { return out(row, col - 2); };
103  auto getUp = [&]() { return out(row - 2, col); };
104  auto getLeftUp = [&]() { return out(row - 2, col - 2); };
105 
106  int pred;
107  if (row < 2 && col < 2)
108  pred = 0;
109  else if (row < 2)
110  pred = getLeft();
111  else if (col < 2)
112  pred = getUp();
113  else {
114  int left = getLeft();
115  int up = getUp();
116  int leftUp = getLeftUp();
117 
118  int leftMinusNw = left - leftUp;
119  int upMinusNw = up - leftUp;
120 
121  // Check if sign is different, and they are both not zero
122  if ((SignBit(leftMinusNw) ^ SignBit(upMinusNw)) &&
123  (leftMinusNw != 0 && upMinusNw != 0)) {
124  if (std::abs(leftMinusNw) > 32 || std::abs(upMinusNw) > 32)
125  pred = left + upMinusNw;
126  else
127  pred = (left + up) >> 1;
128  } else
129  pred = std::abs(leftMinusNw) > std::abs(upMinusNw) ? left : up;
130  }
131 
132  return pred;
133 }

135 void OlympusDecompressor::decompressRow(BitPumpMSB* bits, int row) const {
136  assert(mRaw->dim.y > 0);
137  assert(mRaw->dim.x > 0);
138  assert(mRaw->dim.x % 2 == 0);
139 
140  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
141 
142  std::array<std::array<int, 3>, 2> acarry{{}};
143 
144  for (int col = 0; col < out.width; col++) {
145  int c = col & 1;
146 
147  std::array<int, 3>& carry = acarry[c];
148 
149  int diff = parseCarry(bits, &carry);
150  int pred = getPred(out, row, col);
151 
152  out(row, col) = pred + diff;
153  }
154 }

156 void OlympusDecompressor::decompress(ByteStream input) const {
157  assert(mRaw->dim.y > 0);
158  assert(mRaw->dim.x > 0);
159  assert(mRaw->dim.x % 2 == 0);
160 
161  input.skipBytes(7);
162  BitPumpMSB bits(input);
163 
164  for (int y = 0; y < mRaw->dim.y; y++)
165  decompressRow(&bits, y);
166 }
