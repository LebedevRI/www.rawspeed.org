113 void JpegDecompressor::decode(uint32_t offX,
114  uint32_t offY) { /* Each slice is a JPEG image */
115  struct JpegDecompressStruct dinfo;
116 
117  const auto size = input.getRemainSize();
118 
119  JPEG_MEMSRC(&dinfo, input.getData(size), size);
120 
121  if (JPEG_HEADER_OK != jpeg_read_header(&dinfo, static_cast<boolean>(true)))
122  ThrowRDE("Unable to read JPEG header");
123 
124  jpeg_start_decompress(&dinfo);
125  if (dinfo.output_components != static_cast<int>(mRaw->getCpp()))
126  ThrowRDE("Component count doesn't match");
127  int row_stride = dinfo.output_width * dinfo.output_components;
128 
129  unique_ptr<uint8_t[], // NOLINT
130  decltype(&alignedFree)>
131  complete_buffer(
132  alignedMallocArray<uint8_t, 16>(dinfo.output_height, row_stride),
133  &alignedFree);
134 
135  const Array2DRef<uint8_t> tmp(&complete_buffer[0],
136  dinfo.output_components * dinfo.output_width,
137  dinfo.output_height, row_stride);
138 
139  while (dinfo.output_scanline < dinfo.output_height) {
140  auto* rowOut = static_cast<JSAMPROW>(&tmp(dinfo.output_scanline, 0));
141  if (0 == jpeg_read_scanlines(&dinfo, &rowOut, 1))
142  ThrowRDE("JPEG Error while decompressing image.");
143  }
144  jpeg_finish_decompress(&dinfo);
145 
146  // Now the image is decoded, and we copy the image data
147  int copy_w = min(mRaw->dim.x - offX, dinfo.output_width);
148  int copy_h = min(mRaw->dim.y - offY, dinfo.output_height);
149 
150  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
151  for (int row = 0; row < copy_h; row++) {
152  for (int col = 0; col < dinfo.output_components * copy_w; col++)
153  out(row + offY, dinfo.output_components * offX + col) = tmp(row, col);
154  }
155 }
