42 FujiDecompressor::FujiDecompressor(const RawImage& img, ByteStream input_)
43  : mRaw(img), input(std::move(input_)) {
44  if (mRaw->getCpp() != 1 || mRaw->getDataType() != TYPE_USHORT16 ||
45  mRaw->getBpp() != sizeof(uint16_t))
46  ThrowRDE("Unexpected component count / data type");
47 
48  input.setByteOrder(Endianness::big);
49 
50  header = FujiHeader(&input);
51  if (!header)
52  ThrowRDE("compressed RAF header check");
53 
54  if (mRaw->dim != iPoint2D(header.raw_width, header.raw_height))
55  ThrowRDE("RAF header specifies different dimensions!");
56 
57  if (12 == header.raw_bits) {
58  ThrowRDE("Aha, finally, a 12-bit compressed RAF! Please consider providing "
59  "samples on <https://raw.pixls.us/>, thanks!");
60  }
61 
62  for (int i = 0; i < 6; i++) {
63  for (int j = 0; j < 6; j++) {
64  const CFAColor c = mRaw->cfa.getColorAt(j, i);
65  switch (c) {
66  case CFA_RED:
67  case CFA_GREEN:
68  case CFA_BLUE:
69  CFA[i][j] = c;
70  break;
71  default:
72  ThrowRDE("Got unexpected color %u", c);
73  }
74  }
75  }
76 
77  fuji_compressed_load_raw();
78 }

80 FujiDecompressor::fuji_compressed_params::fuji_compressed_params(
81  const FujiDecompressor& d) {
82  int cur_val;
83  char* qt;
84 
85  if ((d.header.block_size % 3 && d.header.raw_type == 16) ||
86  (d.header.block_size & 1 && d.header.raw_type == 0)) {
87  ThrowRDE("fuji_block_checks");
88  }
89 
90 
91  if (d.header.raw_type == 16) {
92  line_width = (d.header.block_size * 2) / 3;
93  } else {
94  line_width = d.header.block_size >> 1;
95  }
96 
97  q_point[0] = 0;
98  q_point[1] = 0x12;
99  q_point[2] = 0x43;
100  q_point[3] = 0x114;
101  q_point[4] = (1 << d.header.raw_bits) - 1;
102  min_value = 0x40;
103 
104  cur_val = -q_point[4];
105  q_table.resize(2 * (1 << d.header.raw_bits));
106 
107  for (qt = &q_table[0]; cur_val <= q_point[4]; ++qt, ++cur_val) {
108  if (cur_val <= -q_point[3]) {
109  *qt = -4;
110  } else if (cur_val <= -q_point[2]) {
111  *qt = -3;
112  } else if (cur_val <= -q_point[1]) {
113  *qt = -2;
114  } else if (cur_val < 0) {
115  *qt = -1;
116  } else if (cur_val == 0) {
117  *qt = 0;
118  } else if (cur_val < q_point[1]) {
119  *qt = 1;
120  } else if (cur_val < q_point[2]) {
121  *qt = 2;
122  } else if (cur_val < q_point[3]) {
123  *qt = 3;
124  } else {
125  *qt = 4;
126  }
127  }
128 
129  // populting gradients
130  if (q_point[4] == 0xFFFF) { // (1 << d.header.raw_bits) - 1
131  total_values = 0x10000; // 1 << d.header.raw_bits
132  raw_bits = 16; // d.header.raw_bits
133  max_bits = 64; // d.header.raw_bits * (64 / d.header.raw_bits)
134  maxDiff = 1024; // 1 << (d.header.raw_bits - 6)
135  } else if (q_point[4] == 0x3FFF) {
136  total_values = 0x4000;
137  raw_bits = 14;
138  max_bits = 56;
139  maxDiff = 256;
140  } else if (q_point[4] == 0xFFF) {
141  ThrowRDE("Aha, finally, a 12-bit compressed RAF! Please consider providing "
142  "samples on <https://raw.pixls.us/>, thanks!");
143 
144  /* kept for future, once there is a sample.
145  total_values = 4096;
146  raw_bits = 12;
147  max_bits = 48; // out-of-family, there's greater pattern at play.
148  maxDiff = 64;
149  */
150  } else {
151  ThrowRDE("FUJI q_point");
152  }
153 }

180 template <typename T>
181 void FujiDecompressor::copy_line(fuji_compressed_block* info,
182  const FujiStrip& strip, int cur_line,
183  T&& idx) const {
184  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
185 
186  std::array<uint16_t*, 3> lineBufB;
187  std::array<uint16_t*, 6> lineBufG;
188  std::array<uint16_t*, 3> lineBufR;
189 
190  for (int i = 0; i < 3; i++) {
191  lineBufR[i] = info->linebuf[R2 + i] + 1;
192  lineBufB[i] = info->linebuf[B2 + i] + 1;
193  }
194 
195  for (int i = 0; i < 6; i++) {
196  lineBufG[i] = info->linebuf[G2 + i] + 1;
197  }
198 
199  for (int row_count = 0; row_count < FujiStrip::lineHeight(); row_count++) {
200  for (int pixel_count = 0; pixel_count < strip.width(); pixel_count++) {
201  uint16_t* line_buf = nullptr;
202 
203  switch (CFA[row_count][pixel_count % 6]) {
204  case CFA_RED: // red
205  line_buf = lineBufR[row_count >> 1];
206  break;
207 
208  case CFA_GREEN: // green
209  line_buf = lineBufG[row_count];
210  break;
211 
212  case CFA_BLUE: // blue
213  line_buf = lineBufB[row_count >> 1];
214  break;
215 
216  default:
217  __builtin_unreachable();
218  }
219 
220  out(strip.offsetY(cur_line) + row_count, strip.offsetX() + pixel_count) =
221  line_buf[idx(pixel_count)];
222  }
223  }

245 inline void FujiDecompressor::fuji_zerobits(BitPumpMSB* pump, int* count) {
246  *count = 0;
247 
248  // Count-and-skip all the leading `0`s.
249  while (true) {
250  uint32_t batch = (pump->peekBits(31) << 1) | 0b1;
251  int numZerosInThisBatch = __builtin_clz(batch);
252  *count += numZerosInThisBatch;
253  bool allZeroes = numZerosInThisBatch == 31;
254  int numBitsToSkip = numZerosInThisBatch;
255  if (!allZeroes)
256  numBitsToSkip += 1; // Also skip the first `1`.
257  pump->skipBitsNoFill(numBitsToSkip);
258  if (!allZeroes)
259  break; // We're done!
260  }
261 }

263 int __attribute__((const)) FujiDecompressor::bitDiff(int value1, int value2) {
264  int decBits = 0;
265 
266  if (value2 >= value1)
267  return decBits;
268 
269  while (decBits <= 12) {
270  ++decBits;
271 
272  if ((value2 << decBits) >= value1)
273  return decBits;
274  }
275 
276  return decBits;
277 }

279 template <typename T1, typename T2>
280 void FujiDecompressor::fuji_decode_sample(
281  T1&& func_0, T2&& func_1, fuji_compressed_block* info, uint16_t* line_buf,
282  int* pos, std::array<int_pair, 41>* grads) const {
283  int interp_val = 0;
284 
285  int sample = 0;
286  int code = 0;
287  uint16_t* line_buf_cur = line_buf + *pos;
288 
289  int grad;
290  int gradient;
291 
292  func_0(line_buf_cur, &interp_val, &grad, &gradient);
293 
294  fuji_zerobits(&(info->pump), &sample);
295 
296  if (sample < common_info.max_bits - common_info.raw_bits - 1) {
297  int decBits = bitDiff((*grads)[gradient].value1, (*grads)[gradient].value2);
298  code = 0;
299  if (decBits)
300  code = info->pump.getBits(decBits);
301  code += sample << decBits;
302  } else {
303  code = info->pump.getBits(common_info.raw_bits);
304  code++;
305  }
306 
307  if (code < 0 || code >= common_info.total_values) {
308  ThrowRDE("fuji_decode_sample");
309  }
310 
311  if (code & 1) {
312  code = -1 - code / 2;
313  } else {
314  code /= 2;
315  }
316 
317  (*grads)[gradient].value1 += std::abs(code);
318 
319  if ((*grads)[gradient].value2 == common_info.min_value) {
320  (*grads)[gradient].value1 >>= 1;
321  (*grads)[gradient].value2 >>= 1;
322  }
323 
324  (*grads)[gradient].value2++;
325 
326  interp_val = func_1(grad, interp_val, code);
327 
328  if (interp_val < 0) {
329  interp_val += common_info.total_values;
330  } else if (interp_val > common_info.q_point[4]) {
331  interp_val -= common_info.total_values;
332  }
333 
334  if (interp_val >= 0) {
335  line_buf_cur[0] = std::min(interp_val, common_info.q_point[4]);
336  } else {
337  line_buf_cur[0] = 0;
338  }
339 
340  *pos += 2;
341 }
342 
343 #define fuji_quant_gradient(v1, v2)
344  (9 * ci.q_table[ci.q_point[4] + (v1)] + ci.q_table[ci.q_point[4] + (v2)])

346 void FujiDecompressor::fuji_decode_sample_even(
347  fuji_compressed_block* info, uint16_t* line_buf, int* pos,
348  std::array<int_pair, 41>* grads) const {
349  const auto& ci = common_info;
350  fuji_decode_sample(
351  [&ci](const uint16_t* line_buf_cur, int* interp_val, int* grad,
352  int* gradient) {
353  int Rb = line_buf_cur[-2 - ci.line_width];
354  int Rc = line_buf_cur[-3 - ci.line_width];
355  int Rd = line_buf_cur[-1 - ci.line_width];
356  int Rf = line_buf_cur[-4 - 2 * ci.line_width];
357 
358  int diffRcRb;
359  int diffRfRb;
360  int diffRdRb;
361 
362  *grad = fuji_quant_gradient(Rb - Rf, Rc - Rb);
363  *gradient = std::abs(*grad);
364  diffRcRb = std::abs(Rc - Rb);
365  diffRfRb = std::abs(Rf - Rb);
366  diffRdRb = std::abs(Rd - Rb);
367 
368  if (diffRcRb > diffRfRb && diffRcRb > diffRdRb) {
369  *interp_val = Rf + Rd + 2 * Rb;
370  } else if (diffRdRb > diffRcRb && diffRdRb > diffRfRb) {
371  *interp_val = Rf + Rc + 2 * Rb;
372  } else {
373  *interp_val = Rd + Rc + 2 * Rb;
374  }
375  },
376  [](int grad, int interp_val, int code) {
377  if (grad < 0) {
378  interp_val = (interp_val >> 2) - code;
379  } else {
380  interp_val = (interp_val >> 2) + code;
381  }
382 
383  return interp_val;
384  },
385  info, line_buf, pos, grads);
386 }

388 void FujiDecompressor::fuji_decode_sample_odd(
389  fuji_compressed_block* info, uint16_t* line_buf, int* pos,
390  std::array<int_pair, 41>* grads) const {
391  const auto& ci = common_info;
392  fuji_decode_sample(
393  [&ci](const uint16_t* line_buf_cur, int* interp_val, int* grad,
394  int* gradient) {
395  int Ra = line_buf_cur[-1];
396  int Rb = line_buf_cur[-2 - ci.line_width];
397  int Rc = line_buf_cur[-3 - ci.line_width];
398  int Rd = line_buf_cur[-1 - ci.line_width];
399  int Rg = line_buf_cur[1];
400 
401  *grad = fuji_quant_gradient(Rb - Rc, Rc - Ra);
402  *gradient = std::abs(*grad);
403 
404  if ((Rb > Rc && Rb > Rd) || (Rb < Rc && Rb < Rd)) {
405  *interp_val = (Rg + Ra + 2 * Rb) >> 2;
406  } else {
407  *interp_val = (Ra + Rg) >> 1;
408  }
409  },
410  [](int grad, int interp_val, int code) {
411  if (grad < 0) {
412  interp_val -= code;
413  } else {
414  interp_val += code;
415  }
416 
417  return interp_val;
418  },
419  info, line_buf, pos, grads);
420 }

424 void FujiDecompressor::fuji_decode_interpolation_even(int line_width,
425  uint16_t* line_buf,
426  int* pos) {
427  uint16_t* line_buf_cur = line_buf + *pos;
428  int Rb = line_buf_cur[-2 - line_width];
429  int Rc = line_buf_cur[-3 - line_width];
430  int Rd = line_buf_cur[-1 - line_width];
431  int Rf = line_buf_cur[-4 - 2 * line_width];
432  int diffRcRb = std::abs(Rc - Rb);
433  int diffRfRb = std::abs(Rf - Rb);
434  int diffRdRb = std::abs(Rd - Rb);
435 
436  if (diffRcRb > diffRfRb && diffRcRb > diffRdRb) {
437  *line_buf_cur = (Rf + Rd + 2 * Rb) >> 2;
438  } else if (diffRdRb > diffRcRb && diffRdRb > diffRfRb) {
439  *line_buf_cur = (Rf + Rc + 2 * Rb) >> 2;
440  } else {
441  *line_buf_cur = (Rd + Rc + 2 * Rb) >> 2;
442  }
443 
444  *pos += 2;
445 }

470 void FujiDecompressor::xtrans_decode_block(fuji_compressed_block* info,
471  int cur_line) const {
472  struct ColorPos {
473  int even = 0;
474  int odd = 1;
475 
476  void reset() {
477  even = 0;
478  odd = 1;
479  }
480  };
481 
482  ColorPos r;
483  ColorPos g;
484  ColorPos b;
485 
486  const int line_width = common_info.line_width;
487 
488  // FIXME: GCC5 sucks.
489  // https://github.com/darktable-org/rawspeed/issues/112
490  // https://bugs.debian.org/cgi-bin/bugreport.cgi?bug=871250
491  // https://bugs.launchpad.net/linuxmint/+bug/1709234
492  auto pass = [&, line_width](auto&& even_func, xt_lines c0, xt_lines c1,
493  int grad, ColorPos& c0_pos, ColorPos& c1_pos) {
494  while (g.even < line_width || g.odd < line_width) {
495  if (g.even < line_width)
496  even_func(c0, c1, grad, c0_pos, c1_pos);
497 
498  if (g.even > 8) {
499  fuji_decode_sample_odd(info, info->linebuf[c0] + 1, &c0_pos.odd,
500  &(info->grad_odd[grad]));
501  fuji_decode_sample_odd(info, info->linebuf[c1] + 1, &c1_pos.odd,
502  &(info->grad_odd[grad]));
503  }
504  }
505  };
506 
507  pass(
508  [&](xt_lines c0, xt_lines c1, int grad, ColorPos& c0_pos,
509  ColorPos& c1_pos) {
510  fuji_decode_interpolation_even(line_width, info->linebuf[c0] + 1,
511  &c0_pos.even);
512  fuji_decode_sample_even(info, info->linebuf[c1] + 1, &c1_pos.even,
513  &(info->grad_even[grad]));
514  },
515  R2, G2, 0, r, g);
516 
517  fuji_extend_red(info->linebuf, line_width);
518  fuji_extend_green(info->linebuf, line_width);
519 
520  g.reset();
521 
522  pass(
523  [&](xt_lines c0, xt_lines c1, int grad, ColorPos& c0_pos,
524  ColorPos& c1_pos) {
525  fuji_decode_sample_even(info, info->linebuf[c0] + 1, &c0_pos.even,
526  &(info->grad_even[grad]));
527  fuji_decode_interpolation_even(line_width, info->linebuf[c1] + 1,
528  &c1_pos.even);
529  },
530  G3, B2, 1, g, b);
531 
532  fuji_extend_green(info->linebuf, line_width);
533  fuji_extend_blue(info->linebuf, line_width);
534 
535  r.reset();
536  g.reset();
537 
538  pass(
539  [&](xt_lines c0, xt_lines c1, int grad, ColorPos& c0_pos,
540  ColorPos& c1_pos) {
541  if (c0_pos.even & 3) {
542  fuji_decode_sample_even(info, info->linebuf[c0] + 1, &c0_pos.even,
543  &(info->grad_even[grad]));
544  } else {
545  fuji_decode_interpolation_even(line_width, info->linebuf[c0] + 1,
546  &c0_pos.even);
547  }
548 
549  fuji_decode_interpolation_even(line_width, info->linebuf[c1] + 1,
550  &c1_pos.even);
551  },
552  R3, G4, 2, r, g);
553 
554  fuji_extend_red(info->linebuf, line_width);
555  fuji_extend_green(info->linebuf, line_width);
556 
557  g.reset();
558  b.reset();
559 
560  pass(
561  [&](xt_lines c0, xt_lines c1, int grad, ColorPos& c0_pos,
562  ColorPos& c1_pos) {
563  fuji_decode_sample_even(info, info->linebuf[c0] + 1, &c0_pos.even,
564  &(info->grad_even[grad]));
565 
566  if ((c1_pos.even & 3) == 2) {
567  fuji_decode_interpolation_even(line_width, info->linebuf[c1] + 1,
568  &c1_pos.even);
569  } else {
570  fuji_decode_sample_even(info, info->linebuf[c1] + 1, &c1_pos.even,

572  }
573  },
574  G5, B3, 0, g, b);

576  fuji_extend_green(info->linebuf, line_width);
577  fuji_extend_blue(info->linebuf, line_width);

579  r.reset();

581 
582  pass(
583  [&](xt_lines c0, xt_lines c1, int grad, ColorPos& c0_pos,
584  ColorPos& c1_pos) {
585  if ((c0_pos.even & 3) == 2) {
586  fuji_decode_interpolation_even(line_width, info->linebuf[c0] + 1,
587  &c0_pos.even);
588  } else {
589  fuji_decode_sample_even(info, info->linebuf[c0] + 1, &c0_pos.even,

624 void FujiDecompressor::fuji_bayer_decode_block(fuji_compressed_block* info,
625  int cur_line) const {
626  struct ColorPos {
627  int even = 0;
628  int odd = 1;
629 
630  void reset() {
631  even = 0;
632  odd = 1;
633  }
634  };
635 
636  ColorPos r;
637  ColorPos g;
638  ColorPos b;
639 
640  const int line_width = common_info.line_width;
641 
642  auto pass = [&](xt_lines c0, xt_lines c1, int grad, ColorPos& c0_pos,
643  ColorPos& c1_pos) {
644  while (g.even < line_width || g.odd < line_width) {
645  if (g.even < line_width) {
646  fuji_decode_sample_even(info, info->linebuf[c0] + 1, &c0_pos.even,
647  &(info->grad_even[grad]));
648  fuji_decode_sample_even(info, info->linebuf[c1] + 1, &c1_pos.even,
649  &(info->grad_even[grad]));
650  }
651 
652  if (g.even > 8) {
653  fuji_decode_sample_odd(info, info->linebuf[c0] + 1, &c0_pos.odd,
654  &(info->grad_odd[grad]));
655  fuji_decode_sample_odd(info, info->linebuf[c1] + 1, &c1_pos.odd,
656  &(info->grad_odd[grad]));
657  }
658  }
659  };
660 
661  auto pass_RG = [&](xt_lines c0, xt_lines c1, int grad) {
662  pass(c0, c1, grad, r, g);
663 
664  fuji_extend_red(info->linebuf, line_width);
665  fuji_extend_green(info->linebuf, line_width);
666  };
667 
668  auto pass_GB = [&](xt_lines c0, xt_lines c1, int grad) {
669  pass(c0, c1, grad, g, b);
670 
671  fuji_extend_green(info->linebuf, line_width);
672  fuji_extend_blue(info->linebuf, line_width);
673  };
674 
675  pass_RG(R2, G2, 0);
676 
677  g.reset();
678 
679  pass_GB(G3, B2, 1);
680 
681  r.reset();
682  g.reset();
683 
684  pass_RG(R3, G4, 2);
685 
686  g.reset();
687  b.reset();
688 
689  pass_GB(G5, B3, 0);
690 
691  r.reset();
692  g.reset();
693 
694  pass_RG(R4, G6, 1);
695 
696  g.reset();
697  b.reset();
698 
699  pass_GB(G7, B4, 2);
700 }

702 void FujiDecompressor::fuji_decode_strip(
703  fuji_compressed_block* info_block, const FujiStrip& strip) const {
704  BitPumpMSB pump(strip.bs);
705 
706  const unsigned line_size = sizeof(uint16_t) * (common_info.line_width + 2);
707 
708  struct i_pair {
709  int a;
710  int b;
711  };
712 
713  const std::array<i_pair, 6> mtable = {
714  {{R0, R3}, {R1, R4}, {G0, G6}, {G1, G7}, {B0, B3}, {B1, B4}}};
715  const std::array<i_pair, 3> ztable = {{{R2, 3}, {G2, 6}, {B2, 3}}};
716 
717  for (int cur_line = 0; cur_line < strip.height(); cur_line++) {
718  if (header.raw_type == 16) {
719  xtrans_decode_block(info_block, cur_line);
720  } else {
721  fuji_bayer_decode_block(info_block, cur_line);
722  }
723 
724  // copy data from line buffers and advance
725  for (auto i : mtable) {
726  memcpy(info_block->linebuf[i.a], info_block->linebuf[i.b], line_size);
727  }
728 
729  if (header.raw_type == 16) {
730  copy_line_to_xtrans(info_block, strip, cur_line);
731  } else {
732  copy_line_to_bayer(info_block, strip, cur_line);
733  }
734 
735  for (auto i : ztable) {
736  memset(info_block->linebuf[i.a], 0, i.b * line_size);
737  info_block->linebuf[i.a][0] = info_block->linebuf[i.a - 1][1];
738  info_block->linebuf[i.a][common_info.line_width + 1] =
739  info_block->linebuf[i.a - 1][common_info.line_width];
740  }
741  }
742 }

744 void FujiDecompressor::fuji_compressed_load_raw() {
745  common_info = fuji_compressed_params(*this);
746 
747  // read block sizes
748  std::vector<uint32_t> block_sizes;
749  block_sizes.resize(header.blocks_in_row);
750  for (auto& block_size : block_sizes)
751  block_size = input.getU32();
752 
753  // some padding?
754  const uint64_t raw_offset = sizeof(uint32_t) * header.blocks_in_row;
755  if (raw_offset & 0xC) {
756  const int padding = 0x10 - (raw_offset & 0xC);
757  input.skipBytes(padding);
758  }
759 
760  // calculating raw block offsets
761  strips.reserve(header.blocks_in_row);
762 
763  int block = 0;
764  for (const auto& block_size : block_sizes) {
765  strips.emplace_back(header, block, input.getStream(block_size));
766  block++;
767  }
768 }
