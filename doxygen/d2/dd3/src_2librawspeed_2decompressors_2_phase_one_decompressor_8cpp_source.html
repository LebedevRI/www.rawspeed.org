61 void PhaseOneDecompressor::prepareStrips() {
62  // The 'strips' vector should contain exactly one element per row of image.
63 
64  // If the length is different, then the 'strips' vector is clearly incorrect.
65  if (strips.size() != static_cast<decltype(strips)::size_type>(mRaw->dim.y)) {
66  ThrowRDE("Height (%u) vs strip count %zu mismatch", mRaw->dim.y,
67  strips.size());
68  }
69 
70  // Now, the strips in 'strips' vector aren't in order.
71  // The 'decltype(strips)::value_type::n' is the row number of a strip.
72  // We need to make sure that we have every row (0..mRaw->dim.y-1), once.
73  // For that, first let's sort them to have monothonically increasting `n`.
74  // This will also serialize the per-line outputting.
75  std::sort(
76  strips.begin(), strips.end(),
77  [](const PhaseOneStrip& a, const PhaseOneStrip& b) { return a.n < b.n; });
78  // And now ensure that slice number matches the slice's row.
79  for (decltype(strips)::size_type i = 0; i < strips.size(); ++i)
80  if (static_cast<decltype(strips)::size_type>(strips[i].n) != i)
81  ThrowRDE("Strips validation issue.");
82  // All good.
83 }

85 void PhaseOneDecompressor::decompressStrip(const PhaseOneStrip& strip) const {
86  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
87 
88  assert(out.width > 0);
89  assert(out.width % 2 == 0);
90 
91  static constexpr std::array<const int, 10> length = {8, 7, 6, 9, 11,
92  10, 5, 12, 14, 13};
93 
94  BitPumpMSB32 pump(strip.bs);
95 
96  std::array<int32_t, 2> pred;
97  pred.fill(0);
98  std::array<int, 2> len;
99  const int row = strip.n;
100  for (int col = 0; col < out.width; col++) {
101  pump.fill(32);
102  if (static_cast<unsigned>(col) >=
103  (out.width & ~7U)) // last 'width % 8' pixels.
104  len[0] = len[1] = 14;
105  else if ((col & 7) == 0) {
106  for (int& i : len) {
107  int j = 0;
108 
109  for (; j < 5; j++) {
110  if (pump.getBitsNoFill(1) != 0) {
111  if (col == 0)
112  ThrowRDE("Can not initialize lengths. Data is corrupt.");
113 
114  // else, we have previously initialized lengths, so we are fine
115  break;
116  }
117  }
118 
119  assert((col == 0 && j > 0) || col != 0);
120  if (j > 0)
121  i = length[2 * (j - 1) + pump.getBitsNoFill(1)];
122  }
123  }
124 
125  int i = len[col & 1];
126  if (i == 14)
127  out(row, col) = pred[col & 1] = pump.getBitsNoFill(16);
128  else {
129  pred[col & 1] +=
130  static_cast<signed>(pump.getBitsNoFill(i)) + 1 - (1 << (i - 1));
131  // FIXME: is the truncation the right solution here?
132  out(row, col) = uint16_t(pred[col & 1]);
133  }
134  }
135 }

137 void PhaseOneDecompressor::decompressThread() const noexcept {
138 #ifdef HAVE_OPENMP
139 #pragma omp for schedule(static)
140 #endif
141  for (auto strip = strips.cbegin(); strip < strips.cend(); ++strip) {
142  try {
143  decompressStrip(*strip);
144  } catch (RawspeedException& err) {
145  // Propagate the exception out of OpenMP magic.
146  mRaw->setError(err.what());
147  }
148  }
149 }
