59 void RawImageDataU16::calculateBlackAreas() {
60  vector<unsigned int> histogram(4 * 65536);
61  fill(histogram.begin(), histogram.end(), 0);
62 
63  int totalpixels = 0;
64 
65  for (auto area : blackAreas) {
66  /* Make sure area sizes are multiple of two,
67  so we have the same amount of pixels for each CFA group */
68  area.size = area.size - (area.size&1);
69 
70  /* Process horizontal area */
71  if (!area.isVertical) {
72  if (static_cast<int>(area.offset) + static_cast<int>(area.size) >
73  uncropped_dim.y)
74  ThrowRDE("Offset + size is larger than height of image");
75  for (uint32_t y = area.offset; y < area.offset + area.size; y++) {
76  auto* pixel =
77  reinterpret_cast<uint16_t*>(getDataUncropped(mOffset.x, y));
78  auto* localhist = &histogram[(y & 1) * (65536UL * 2UL)];
79  for (int x = mOffset.x; x < dim.x+mOffset.x; x++) {
80  const auto hBin = ((x & 1) << 16) + *pixel;
81  localhist[hBin]++;
82  }
83  }
84  totalpixels += area.size * dim.x;
85  }
86 
87  /* Process vertical area */
88  if (area.isVertical) {
89  if (static_cast<int>(area.offset) + static_cast<int>(area.size) >
90  uncropped_dim.x)
91  ThrowRDE("Offset + size is larger than width of image");
92  for (int y = mOffset.y; y < dim.y+mOffset.y; y++) {
93  auto* pixel =
94  reinterpret_cast<uint16_t*>(getDataUncropped(area.offset, y));
95  auto* localhist = &histogram[(y & 1) * (65536UL * 2UL)];
96  for (uint32_t x = area.offset; x < area.size + area.offset; x++) {
97  const auto hBin = ((x & 1) << 16) + *pixel;
98  localhist[hBin]++;
99  }
100  }
101  totalpixels += area.size * dim.y;
102  }
103  }
104 
105  if (!totalpixels) {
106  for (int &i : blackLevelSeparate)
107  i = blackLevel;
108  return;
109  }
110 
111  /* Calculate median value of black areas for each component */
112  /* Adjust the number of total pixels so it is the same as the median of each histogram */
113  totalpixels /= 4*2;
114 
115  for (int i = 0 ; i < 4; i++) {
116  auto* localhist = &histogram[i * 65536UL];
117  int acc_pixels = localhist[0];
118  int pixel_value = 0;
119  while (acc_pixels <= totalpixels && pixel_value < 65535) {
120  pixel_value++;
121  acc_pixels += localhist[pixel_value];
122  }
123  blackLevelSeparate[i] = pixel_value;
124  }
125 
126  /* If this is not a CFA image, we do not use separate blacklevels, use average */
127  if (!isCFA) {
128  int total = 0;
129  for (int i : blackLevelSeparate)
130  total += i;
131  for (int &i : blackLevelSeparate)
132  i = (total + 2) >> 2;
133  }
134 }

136 void RawImageDataU16::scaleBlackWhite() {
137  const int skipBorder = 250;
138  int gw = (dim.x - skipBorder) * cpp;
139  if ((blackAreas.empty() && blackLevelSeparate[0] < 0 && blackLevel < 0) || whitePoint >= 65536) { // Estimate
140  int b = 65536;
141  int m = 0;
142  for (int row = skipBorder; row < (dim.y - skipBorder);row++) {
143  auto* pixel = reinterpret_cast<uint16_t*>(getData(skipBorder, row));
144  for (int col = skipBorder ; col < gw ; col++) {
145  b = min(static_cast<int>(*pixel), b);
146  m = max(static_cast<int>(*pixel), m);
147  pixel++;
148  }
149  }
150  if (blackLevel < 0)
151  blackLevel = b;
152  if (whitePoint >= 65536)
153  whitePoint = m;
154  writeLog(DEBUG_PRIO_INFO, "ISO:%d, Estimated black:%d, Estimated white: %d",
155  metadata.isoSpeed, blackLevel, whitePoint);
156  }
157 
158  /* Skip, if not needed */
159  if ((blackAreas.empty() && blackLevel == 0 && whitePoint == 65535 &&
160  blackLevelSeparate[0] < 0) ||
161  dim.area() <= 0)
162  return;
163 
164  /* If filter has not set separate blacklevel, compute or fetch it */
165  if (blackLevelSeparate[0] < 0)
166  calculateBlackAreas();
167 
168  startWorker(RawImageWorker::SCALE_VALUES, true);
169 }

192 void RawImageDataU16::scaleValues_SSE2(int start_y, int end_y) {
193  int depth_values = whitePoint - blackLevelSeparate[0];
194  float app_scale = 65535.0F / depth_values;
195 
196  // Scale in 30.2 fp
197  auto full_scale_fp = static_cast<int>(app_scale * 4.0F);
198  // Half Scale in 18.14 fp
199  auto half_scale_fp = static_cast<int>(app_scale * 4095.0F);
200 
201  __m128i sseround;
202  __m128i ssesub2;
203  __m128i ssesign;
204  __m128i rand_mul;
205  __m128i rand_mask;
206  __m128i sse_full_scale_fp;
207  __m128i sse_half_scale_fp;
208 
209  auto* sub_mul = alignedMallocArray<uint32_t, 16, __m128i>(4);
210  if (!sub_mul)
211  ThrowRDE("Out of memory, failed to allocate 128 bytes");
212 
213  assert(sub_mul != nullptr);
214 
215  uint32_t gw = pitch / 16;
216  // 10 bit fraction
217  uint32_t mul = static_cast<int>(
218  1024.0F * 65535.0F /
219  static_cast<float>(whitePoint - blackLevelSeparate[mOffset.x & 1]));
220  mul |= (static_cast<int>(
221  1024.0F * 65535.0F /
222  static_cast<float>(whitePoint -
223  blackLevelSeparate[(mOffset.x + 1) & 1])))
224  << 16;
225  uint32_t b = blackLevelSeparate[mOffset.x & 1] |
226  (blackLevelSeparate[(mOffset.x + 1) & 1] << 16);
227 
228  for (int i = 0; i < 4; i++) {
229  sub_mul[i] = b; // Subtract even lines
230  sub_mul[4 + i] = mul; // Multiply even lines
231  }
232 
233  mul = static_cast<int>(
234  1024.0F * 65535.0F /
235  static_cast<float>(whitePoint - blackLevelSeparate[2 + (mOffset.x & 1)]));
236  mul |= (static_cast<int>(
237  1024.0F * 65535.0F /
238  static_cast<float>(whitePoint -
239  blackLevelSeparate[2 + ((mOffset.x + 1) & 1)])))
240  << 16;
241  b = blackLevelSeparate[2 + (mOffset.x & 1)] |
242  (blackLevelSeparate[2 + ((mOffset.x + 1) & 1)] << 16);
243 
244  for (int i = 0; i < 4; i++) {
245  sub_mul[8 + i] = b; // Subtract odd lines
246  sub_mul[12 + i] = mul; // Multiply odd lines
247  }
248 
249  sseround = _mm_set_epi32(512, 512, 512, 512);
250  ssesub2 = _mm_set_epi32(32768, 32768, 32768, 32768);
251  ssesign = _mm_set_epi32(0x80008000, 0x80008000, 0x80008000, 0x80008000);
252  sse_full_scale_fp = _mm_set1_epi32(full_scale_fp | (full_scale_fp << 16));
253  sse_half_scale_fp = _mm_set1_epi32(half_scale_fp >> 4);
254 
255  if (mDitherScale) {
256  rand_mul = _mm_set1_epi32(0x4d9f1d32);
257  } else {
258  rand_mul = _mm_set1_epi32(0);
259  }
260  rand_mask = _mm_set1_epi32(0x00ff00ff); // 8 random bits
261 
262  for (int y = start_y; y < end_y; y++) {
263  __m128i sserandom;
264  if (mDitherScale) {
265  sserandom =
266  _mm_set_epi32(dim.x * 1676 + y * 18000, dim.x * 2342 + y * 34311,
267  dim.x * 4272 + y * 12123, dim.x * 1234 + y * 23464);
268  } else {
269  sserandom = _mm_setzero_si128();
270  }
271  auto* pixel = reinterpret_cast<__m128i*>(&data[(mOffset.y + y) * pitch]);
272  __m128i ssescale;
273  __m128i ssesub;
274  if (((y + mOffset.y) & 1) == 0) {
275  ssesub = _mm_load_si128(reinterpret_cast<__m128i*>(&sub_mul[0]));
276  ssescale = _mm_load_si128(reinterpret_cast<__m128i*>(&sub_mul[4]));
277  } else {
278  ssesub = _mm_load_si128(reinterpret_cast<__m128i*>(&sub_mul[8]));
279  ssescale = _mm_load_si128(reinterpret_cast<__m128i*>(&sub_mul[12]));
280  }
281 
282  for (uint32_t x = 0; x < gw; x++) {
283  __m128i pix_high;
284  __m128i temp;
285  _mm_prefetch(reinterpret_cast<char*>(pixel + 1), _MM_HINT_T0);
286  __m128i pix_low = _mm_load_si128(pixel);
287  // Subtract black
288  pix_low = _mm_subs_epu16(pix_low, ssesub);
289  // Multiply the two unsigned shorts and combine it to 32 bit result
290  pix_high = _mm_mulhi_epu16(pix_low, ssescale);
291  temp = _mm_mullo_epi16(pix_low, ssescale);
292  pix_low = _mm_unpacklo_epi16(temp, pix_high);
293  pix_high = _mm_unpackhi_epi16(temp, pix_high);
294  // Add rounder
295  pix_low = _mm_add_epi32(pix_low, sseround);
296  pix_high = _mm_add_epi32(pix_high, sseround);
297 
298  sserandom = _mm_xor_si128(_mm_mulhi_epi16(sserandom, rand_mul),
299  _mm_mullo_epi16(sserandom, rand_mul));
300  __m128i rand_masked =
301  _mm_and_si128(sserandom, rand_mask); // Get 8 random bits
302  rand_masked = _mm_mullo_epi16(rand_masked, sse_full_scale_fp);
303 
304  __m128i zero = _mm_setzero_si128();
305  __m128i rand_lo = _mm_sub_epi32(sse_half_scale_fp,
306  _mm_unpacklo_epi16(rand_masked, zero));
307  __m128i rand_hi = _mm_sub_epi32(sse_half_scale_fp,
308  _mm_unpackhi_epi16(rand_masked, zero));
309 
310  pix_low = _mm_add_epi32(pix_low, rand_lo);
311  pix_high = _mm_add_epi32(pix_high, rand_hi);
312 
313  // Shift down
314  pix_low = _mm_srai_epi32(pix_low, 10);
315  pix_high = _mm_srai_epi32(pix_high, 10);
316  // Subtract to avoid clipping
317  pix_low = _mm_sub_epi32(pix_low, ssesub2);
318  pix_high = _mm_sub_epi32(pix_high, ssesub2);
319  // Pack
320  pix_low = _mm_packs_epi32(pix_low, pix_high);
321  // Shift sign off
322  pix_low = _mm_xor_si128(pix_low, ssesign);
323  _mm_store_si128(pixel, pix_low);
324  pixel++;
325  }
326  }
327  alignedFree(sub_mul);
328 }

331 void RawImageDataU16::scaleValues_plain(int start_y, int end_y) {
332  int depth_values = whitePoint - blackLevelSeparate[0];
333  float app_scale = 65535.0F / depth_values;
334 
335  // Scale in 30.2 fp
336  auto full_scale_fp = static_cast<int>(app_scale * 4.0F);
337  // Half Scale in 18.14 fp
338  auto half_scale_fp = static_cast<int>(app_scale * 4095.0F);
339 
340  // Not SSE2
341  int gw = dim.x * cpp;
342  std::array<int, 4> mul;
343  std::array<int, 4> sub;
344  for (int i = 0; i < 4; i++) {
345  int v = i;
346  if ((mOffset.x & 1) != 0)
347  v ^= 1;
348  if ((mOffset.y & 1) != 0)
349  v ^= 2;
350  mul[i] = static_cast<int>(
351  16384.0F * 65535.0F /
352  static_cast<float>(whitePoint - blackLevelSeparate[v]));
353  sub[i] = blackLevelSeparate[v];
354  }
355  for (int y = start_y; y < end_y; y++) {
356  int v = dim.x + y * 36969;
357  auto* pixel = reinterpret_cast<uint16_t*>(getData(0, y));
358  int* mul_local = &mul[2 * (y & 1)];
359  int* sub_local = &sub[2 * (y & 1)];
360  for (int x = 0; x < gw; x++) {
361  int rand;
362  if (mDitherScale) {
363  v = 18000 * (v & 65535) + (v >> 16);
364  rand = half_scale_fp - (full_scale_fp * (v & 2047));
365  } else {
366  rand = 0;
367  }
368  pixel[x] = clampBits(
369  ((pixel[x] - sub_local[x & 1]) * mul_local[x & 1] + 8192 + rand) >>
370  14,
371  16);
372  }
373  }
374 }

381 void RawImageDataU16::fixBadPixel(uint32_t x, uint32_t y, int component) {
382  array<int, 4> values;
383  array<int, 4> dist;
384  array<int, 4> weight;
385 
386  values.fill(-1);
387  dist.fill(0);
388  weight.fill(0);
389 
390  uint8_t* bad_line = &mBadPixelMap[y * mBadPixelMapPitch];
391  int step = isCFA ? 2 : 1;
392 
393  // Find pixel to the left
394  int x_find = static_cast<int>(x) - step;
395  int curr = 0;
396  while (x_find >= 0 && values[curr] < 0) {
397  if (0 == ((bad_line[x_find>>3] >> (x_find&7)) & 1)) {
398  values[curr] =
399  (reinterpret_cast<uint16_t*>(getDataUncropped(x_find, y)))[component];
400  dist[curr] = static_cast<int>(x) - x_find;
401  }
402  x_find -= step;
403  }
404  // Find pixel to the right
405  x_find = static_cast<int>(x) + step;
406  curr = 1;
407  while (x_find < uncropped_dim.x && values[curr] < 0) {
408  if (0 == ((bad_line[x_find>>3] >> (x_find&7)) & 1)) {
409  values[curr] =
410  (reinterpret_cast<uint16_t*>(getDataUncropped(x_find, y)))[component];
411  dist[curr] = x_find - static_cast<int>(x);
412  }
413  x_find += step;
414  }
415 
416  bad_line = &mBadPixelMap[x>>3];
417  // Find pixel upwards
418  int y_find = static_cast<int>(y) - step;
419  curr = 2;
420  while (y_find >= 0 && values[curr] < 0) {
421  if (0 == ((bad_line[y_find*mBadPixelMapPitch] >> (x&7)) & 1)) {
422  values[curr] =
423  (reinterpret_cast<uint16_t*>(getDataUncropped(x, y_find)))[component];
424  dist[curr] = static_cast<int>(y) - y_find;
425  }
426  y_find -= step;
427  }
428  // Find pixel downwards
429  y_find = static_cast<int>(y) + step;
430  curr = 3;
431  while (y_find < uncropped_dim.y && values[curr] < 0) {
432  if (0 == ((bad_line[y_find*mBadPixelMapPitch] >> (x&7)) & 1)) {
433  values[curr] =
434  (reinterpret_cast<uint16_t*>(getDataUncropped(x, y_find)))[component];
435  dist[curr] = y_find - static_cast<int>(y);
436  }
437  y_find += step;
438  }
439 
440  // Find x weights
441  int total_dist_x = dist[0] + dist[1];
442 
443  int total_shifts = 7;
444  if (total_dist_x) {
445  weight[0] = dist[0] ? (total_dist_x - dist[0]) * 256 / total_dist_x : 0;
446  weight[1] = 256 - weight[0];
447  total_shifts++;
448  }
449 
450  // Find y weights
451  int total_dist_y = dist[2] + dist[3];
452  if (total_dist_y) {
453  weight[2] = dist[2] ? (total_dist_y - dist[2]) * 256 / total_dist_y : 0;
454  weight[3] = 256-weight[2];
455  total_shifts++;
456  }
457 
458  int total_pixel = 0;
459  for (int i = 0; i < 4; i++)
460  if (values[i] >= 0)
461  total_pixel += values[i] * weight[i];
462 
463  total_pixel >>= total_shifts;
464  auto* pix = reinterpret_cast<uint16_t*>(getDataUncropped(x, y));
465  pix[component] = clampBits(total_pixel, 16);
466 
467  /* Process other pixels - could be done inline, since we have the weights */
468  if (cpp > 1 && component == 0)
469  for (int i = 1; i < static_cast<int>(cpp); i++)
470  fixBadPixel(x,y,i);
471 }
