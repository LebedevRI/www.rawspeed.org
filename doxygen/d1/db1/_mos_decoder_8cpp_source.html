51 bool MosDecoder::isAppropriateDecoder(const TiffRootIFD* rootIFD,
52  const Buffer* file) {
53  try {
54  const auto id = rootIFD->getID();
55  const std::string& make = id.make;
56 
57  // This is messy. see https://github.com/darktable-org/rawspeed/issues/116
58  // Old Leafs are MOS, new ones are IIQ. Use IIQ's magic to differentiate.
59  return make == "Leaf" && !IiqDecoder::isAppropriateDecoder(file);
60  } catch (const TiffParserException&) {
61  // Last ditch effort to identify Leaf cameras that don't have a Tiff Make
62  // set
63  TiffEntry* softwareIFD = rootIFD->getEntryRecursive(SOFTWARE);
64  if (!softwareIFD)
65  return false;
66 
67  const string software = trimSpaces(softwareIFD->getString());
68  return software == "Camera Library";
69  }
70 }

90 string MosDecoder::getXMPTag(const string &xmp, const string &tag) {
91  string::size_type start = xmp.find("<tiff:"+tag+">");
92  string::size_type end = xmp.find("</tiff:"+tag+">");
93  if (start == string::npos || end == string::npos || end <= start)
94  ThrowRDE("Couldn't find tag '%s' in the XMP", tag.c_str());
95  int startlen = tag.size()+7;
96  return xmp.substr(start+startlen, end-start-startlen);
97 }

99 RawImage MosDecoder::decodeRawInternal() {
100  uint32_t off = 0;
101 
102  const TiffIFD *raw = nullptr;
103 
104  if (mRootIFD->hasEntryRecursive(TILEOFFSETS)) {
105  raw = mRootIFD->getIFDWithTag(TILEOFFSETS);
106  off = raw->getEntry(TILEOFFSETS)->getU32();
107  } else {
108  raw = mRootIFD->getIFDWithTag(CFAPATTERN);
109  off = raw->getEntry(STRIPOFFSETS)->getU32();
110  }
111 
112  uint32_t width = raw->getEntry(IMAGEWIDTH)->getU32();
113  uint32_t height = raw->getEntry(IMAGELENGTH)->getU32();
114 
115  // FIXME: could be wrong. max "active pixels" - "80 MP"
116  if (width == 0 || height == 0 || width > 10328 || height > 7760)
117  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
118 
119  mRaw->dim = iPoint2D(width, height);
120  mRaw->createData();
121 
122  const ByteStream bs(DataBuffer(mFile->getSubView(off), Endianness::little));
123  if (bs.getRemainSize() == 0)
124  ThrowRDE("Input buffer is empty");
125 
126  UncompressedDecompressor u(bs, mRaw);
127 
128  int compression = raw->getEntry(COMPRESSION)->getU32();
129  if (1 == compression) {
130  const Endianness endianness =
131  getTiffByteOrder(ByteStream(DataBuffer(*mFile, Endianness::little)), 0);
132 
133  if (Endianness::big == endianness)
134  u.decodeRawUnpacked<16, Endianness::big>(width, height);
135  else
136  u.decodeRawUnpacked<16, Endianness::little>(width, height);
137  }
138  else if (99 == compression || 7 == compression) {
139  ThrowRDE("Leaf LJpeg not yet supported");
140  //LJpegPlain l(mFile, mRaw);
141  //l.startDecoder(off, mFile->getSize()-off, 0, 0);
142  } else
143  ThrowRDE("Unsupported compression: %d", compression);
144 
145  return mRaw;
146 }

152 void MosDecoder::decodeMetaDataInternal(const CameraMetaData* meta) {
153  RawDecoder::setMetaData(meta, make, model, "", 0);
154 
155  // Fetch the white balance (see dcraw.c parse_mos for more metadata that can be gotten)
156  if (mRootIFD->hasEntryRecursive(LEAFMETADATA)) {
157  ByteStream bs = mRootIFD->getEntryRecursive(LEAFMETADATA)->getData();
158 
159  // We need at least a couple of bytes:
160  // "NeutObj_neutrals" + 28 bytes binary + 4x uint as strings + 3x space + \0
161  const uint32_t minSize = 16 + 28 + 4 + 3 + 1;
162 
163  // dcraw does actual parsing, since we just want one field we bruteforce it
164  while (bs.getRemainSize() > minSize) {
165  if (bs.skipPrefix("NeutObj_neutrals", 16)) {
166  bs.skipBytes(28);
167  // check for nulltermination of string inside bounds
168  if (!memchr(bs.peekData(bs.getRemainSize()), 0, bs.getRemainSize()))
169  break;
170  std::array<uint32_t, 4> tmp = {{}};
171  std::istringstream iss(bs.peekString());
172  iss >> tmp[0] >> tmp[1] >> tmp[2] >> tmp[3];
173  if (!iss.fail() && tmp[0] > 0 && tmp[1] > 0 && tmp[2] > 0 &&
174  tmp[3] > 0) {
175  mRaw->metadata.wbCoeffs[0] = static_cast<float>(tmp[0]) / tmp[1];
176  mRaw->metadata.wbCoeffs[1] = static_cast<float>(tmp[0]) / tmp[2];
177  mRaw->metadata.wbCoeffs[2] = static_cast<float>(tmp[0]) / tmp[3];
178  }
179  break;
180  }
181  bs.skipBytes(1);
182  }
183  }
