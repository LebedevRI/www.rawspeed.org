45 enum class VC5Tag : int16_t {
46  NoTag = 0x0, // synthetic, not an actual tag
47 
48  ChannelCount = 0x000c,
49  ImageWidth = 0x0014,
50  ImageHeight = 0x0015,
51  LowpassPrecision = 0x0023,
52  SubbandCount = 0x000E,
53  SubbandNumber = 0x0030,
54  Quantization = 0x0035,
55  ChannelNumber = 0x003e,
56  ImageFormat = 0x0054,
57  MaxBitsPerComponent = 0x0066,
58  PatternWidth = 0x006a,
59  PatternHeight = 0x006b,
60  ComponentsPerSample = 0x006c,
61  PrescaleShift = 0x006d,
62 
63  LARGE_CHUNK = 0x2000,
64  SMALL_CHUNK = 0x4000,
65  UniqueImageIdentifier = 0x4004,
66  LargeCodeblock = 0x6000,
67 
68  Optional = int16_t(0x8000U), // only signbit set
69 };
70 inline VC5Tag operator&(VC5Tag LHS, VC5Tag RHS) {
71  using value_type = std::underlying_type<VC5Tag>::type;
72  return static_cast<VC5Tag>(static_cast<value_type>(LHS) &
73  static_cast<value_type>(RHS));
74 }
75 inline bool matches(VC5Tag LHS, VC5Tag RHS) {
76  // Are there any common bit set?
77  return (LHS & RHS) != VC5Tag::NoTag;
78 }
79 inline bool is(VC5Tag LHS, VC5Tag RHS) {
80  // Does LHS have all the RHS bits set?
81  return (LHS & RHS) == RHS;
82 }
83 inline VC5Tag operator-(VC5Tag tag) {
84  using value_type = std::underlying_type<VC5Tag>::type;
85  // Negate
86  return static_cast<VC5Tag>(-static_cast<value_type>(tag));
87 }
