161  /* Crop, so area is positive, and return true, if there is any area left */
162  /* This will ensure that bottomright is never on the left/top of the offset */
163  bool cropArea() {
164  dim.x = std::max(0, dim.x);
165  dim.y = std::max(0, dim.y);
166  return hasPositiveArea();
167  }
168 
169  /* This will make sure that offset is positive, and make the area smaller if
170  * needed */
171  /* This will return true if there is any area left */
172  bool cropOffsetToZero() {
173  iPoint2D crop_pixels;
174  if (pos.x < 0) {
175  crop_pixels.x = -(pos.x);
176  pos.x = 0;
177  }
178  if (pos.y < 0) {
179  crop_pixels.y = -pos.y;
180  pos.y = 0;
181  }
182  dim -= crop_pixels;
183  return cropArea();
184  }

186  iRectangle2D getOverlap(const iRectangle2D& other) const {
187  iRectangle2D overlap;
188  iPoint2D br1 = getBottomRight();
189  iPoint2D br2 = other.getBottomRight();
190  overlap.setAbsolute(std::max(pos.x, other.pos.x),
191  std::max(pos.y, other.pos.y), std::min(br1.x, br2.x),
192  std::min(br1.y, br2.y));
193  return overlap;
194  }

202  std::max(br2.y, br2.y));
203  return combined;
204  }
