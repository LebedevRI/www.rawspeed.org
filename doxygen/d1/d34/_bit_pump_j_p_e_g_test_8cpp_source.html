69 TEST(BitPumpJPEGTest, 0xFF0x00Is0xFFTest) {
70  // If 0xFF0x00 byte sequence is found, it is just 0xFF, i.e. 0x00 is ignored.
71  static const std::array<uint8_t, 2 + 4> data{
72  {0xFF, 0x00, 0b10100100, 0b01000010, 0b00001000, 0b00011111}};
73 
74  const Buffer b(data.data(), data.size());
75 
76  for (auto e : {Endianness::little, Endianness::big}) {
77  const DataBuffer db(b, e);
78  const ByteStream bs(db);

80  BitPumpJPEG p(bs);
