63  struct FujiStrip {
64  // part of which 'image' this block is
65  const FujiHeader& h;
66 
67  // which strip is this, 0 .. h.blocks_in_row-1
68  const int n;
69 
70  // the compressed data of this strip
71  const ByteStream bs;
72 
73  FujiStrip(const FujiHeader& h_, int block, ByteStream bs_)
74  : h(h_), n(block), bs(std::move(bs_)) {
75  assert(n >= 0 && n < h.blocks_in_row);
76  }
77 
78  // each strip's line corresponds to 6 output lines.
79  static int lineHeight() { return 6; }
80 
81  // how many vertical lines does this block encode?
82  int height() const { return h.total_lines; }
83 
84  // how many horizontal pixels does this block encode?
85  int width() const {
86  // if this is not the last block, we are good.
87  if ((n + 1) != h.blocks_in_row)
88  return h.block_size;
89 
90  // ok, this is the last block...
91 
92  assert(h.block_size * h.blocks_in_row >= h.raw_width);
93  return h.raw_width - offsetX();
94  }
95 
96  // where vertically does this block start?
97  int offsetY(int line = 0) const {
98  (void)height(); // A note for NDEBUG builds that *this is used.
99  assert(line >= 0 && line < height());
100  return lineHeight() * line;
101  }
102 
103  // where horizontally does this block start?
104  int offsetX() const { return h.block_size * n; }
105  };
