127 template <typename T>
128 bool BinaryHuffmanTree<T>::Branch::pruneLeaflessBranches(
129  std::unique_ptr<Node>* top) {
130  if (!top)
131  return false;
132 
133  bool foundLeafs = false; // Any leafs in this branch?
134  (*top)->getAsBranch().forEachNode([&foundLeafs](std::unique_ptr<Node>* n) {
135  assert(n);
136  if (!(*n))
137  return false; // Nothing to do here, node is empty already, keep going.
138  switch (static_cast<typename Node::Type>(**n)) {
139  case Node::Type::Branch:
140  // Recurse. Any leafs in this branch?
141  if (Branch::pruneLeaflessBranches(n))
142  foundLeafs = true;
143  else
144  n->reset(); // Aha, dead branch, prune it!
145  break;
146  case Node::Type::Leaf:
147  foundLeafs = true; // Ok, this is a Leaf, great.
148  break;
149  }
150  return false; // keep going.
151  });
152 
153  if (!foundLeafs)
154  top->reset();
155 
156  return foundLeafs;
157 }

159 template <typename T>
160 std::vector<typename BinaryHuffmanTree<T>::Branch*>
161 BinaryHuffmanTree<T>::getAllBranchesOfDepth(int depth) {
162  assert(depth >= 0);
163 
164  if (0 == depth) {
165  // The root (depth == 0) is is special, and is *always* a Branch.
166  if (!root)
167  root = std::make_unique<Branch>();
168  return {&root->getAsBranch()};
169  }
170 
171  // Recursively get all branches of previous depth
172  auto prevBranches = getAllBranchesOfDepth(depth - 1);
173 
174  // Early return in case of no branches on previous depth
175  if (prevBranches.empty())
176  return {};
177 
178  // We will have at most twice as much branches as at the previous depth.
179  decltype(prevBranches) branches;
180  branches.reserve(2U * prevBranches.size());
181 
182  for (const auto& prevBranch : prevBranches) {
183  assert(prevBranch);
184 
185  prevBranch->forEachNode([&branches](std::unique_ptr<Node>* n) {
186  assert(n);
187  // If the Node is vacant, make it a branch.
188  // The user was supposed to create all the required Leafs before.
189  // We shall prune Leaf-less branches at the end
190  if (!(*n))
191  *n = std::make_unique<Branch>();
192  // If this is a branch, add it to the list.
193  if (Node::Type::Branch == static_cast<typename Node::Type>(**n))
194  branches.emplace_back(&((*n)->getAsBranch()));
195  return false; // keep going;
196  });
197  }
198  assert(branches.size() <= 2U * prevBranches.size());
199 
200  return branches;
201 }

203 template <typename T>
204 std::vector<std::unique_ptr<typename BinaryHuffmanTree<T>::Node>*>
205 BinaryHuffmanTree<T>::getAllVacantNodesAtDepth(int depth) {
206  assert(depth > 0);
207 
208  // Get all branches of previous depth
209  auto prevBranches = getAllBranchesOfDepth(depth - 1);
210 
211  // Early return in case of no branches on previous depth
212  if (prevBranches.empty())
213  return {};
214 
215  // We will have at most two nodes per each branch on the previous depth.
216  std::vector<std::unique_ptr<BinaryHuffmanTree<T>::Node>*> nodes;
217  nodes.reserve(2U * prevBranches.size());
218 
219  for (const auto& prevBranch : prevBranches) {
220  assert(prevBranch);
221 
222  auto& b = prevBranch->getAsBranch();
223 
224  b.forEachNode([&nodes](std::unique_ptr<Node>* n) {
225  assert(n);
226  if (!(*n)) // If there is no node already, then record it.
227  nodes.emplace_back(n);
228  return false; // keep going;
229  });
230  }
231  assert(nodes.size() <= 2U * prevBranches.size());
232 
233  return nodes;
234 }
