46 template <typename Tr, typename Tv>
47 inline constexpr bool __attribute__((const))
48 RangeContains(const Tr& r, Tv pos) {
49  if (pos < r.begin())
50  return false;
51 
52  assert(pos >= r.begin());
53  return r.end() > pos;

56 template <typename T>
57 inline constexpr bool __attribute__((const))
58 RangesOverlap(const T& lhs, const T& rhs) {
59  if (&lhs == &rhs)
60  return true;
61 
62  if (lhs.begin() == rhs.begin())
63  return true;
64 
65  const std::pair<const T&, const T&> ordered =
66  std::minmax(lhs, rhs, [](const T& r0, const T& r1) {
67  assert(r0.begin() != r1.begin());
68  return r0.begin() < r1.begin();
69  });
70 
71  assert(ordered.first.begin() < ordered.second.begin());
72  return RangeContains(ordered.first, ordered.second.begin());
73 }
