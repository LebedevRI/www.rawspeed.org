47 inline void copyPixels(uint8_t* dest, int dstPitch, const uint8_t* src,
48  int srcPitch, int rowSize, int height) {
49  if (height == 1 || (dstPitch == srcPitch && srcPitch == rowSize))
50  memcpy(dest, src, static_cast<size_t>(rowSize) * height);
51  else {
52  for (int y = height; y > 0; --y) {
53  memcpy(dest, src, rowSize);
54  dest += dstPitch;
55  src += srcPitch;
56  }
57  }
58 }

65 constexpr inline size_t __attribute__((const))
66 roundToMultiple(size_t value, size_t multiple, bool roundDown) {
67  if ((multiple == 0) || (value % multiple == 0))
68  return value;
69  // Drop remainder.
70  size_t roundedDown = value - (value % multiple);
71  if (roundDown) // If we were rounding down, then that's it.
72  return roundedDown;
73  // Else, just add one multiple.
74  return roundedDown + multiple;
75 }

121 // Clamps the given unsigned value to the range 0 .. 2^n-1, with n <= 16
122 template <class T>
123 inline constexpr __attribute__((const)) uint16_t clampBits(
124  T value, unsigned int nBits,
125  typename std::enable_if<std::is_unsigned<T>::value>::type* /*unused*/ =
126  nullptr) {
127  // We expect to produce uint16_t.
128  assert(nBits <= 16);
129  // Check that the clamp is not a no-op. Not of uint16_t to 16 bits e.g.
130  // (Well, not really, if we are called from clampBits<signed>, it's ok..).
131  assert(bitwidth<T>() > nBits); // If nBits >= bitwidth, then shift is UB.
132  const T maxVal = (T(1) << nBits) - T(1);
133  return std::min(value, maxVal);
134 }

136 // Clamps the given signed value to the range 0 .. 2^n-1, with n <= 16
137 template <typename T>
138 inline constexpr uint16_t __attribute__((const))
139 clampBits(T value, unsigned int nBits,
140  typename std::enable_if<std::is_signed<T>::value>::type* /*unused*/ =
141  nullptr) {
142  // If the value is negative, clamp it to zero.
143  value = std::max(value, T(0));
144  // Now, let the unsigned case clamp to the upper limit.
145  using UnsignedT = typename std::make_unsigned<T>::type;
146  return clampBits<UnsignedT>(value, nBits);

184 // Trim both leading and trailing spaces from the string
185 inline std::string trimSpaces(const std::string& str)
186 {
187  // Find the first character position after excluding leading blank spaces
188  size_t startpos = str.find_first_not_of(" \t");
189 
190  // Find the first character position from reverse af
191  size_t endpos = str.find_last_not_of(" \t");
192

194  if ((startpos == std::string::npos) || (endpos == std::string::npos))
195  return "";
196 
197  return str.substr(startpos, endpos - startpos + 1);

200 inline std::vector<std::string> splitString(const std::string& input,
201  char c = ' ')
202 {
203  std::vector<std::string> result;
204  const char* str = input.c_str();
205 
206  while (true) {
207  const char* begin = str;
208 
209  while (*str != c && *str != '\0')
210  str++;
211 
212  if (begin != str)
213  result.emplace_back(begin, str);
214 
215  const bool isNullTerminator = (*str == '\0');
216  str++;
217 
218  if (isNullTerminator)
219  break;
220  }
221 
222  return result;
223 }
