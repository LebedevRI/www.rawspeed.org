119 // yes, this is not cool. but i see no way to compute the hash of the
120 // full image, without duplicating image, and copying excluding padding
121 md5::md5_state imgDataHash(const RawImage& raw) {
122  md5::md5_state ret = md5::md5_init;
123 
124  const iPoint2D dimUncropped = raw->getUncroppedDim();
125 
126  vector<md5::md5_state> line_hashes;
127  line_hashes.resize(dimUncropped.y, md5::md5_init);
128 
129  for (int j = 0; j < dimUncropped.y; j++) {
130  auto* d = raw->getDataUncropped(0, j);
131  md5::md5_hash(d, raw->pitch - raw->padding, &line_hashes[j]);
132  }
133 
134  md5::md5_hash(reinterpret_cast<const uint8_t*>(&line_hashes[0]),
135  sizeof(line_hashes[0]) * line_hashes.size(), &ret);
136 
137  return ret;
138 }

234 void writePPM(const RawImage& raw, const string& fn) {
235  file_ptr f(fopen((fn + ".ppm").c_str(), "wb"), &fclose);
236 
237  const iPoint2D dimUncropped = raw->getUncroppedDim();
238  int width = dimUncropped.x;
239  int height = dimUncropped.y;
240  string format = raw->getCpp() == 1 ? "P5" : "P6";
241 
242  // Write PPM header
243  fprintf(f.get(), "%s\n%d %d\n65535\n", format.c_str(), width, height);
244 
245  width *= raw->getCpp();
246 
247  // Write pixels
248  for (int y = 0; y < height; ++y) {
249  auto* row = reinterpret_cast<uint16_t*>(raw->getDataUncropped(0, y));
250  // PPM is big-endian
251  for (int x = 0; x < width; ++x)
252  row[x] = getU16BE(row + x);
253 
254  fwrite(row, sizeof(*row), width, f.get());
255  }
256 }

258 void writePFM(const RawImage& raw, const string& fn) {
259  file_ptr f(fopen((fn + ".pfm").c_str(), "wb"), &fclose);
260 
261  const iPoint2D dimUncropped = raw->getUncroppedDim();
262  int width = dimUncropped.x;
263  int height = dimUncropped.y;
264  string format = raw->getCpp() == 1 ? "Pf" : "PF";
265 
266  // Write PFM header. if scale < 0, it is little-endian, if >= 0 - big-endian
267  int len = fprintf(f.get(), "%s\n%d %d\n-1.0", format.c_str(), width, height);
268 
269  // make sure that data starts at aligned offset. for sse
270  static const auto dataAlignment = 16;
271 
272  // regardless of padding, we need to write \n separator
273  const int realLen = len + 1;
274  // the first byte after that \n will be aligned
275  const int paddedLen = roundUp(realLen, dataAlignment);
276  assert(paddedLen > len);
277  assert(rawspeed::isAligned(paddedLen, dataAlignment));
278 
279  // how much padding?
280  const int padding = paddedLen - realLen;
281  assert(padding >= 0);
282  assert(rawspeed::isAligned(realLen + padding, dataAlignment));
283 
284  // and actually write padding + new line
285  len += fprintf(f.get(), "%0*i\n", padding, 0);
286  assert(paddedLen == len);
287 
288  // did we write a multiple of an alignment value?
289  assert(rawspeed::isAligned(len, dataAlignment));
290  assert(ftell(f.get()) == len);
291  assert(rawspeed::isAligned(ftell(f.get()), dataAlignment));
292 
293  width *= raw->getCpp();
294 
295  // Write pixels
296  for (int y = 0; y < height; ++y) {
297  // NOTE: pfm has rows in reverse order
298  const int row_in = height - 1 - y;
299  auto* row = reinterpret_cast<float*>(raw->getDataUncropped(0, row_in));
300 
301  // PFM can have any endiannes, let's write little-endian
302  for (int x = 0; x < width; ++x)
303  row[x] = getU32LE(row + x);
304 
305  fwrite(row, sizeof(*row), width, f.get());
306  }
307 }

322 size_t process(const string& filename, const CameraMetaData* metadata,
323  const options& o) {
324 
325  const string hashfile(filename + ".hash");
326 
327  // if creating hash and hash exists -> skip current file
328  // if not creating and hash is missing -> skip as well
329  // unless in force mode
330  ifstream hf(hashfile);
331  if (hf.good() == o.create && !o.force) {
332 #if !defined(__has_feature) || !__has_feature(thread_sanitizer)
333 #ifdef HAVE_OPENMP
334 #pragma omp critical(io)
335 #endif
336  cout << left << setw(55) << filename << ": hash "
337  << (o.create ? "exists" : "missing") << ", skipping" << endl;
338 #endif
339  return 0;
340  }
341 
342 // to narrow down the list of files that could have causes the crash
343 #if !defined(__has_feature) || !__has_feature(thread_sanitizer)
344 #ifdef HAVE_OPENMP
345 #pragma omp critical(io)
346 #endif
347  cout << left << setw(55) << filename << ": starting decoding ... " << endl;
348 #endif
349 
350  FileReader reader(filename.c_str());
351 
352  auto map(reader.readFile());
353  // Buffer* map = readFile( argv[1] );
354 
355  Timer t;
356 
357  RawParser parser(map.get());
358  auto decoder(parser.getDecoder(metadata));
359  // RawDecoder* decoder = parseRaw( map );
360 
361  decoder->failOnUnknown = false;
362  decoder->checkSupport(metadata);
363 
364  decoder->decodeRaw();
365  decoder->decodeMetaData(metadata);
366  RawImage raw = decoder->mRaw;
367  // RawImage raw = decoder->decode();
368 
369  auto time = t();
370 #if !defined(__has_feature) || !__has_feature(thread_sanitizer)
371 #ifdef HAVE_OPENMP
372 #pragma omp critical(io)
373 #endif
374  cout << left << setw(55) << filename << ": " << internal << setw(3)
375  << map->getSize() / 1000000 << " MB / " << setw(4) << time << " ms"
376  << endl;
377 #endif
378 
379  if (o.create) {
380  // write the hash. if force is set, then we are potentially overwriting here
381  ofstream f(hashfile);
382  f << img_hash(raw);
383  if (o.dump)
384  writeImage(raw, filename);
385  } else {
386  // do generate the hash string regardless.
387  string h = img_hash(raw);
388 
389  // normally, here we would compare the old hash with the new one
390  // but if the force is set, and the hash does not exist, do nothing.
391  if (!hf.good() && o.force)
392  return time;
393 
394  string truth((istreambuf_iterator<char>(hf)), istreambuf_iterator<char>());
395  if (h != truth) {
396  ofstream f(filename + ".hash.failed");
397  f << h;
398  if (o.dump)
399  writeImage(raw, filename + ".failed");
400  throw RstestHashMismatch("hash/metadata mismatch", time);
401  }
402  }
403 
404  return time;
405 }

451 
452  return 1;

455 static int usage(const char* progname) {
456  cout << "usage: " << progname << R"(
457  [-h] print this help
458  [-c] for each file: decode, compute hash and store it.
459  If hash exists, it does not recompute it, unless option -f is set!
460  [-f] if -c is set, then it will override the existing hashes.
461  If -c is not set, and the hash does not exist, then just decode,
462  but do not write the hash!
463  [-d] store decoded image as PPM
464  <FILE[S]> the file[s] to work on.
465 
466  With no options given, each raw with an accompanying hash will be decoded
467  and compared (unless option -f is set!) to the existing hash. A summary of
468  all errors/failed hash comparisons will be reported at the end.
469 
470  Suggested workflow for easy regression testing:
471  1. remove all .hash files and build 'trusted' version of this program
472  2. run with option '-c' -> creates .hash for all supported files
473  3. build new version to test for regressions
474  4. run with no option -> checks files with existing .hash
475  If the second run shows no errors, you have no regressions,
476  otherwise, the diff between hashes is appended to rstest.log
477 )";
478  return 0;

493  auto hasFlag = [argc, &remaining_argc, argv](const string& flag) {
494  bool found = false;
495  for (int i = 1; i < argc; ++i) {
496  if (!argv[i] || argv[i] != flag)
497  continue;
498  found = true;
499  argv[i] = nullptr;

501  }

503  };
504 
505  if (1 == argc || hasFlag("-h"))
506  return usage(argv[0]);

508  options o;

510  o.force = hasFlag("-f");
511  o.dump = hasFlag("-d");

515 #else

521 #ifdef HAVE_OPENMP
522 #pragma omp parallel for default(none) firstprivate(argc, argv, o)
523  OMPSHAREDCLAUSE(metadata) shared(cerr, failedTests) schedule(dynamic, 1)
524  reduction(+ : time) if(remaining_argc > 2)
525 #endif
526  for (int i = 1; i < argc; ++i) {
527  if (!argv[i])

529 
530  try {
531  try {

534  time += e.time;

541  {

544  cerr << msg << endl;
545 #endif
546  failedTests.emplace(argv[i], msg);
547  }
548  }
549  }
550 
551  cout << "Total decoding time: " << time / 1000.0 << "s" << endl << endl;
552 
553  return results(failedTests, o);
