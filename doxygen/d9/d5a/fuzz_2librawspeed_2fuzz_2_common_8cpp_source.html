56 rawspeed::ColorFilterArray CreateCFA(rawspeed::ByteStream* bs) {
57  assert(bs);
58 
59  const uint32_t cfaWidth = bs->getU32();
60  const uint32_t cfaHeight = bs->getU32();
61 
62  rawspeed::ColorFilterArray cfa;
63 
64  if (cfaHeight &&
65  cfaWidth > std::numeric_limits<decltype(cfaWidth)>::max() / cfaHeight)
66  ThrowIOE("Integer overflow when calculating CFA area");
67  bs->check(cfaWidth * cfaHeight, 4);
68 
69  cfa.setSize(rawspeed::iPoint2D(cfaWidth, cfaHeight));
70 
71  for (auto x = 0U; x < cfaWidth; x++) {
72  for (auto y = 0U; y < cfaHeight; y++) {
73  const uint32_t color = bs->getU32();
74  if (color >= static_cast<uint32_t>(rawspeed::CFA_END))
75  ThrowRSE("Unknown color: %u", color);
76 
77  cfa.setColorAt(rawspeed::iPoint2D(x, y), rawspeed::CFAColor(color));
