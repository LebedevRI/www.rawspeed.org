96  void createHuffmanTable() {
97  int p;
98  int i;
99  int l;
100  int lastp;
101  int si;
102  std::array<char, 257> huffsize;
103  std::array<uint16_t, 257> huffcode;
104  uint16_t code;
105  int size;
106  int value;
107  int ll;
108  int ul;
109 
110  /*
111  * Figure C.1: make table of Huffman code length for each symbol
112  * Note that this is in code-length order.
113  */
114  p = 0;
115  for (l = 1; l <= 16; l++) {
116  for (i = 1; i <= static_cast<int>(dctbl1.bits[l]); i++) {
117  huffsize[p++] = static_cast<char>(l);
118  if (p > 256)
119  ThrowRDE("LJpegDecompressor::createHuffmanTable: Code length too "
120  "long. Corrupt data.");
121  }
122  }
123  huffsize[p] = 0;
124  lastp = p;
125 
126  /*
127  * Figure C.2: generate the codes themselves
128  * Note that this is in code-length order.
129  */
130  code = 0;
131  si = huffsize[0];
132  p = 0;
133  while (huffsize[p]) {
134  while ((static_cast<int>(huffsize[p])) == si) {
135  huffcode[p++] = code;
136  code++;
137  }
138  code <<= 1;
139  si++;
140  if (p > 256)
141  ThrowRDE("createHuffmanTable: Code length too long. Corrupt data.");
142  }
143 
144  /*
145  * Figure F.15: generate decoding tables
146  */
147  dctbl1.mincode[0] = 0;
148  dctbl1.maxcode[0] = 0;
149  p = 0;
150  for (l = 1; l <= 16; l++) {
151  if (dctbl1.bits[l]) {
152  dctbl1.valptr[l] = p;
153  dctbl1.mincode[l] = huffcode[p];
154  p += dctbl1.bits[l];
155  dctbl1.maxcode[l] = huffcode[p - 1];
156  } else {
157  dctbl1.valptr[l] =
158  0xff; // This check must be present to avoid crash on junk
159  dctbl1.maxcode[l] = -1;
160  }
161  if (p > 256)
162  ThrowRDE("createHuffmanTable: Code length too long. Corrupt data.");
163  }
164 
165  /*
166  * We put in this value to ensure HuffDecode terminates.
167  */
168  dctbl1.maxcode[17] = 0xFFFFFL;
169 
170  /*
171  * Build the numbits, value lookup tables.
172  * These table allow us to gather 8 bits from the bits stream,
173  * and immediately lookup the size and value of the huffman codes.
174  * If size is zero, it means that more than 8 bits are in the huffman
175  * code (this happens about 3-4% of the time).
176  */
177  dctbl1.numbits.fill(0);
178  for (p = 0; p < lastp; p++) {
179  size = huffsize[p];
180  if (size <= 8) {
181  value = dctbl1.huffval[p];
182  code = huffcode[p];
183  ll = code << (8 - size);
184  if (size < 8) {
185  ul = ll | bitMask[24 + size];
186  } else {
187  ul = ll;
188  }
189  if (ul > 256 || ll > ul)
190  ThrowRDE("createHuffmanTable: Code length too long. Corrupt data.");
191  for (i = ll; i <= ul; i++) {
192  dctbl1.numbits[i] = size | (value << 4);
193  }
194  }
195  }
196  if (mUseBigtable)
197  createBigTable();
198  dctbl1.initialized = true;
199  }
200 
201  /************************************
202  * Bitable creation
203  *
204  * This is expanding the concept of fast lookups
205  *
206  * A complete table for 14 arbitrary bits will be
207  * created that enables fast lookup of number of bits used,
208  * and final delta result.
209  * Hit rate is about 90-99% for typical LJPEGS, usually about 98%
210  *
211  ************************************/
212 
213  void createBigTable() {
214  const uint32_t bits =
215  14; // HuffDecode functions must be changed, if this is modified.
216  const uint32_t size = 1 << bits;
217  int rv = 0;
218  int temp;
219  uint32_t l;
220 
221  dctbl1.bigTable.resize(size);
222  for (uint32_t i = 0; i < size; i++) {
223  uint16_t input = i << 2; // Calculate input value
224  int code = input >> 8; // Get 8 bits
225  uint32_t val = dctbl1.numbits[code];
226  l = val & 15;
227  if (l) {
228  rv = val >> 4;
229  } else {
230  l = 8;
231  while (code > dctbl1.maxcode[l]) {
232  temp = extractHighBits(input, l, /*effectiveBitwidth=*/15) & 1;
233  code = (code << 1) | temp;
234  l++;
235  }
236 
237  /*
238  * With garbage input we may reach the sentinel value l = 17.
239  */
240 
241  if (l > 16 || dctbl1.valptr[l] == 0xff) {
242  dctbl1.bigTable[i] = 0xff;
243  continue;
244  }
245  rv = dctbl1.huffval[dctbl1.valptr[l] + (code - dctbl1.mincode[l])];
246  }
247 
248  if (rv == 16) {
249  if (mDNGCompatible)
250  dctbl1.bigTable[i] = (-(32768 << 8)) | (16 + l);
251  else
252  dctbl1.bigTable[i] = (-(32768 << 8)) | l;
253  continue;
254  }
255 
256  if (rv + l > bits) {
257  dctbl1.bigTable[i] = 0xff;
258  continue;
259  }
260 
261  if (rv) {
262  int x = extractHighBits(input, l + rv) & ((1 << rv) - 1);
263  if ((x & (1 << (rv - 1))) == 0)
264  x -= (1 << rv) - 1;
265  dctbl1.bigTable[i] =
266  static_cast<int>((static_cast<unsigned>(x) << 8) | (l + rv));
267  } else {
268  dctbl1.bigTable[i] = l;
269  }
270  }
271  }

291  /*
292  *--------------------------------------------------------------
293  *
294  * HuffDecode --
295  *
296  * Taken from Figure F.16: extract next coded symbol from
297  * input stream. This should becode a macro.
298  *
299  * Results:
300  * Next coded symbol
301  *
302  * Side effects:
303  * Bitstream is parsed.
304  *
305  *--------------------------------------------------------------
306  */
307  int decodeDifference(BitPumpMSB& bits) { // NOLINT: google-runtime-references
308  int rv;
309  int l;
310  int temp;
311  int code;
312  unsigned val;
313 
314  bits.fill();
315  code = bits.peekBitsNoFill(14);
316  val = static_cast<unsigned>(dctbl1.bigTable[code]);
317  if ((val & 0xff) != 0xff) {
318  bits.skipBitsNoFill(val & 0xff);
319  return static_cast<int>(val) >> 8;
320  }
321 
322  rv = 0;
323  code = bits.peekBitsNoFill(8);
324  val = dctbl1.numbits[code];
325  l = val & 15;
326  if (l) {
327  bits.skipBitsNoFill(l);
328  rv = static_cast<int>(val) >> 4;
329  } else {
330  bits.skipBitsNoFill(8);
331  l = 8;
332  while (code > dctbl1.maxcode[l]) {
333  temp = bits.getBitsNoFill(1);
334  code = (code << 1) | temp;
335  l++;
336  }
337 
338  if (l > 16) {
339  ThrowRDE("Corrupt JPEG data: bad Huffman code:%u\n", l);
340  } else {
341  rv = dctbl1.huffval[dctbl1.valptr[l] + (code - dctbl1.mincode[l])];
342  }
343  }
344 
345  if (rv == 16)
346  return -32768;
347 
348  /*
349  * Section F.2.2.1: decode the difference and
350  * Figure F.12: extend sign bit
351  */
352  uint32_t len = rv & 15;
353  uint32_t shl = rv >> 4;
354  int diff = ((bits.getBits(len - shl) << 1) + 1) << shl >> 1;
355  if ((diff & (1 << (len - 1))) == 0)
356  diff -= (1 << len) - !shl;
357  return diff;
358  }

363 std::vector<uint16_t> NikonDecompressor::createCurve(ByteStream* metadata,
364  uint32_t bitsPS,
365  uint32_t v0, uint32_t v1,
366  uint32_t* split) {
367  // Nikon Z7 12/14 bit compressed hack.
368  if (v0 == 68 && v1 == 64)
369  bitsPS -= 2;
370 
371  // 'curve' will hold a peace wise linearly interpolated function.
372  // there are 'csize' segments, each is 'step' values long.
373  // the very last value is not part of the used table but necessary
374  // to linearly interpolate the last segment, therefore the '+1/-1'
375  // size adjustments of 'curve'.
376  std::vector<uint16_t> curve((1 << bitsPS & 0x7fff) + 1);
377  assert(curve.size() > 1);
378 
379  for (size_t i = 0; i < curve.size(); i++)
380  curve[i] = i;
381 
382  uint32_t step = 0;
383  uint32_t csize = metadata->getU16();
384  if (csize > 1)
385  step = curve.size() / (csize - 1);
386 
387  if (v0 == 68 && (v1 == 32 || v1 == 64) && step > 0) {
388  if ((csize - 1) * step != curve.size() - 1)
389  ThrowRDE("Bad curve segment count (%u)", csize);
390 
391  for (size_t i = 0; i < csize; i++)
392  curve[i * step] = metadata->getU16();
393  for (size_t i = 0; i < curve.size() - 1; i++) {
394  const uint32_t b_scale = i % step;
395 
396  const uint32_t a_pos = i - b_scale;
397  const uint32_t b_pos = a_pos + step;
398  assert(a_pos < curve.size());
399  assert(b_pos > 0);
400  assert(b_pos < curve.size());
401  assert(a_pos < b_pos);
402 
403  const uint32_t a_scale = step - b_scale;
404  curve[i] = (a_scale * curve[a_pos] + b_scale * curve[b_pos]) / step;
405  }
406 
407  metadata->setPosition(562);
408  *split = metadata->getU16();
409  } else if (v0 != 70) {
410  if (csize == 0 || csize > 0x4001)
411  ThrowRDE("Don't know how to compute curve! csize = %u", csize);
412 
413  curve.resize(csize + 1UL);
414  assert(curve.size() > 1);
415 
416  for (uint32_t i = 0; i < csize; i++) {
417  curve[i] = metadata->getU16();
418  }
419  }
420 
421  // and drop the last value
422  curve.resize(curve.size() - 1);
423  assert(!curve.empty());
424 
425  return curve;
426 }

450  switch (bitsPS) {

452  case 14:
453  break;
454  default:
455  ThrowRDE("Invalid bpp found: %u", bitsPS);

458  uint32_t v0 = metadata.getByte();

465 
466  if (v0 == 70)

483 template <typename Huffman>
484 void NikonDecompressor::decompress(BitPumpMSB* bits, int start_y, int end_y) {
485  Huffman ht = createHuffmanTable<Huffman>(huffSelect);
486 
487  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
488 
489  // allow gcc to devirtualize the calls below
490  auto* rawdata = reinterpret_cast<RawImageDataU16*>(mRaw.get());
491 
492  assert(out.width % 2 == 0);
493  assert(out.width >= 2);
494  for (int row = start_y; row < end_y; row++) {
495  std::array<int, 2> pred = pUp[row & 1];
496  for (int col = 0; col < out.width; col++) {
497  pred[col & 1] += ht.decodeDifference(*bits);
498  if (col < 2)
499  pUp[row & 1][col & 1] = pred[col & 1];
500  rawdata->setWithLookUp(clampBits(pred[col & 1], 15),
501  reinterpret_cast<uint8_t*>(&out(row, col)),
502  &random);
503  }
504  }

507 void NikonDecompressor::decompress(const ByteStream& data,
508  bool uncorrectedRawValues) {
509  RawImageCurveGuard curveHandler(&mRaw, curve, uncorrectedRawValues);
510 
511  BitPumpMSB bits(data);
512 
513  random = bits.peekBits(24);
514 
515  assert(split == 0 || split < static_cast<unsigned>(mRaw->dim.y));
516 
517  if (!split) {
518  decompress<HuffmanTable>(&bits, 0, mRaw->dim.y);
519  } else {
520  decompress<HuffmanTable>(&bits, 0, split);
521  huffSelect += 1;
522  decompress<NikonLASDecompressor>(&bits, split, mRaw->dim.y);
523  }
