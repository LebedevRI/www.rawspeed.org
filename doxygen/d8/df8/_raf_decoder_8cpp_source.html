49 bool RafDecoder::isRAF(const Buffer* input) {
50  static const std::array<char, 16> magic = {{'F', 'U', 'J', 'I', 'F', 'I', 'L',
51  'M', 'C', 'C', 'D', '-', 'R', 'A',
52  'W', ' '}};
53  const unsigned char* data = input->getData(0, magic.size());
54  return 0 == memcmp(data, magic.data(), magic.size());
55 }

67 RawImage RafDecoder::decodeRawInternal() {
68  const auto* raw = mRootIFD->getIFDWithTag(FUJI_STRIPOFFSETS);
69  uint32_t height = 0;
70  uint32_t width = 0;
71 
72  if (raw->hasEntry(FUJI_RAWIMAGEFULLHEIGHT)) {
73  height = raw->getEntry(FUJI_RAWIMAGEFULLHEIGHT)->getU32();
74  width = raw->getEntry(FUJI_RAWIMAGEFULLWIDTH)->getU32();
75  } else if (raw->hasEntry(IMAGEWIDTH)) {
76  TiffEntry *e = raw->getEntry(IMAGEWIDTH);
77  height = e->getU16(0);
78  width = e->getU16(1);
79  } else
80  ThrowRDE("Unable to locate image size");
81 
82  if (width == 0 || height == 0 || width > 11808 || height > 8754)
83  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
84 
85  if (raw->hasEntry(FUJI_LAYOUT)) {
86  TiffEntry *e = raw->getEntry(FUJI_LAYOUT);
87  alt_layout = !(e->getByte(0) >> 7);
88  }
89 
90  TiffEntry *offsets = raw->getEntry(FUJI_STRIPOFFSETS);
91  TiffEntry *counts = raw->getEntry(FUJI_STRIPBYTECOUNTS);
92 
93  if (offsets->count != 1 || counts->count != 1)
94  ThrowRDE("Multiple Strips found: %u %u", offsets->count, counts->count);
95 
96  ByteStream input(offsets->getRootIfdData());
97  input = input.getSubStream(offsets->getU32(), counts->getU32());
98 
99  if (isCompressed()) {
100  mRaw->metadata.mode = "compressed";
101 
102  mRaw->dim = iPoint2D(width, height);
103 
104  FujiDecompressor f(mRaw, input);
105 
106  mRaw->createData();
107 
108  f.decompress();
109 
110  return mRaw;
111  }
112 
113  // x-trans sensors report 14bpp, but data isn't packed
114  // thus, unless someone has any better ideas, let's autodetect it.
115  int bps;
116 
117  // Some fuji SuperCCD cameras include a second raw image next to the first one
118  // that is identical but darker to the first. The two combined can produce
119  // a higher dynamic range image. Right now we're ignoring it.
120  bool double_width;
121 
122  assert(!isCompressed());
123 
124  if (8UL * counts->getU32() >= 2UL * 16UL * width * height) {
125  bps = 16;
126  double_width = true;
127  } else if (8UL * counts->getU32() >= 2UL * 14UL * width * height) {
128  bps = 14;
129  double_width = true;
130  } else if (8UL * counts->getU32() >= 2UL * 12UL * width * height) {
131  bps = 12;
132  double_width = true;
133  } else if (8UL * counts->getU32() >= 16UL * width * height) {
134  bps = 16;
135  double_width = false;
136  } else if (8UL * counts->getU32() >= 14UL * width * height) {
137  bps = 14;
138  double_width = false;
139  } else if (8UL * counts->getU32() >= 12UL * width * height) {
140  bps = 12;
141  double_width = false;
142  } else {
143  ThrowRDE("Can not detect bitdepth. StripByteCounts = %u, width = %u, "
144  "height = %u",
145  counts->getU32(), width, height);
146  }
147 
148  double_width = hints.has("double_width_unpacked");
149  const uint32_t real_width = double_width ? 2U * width : width;
150 
151  mRaw->dim = iPoint2D(real_width, height);
152  mRaw->createData();
153 
154  UncompressedDecompressor u(input, mRaw);
155 
156  if (double_width) {
157  u.decodeRawUnpacked<16, Endianness::little>(width * 2, height);
158  } else if (input.getByteOrder() == Endianness::big &&
159  getHostEndianness() == Endianness::little) {
160  // FIXME: ^ that if seems fishy
161  u.decodeRawUnpacked<16, Endianness::big>(width, height);
162  } else {
163  iPoint2D pos(0, 0);
164  if (hints.has("jpeg32_bitorder")) {
165  u.readUncompressedRaw(mRaw->dim, pos, width * bps / 8, bps,
166  BitOrder_MSB32);
167  } else {
168  u.readUncompressedRaw(mRaw->dim, pos, width * bps / 8, bps, BitOrder_LSB);
169  }
170  }
171 
172  return mRaw;
173 }

191 void RafDecoder::decodeMetaDataInternal(const CameraMetaData* meta) {
192  int iso = 0;
193  if (mRootIFD->hasEntryRecursive(ISOSPEEDRATINGS))
194  iso = mRootIFD->getEntryRecursive(ISOSPEEDRATINGS)->getU32();
195  mRaw->metadata.isoSpeed = iso;
196 
197  // This is where we'd normally call setMetaData but since we may still need
198  // to rotate the image for SuperCCD cameras we do everything ourselves
199  auto id = mRootIFD->getID();
200  const Camera* cam = meta->getCamera(id.make, id.model, mRaw->metadata.mode);
201  if (!cam)
202  ThrowRDE("Couldn't find camera");
203 
204  assert(cam != nullptr);
205 
206  iPoint2D new_size(mRaw->dim);
207  iPoint2D crop_offset = iPoint2D(0,0);
208 
209  if (applyCrop) {
210  new_size = cam->cropSize;
211  crop_offset = cam->cropPos;
212  bool double_width = hints.has("double_width_unpacked");
213  // If crop size is negative, use relative cropping
214  if (new_size.x <= 0)
215  new_size.x = mRaw->dim.x / (double_width ? 2 : 1) - cam->cropPos.x + new_size.x;
216  else
217  new_size.x /= (double_width ? 2 : 1);
218  if (new_size.y <= 0)
219  new_size.y = mRaw->dim.y - cam->cropPos.y + new_size.y;
220  }
221 
222  bool rotate = hints.has("fuji_rotate");
223  rotate = rotate && fujiRotate;
224 
225  // Rotate 45 degrees - could be multithreaded.
226  if (rotate && !this->uncorrectedRawValues) {
227  // Calculate the 45 degree rotated size;
228  uint32_t rotatedsize;
229  uint32_t rotationPos;
230  if (alt_layout) {
231  rotatedsize = new_size.y+new_size.x/2;
232  rotationPos = new_size.x/2 - 1;
233  }
234  else {
235  rotatedsize = new_size.x+new_size.y/2;
236  rotationPos = new_size.x - 1;
237  }
238 
239  iPoint2D final_size(rotatedsize, rotatedsize-1);
240  RawImage rotated = RawImage::create(final_size, TYPE_USHORT16, 1);
241  rotated->clearArea(iRectangle2D(iPoint2D(0,0), rotated->dim));
242  rotated->metadata = mRaw->metadata;
243  rotated->metadata.fujiRotationPos = rotationPos;
244 
245  int dest_pitch = static_cast<int>(rotated->pitch) / 2;
246  auto* dst = reinterpret_cast<uint16_t*>(rotated->getData(0, 0));
247 
248  for (int y = 0; y < new_size.y; y++) {
249  auto* src = reinterpret_cast<uint16_t*>(
250  mRaw->getData(crop_offset.x, crop_offset.y + y));
251  for (int x = 0; x < new_size.x; x++) {
252  int h;
253  int w;
254  if (alt_layout) { // Swapped x and y
255  h = rotatedsize - (new_size.y + 1 - y + (x >> 1));
256  w = ((x+1) >> 1) + y;
257  } else {
258  h = new_size.x - 1 - x + (y >> 1);
259  w = ((y+1) >> 1) + x;
260  }
261  if (h < rotated->dim.y && w < rotated->dim.x)
262  dst[w + h * dest_pitch] = src[x];
263  else
264  ThrowRDE("Trying to write out of bounds");
265  }
266  }
267  mRaw = rotated;
268  } else if (applyCrop) {
269  mRaw->subFrame(iRectangle2D(crop_offset, new_size));
270  }
271 
272  const CameraSensorInfo *sensor = cam->getSensorInfo(iso);
273  mRaw->blackLevel = sensor->mBlackLevel;
274 
275  // at least the (bayer sensor) X100 comes with a tag like this:
276  if (mRootIFD->hasEntryRecursive(FUJI_BLACKLEVEL)) {
277  TiffEntry* sep_black = mRootIFD->getEntryRecursive(FUJI_BLACKLEVEL);
278  if (sep_black->count == 4)
279  {
280  for(int k=0;k<4;k++)
281  mRaw->blackLevelSeparate[k] = sep_black->getU32(k);
282  } else if (sep_black->count == 36) {
283  for (int& k : mRaw->blackLevelSeparate)
284  k = 0;
285 
286  for (int y = 0; y < 6; y++) {
287  for (int x = 0; x < 6; x++)
288  mRaw->blackLevelSeparate[2 * (y % 2) + (x % 2)] +=
289  sep_black->getU32(6 * y + x);
290  }
291 
292  for (int& k : mRaw->blackLevelSeparate)
293  k /= 9;
294  }
295  }
296 
297  mRaw->whitePoint = sensor->mWhiteLevel;
298  mRaw->blackAreas = cam->blackAreas;
299  mRaw->cfa = cam->cfa;
300  mRaw->metadata.canonical_make = cam->canonical_make;
301  mRaw->metadata.canonical_model = cam->canonical_model;
302  mRaw->metadata.canonical_alias = cam->canonical_alias;
303  mRaw->metadata.canonical_id = cam->canonical_id;
304  mRaw->metadata.make = id.make;
305  mRaw->metadata.model = id.model;
306 
307  if (mRootIFD->hasEntryRecursive(FUJI_WB_GRBLEVELS)) {
308  TiffEntry *wb = mRootIFD->getEntryRecursive(FUJI_WB_GRBLEVELS);
309  if (wb->count == 3) {
310  mRaw->metadata.wbCoeffs[0] = wb->getFloat(1);
311  mRaw->metadata.wbCoeffs[1] = wb->getFloat(0);
312  mRaw->metadata.wbCoeffs[2] = wb->getFloat(2);
313  }
314  } else if (mRootIFD->hasEntryRecursive(FUJIOLDWB)) {
315  TiffEntry *wb = mRootIFD->getEntryRecursive(FUJIOLDWB);
316  if (wb->count == 8) {
317  mRaw->metadata.wbCoeffs[0] = wb->getFloat(1);
318  mRaw->metadata.wbCoeffs[1] = wb->getFloat(0);
319  mRaw->metadata.wbCoeffs[2] = wb->getFloat(3);
320  }
321  }
322 }

324 int RafDecoder::isCompressed() {
325  const auto* raw = mRootIFD->getIFDWithTag(FUJI_STRIPOFFSETS);
326  uint32_t height = 0;
327  uint32_t width = 0;
328 
329  if (raw->hasEntry(FUJI_RAWIMAGEFULLHEIGHT)) {
330  height = raw->getEntry(FUJI_RAWIMAGEFULLHEIGHT)->getU32();
331  width = raw->getEntry(FUJI_RAWIMAGEFULLWIDTH)->getU32();
332  } else if (raw->hasEntry(IMAGEWIDTH)) {
333  TiffEntry* e = raw->getEntry(IMAGEWIDTH);
334  height = e->getU16(0);
335  width = e->getU16(1);
336  } else
337  ThrowRDE("Unable to locate image size");
338 
339  if (width == 0 || height == 0 || width > 11808 || height > 8754)
340  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
341 
342  uint32_t count = raw->getEntry(FUJI_STRIPBYTECOUNTS)->getU32();
343 
344  // The uncompressed raf's can be 12/14 bpp, so if it is less than that,
345  // then we are likely in compressed raf.
346  // FIXME: this can't be the correct way to detect this. But i'm not seeing
347  // anything in the diff between exiv2/exiftool dumps of {un,}compressed raws.
348  // Maybe we are supposed to check for valid FujiDecompressor::FujiHeader ?
349  return count * 8 / (width * height) < 12;
350 }
