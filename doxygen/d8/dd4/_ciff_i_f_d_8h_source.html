52  // CIFF IFD are tree-like structure, with branches.
53  // A branch (IFD) can have branches (IFDs) of it's own.
54  // We must be careful to weed-out all the degenerative cases that
55  // can be produced e.g. via fuzzing, or other means.
56  struct Limits final {
57  // How many layers of IFD's can there be?
58  // All RPU samples (as of 2018-02-13) are ok with 3.
59  // However, let's be on the safe side, and pad it by one.
60  static constexpr int Depth = 3 + 1;
61 
62  // How many sub-IFD's can this IFD have?
63  // NOTE: only for the given IFD, *NOT* recursively including all sub-IFD's!
64  // All RPU samples (as of 2018-02-13) are ok with 4.
65  // However, let's be on the safe side, and double it.
66  static constexpr int SubIFDCount = 4 * 2;
67 
68  // How many sub-IFD's can this IFD have, recursively?
69  // All RPU samples (as of 2018-02-13) are ok with 6.
70  // However, let's be on the safe side, and double it.
71  static constexpr int RecursiveSubIFDCount = 6 * 2;
72  };
