44 void ColorFilterArray::setSize(const iPoint2D& _size)
45 {
46  size = _size;
47 
48  if (size.area() > 36) {
49  // Bayer, FC() supports 2x8 pattern
50  // X-Trans is 6x6 pattern
51  // is there anything bigger?
52  ThrowRDE("if your CFA pattern is really %" PRId64 " pixels "
53  "in area we may as well give up now",
54  size.area());
55  }
56  if (size.area() <= 0)
57  return;
58  cfa.resize(size.area());
59  fill(cfa.begin(), cfa.end(), CFA_UNKNOWN);
60 }

133 uint32_t ColorFilterArray::shiftDcrawFilter(uint32_t filter, int x, int y) {
134  // filter is a series of 4 bytes that describe a 2x8 matrix. 2 is the width,
135  // 8 is the height. each byte describes a 2x2 pixel block. so each pixel has
136  // 2 bits of information. This allows to distinguish 4 different colors.
137 
138  if (std::abs(x) & 1) {
139  // A shift in x direction means swapping the first and second half of each
140  // nibble.
141  // see http://graphics.stanford.edu/~seander/bithacks.html#SwappingBitsXOR
142  for (int n = 0; n < 8; ++n) {
143  int i = n * 4;
144  int j = i + 2;
145  uint32_t t = ((filter >> i) ^ (filter >> j)) & ((1U << 2) - 1);
146  filter = filter ^ ((t << i) | (t << j));
147  }
148  }
149 
150  if (y == 0)
151  return filter;
152 
153  // A shift in y direction means rotating the whole int by 4 bits.
154  y *= 4;
155  y = y >= 0 ? y % 32 : 32 - ((-y) % 32);
156  filter = (filter >> y) | (filter << (32 - y));
157 
158  return filter;
159 }

185 static uint32_t toDcrawColor(CFAColor c) {
186  switch (c) {
187  case CFA_FUJI_GREEN:
188  case CFA_RED: return 0;

190  case CFA_GREEN: return 1;

199 uint32_t ColorFilterArray::getDcrawFilter() const {
200  //dcraw magic
201  if (size.x == 6 && size.y == 6)
202  return 9;
203 
204  if (cfa.empty() || size.x > 2 || size.y > 8 || !isPowerOfTwo(size.y))
205  return 1;
206 
207  uint32_t ret = 0;
208  for (int x = 0; x < 2; x++) {
209  for (int y = 0; y < 8; y++) {
210  uint32_t c = toDcrawColor(getColorAt(x, y));
211  int g = (x >> 1) * 8;
212  ret |= c << ((x&1)*2 + y*4 + g);
213  }
214  }
215 
216  writeLog(DEBUG_PRIO_EXTRA, "%s", asString().c_str());
217  writeLog(DEBUG_PRIO_EXTRA, "DCRAW filter:%x", ret);
218 
219  return ret;
220 }
