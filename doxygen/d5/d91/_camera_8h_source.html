45 class Hints
46 {
47  std::map<std::string, std::string> data;
48 public:
49  void add(const std::string& key, const std::string& value)
50  {
51  data.insert({key, value});
52  }
53 
54  bool has(const std::string& key) const
55  {
56  return data.find(key) != data.end();
57  }
58 
59  template <typename T>
60  T get(const std::string& key, T defaultValue) const
61  {
62  auto hint = data.find(key);
63  if (hint != data.end() && !hint->second.empty()) {
64  std::istringstream iss(hint->second);
65  iss >> defaultValue;
66  }
67  return defaultValue;
68  }
69 
70  bool get(const std::string& key, bool defaultValue) const {
71  auto hint = data.find(key);
72  if (hint == data.end())
73  return defaultValue;
74  return "true" == hint->second;
75  }
76 };
