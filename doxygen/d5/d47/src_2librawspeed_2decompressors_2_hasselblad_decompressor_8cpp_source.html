51 // Returns len bits as a signed value.
52 // Highest bit is a sign bit
53 inline int HasselbladDecompressor::getBits(BitPumpMSB32* bs, int len) {
54  if (!len)
55  return 0;
56  int diff = bs->getBits(len);
57  diff = HuffmanTable::extend(diff, len);
58  if (diff == 65535)
59  return -32768;
60  return diff;
61 }

63 void HasselbladDecompressor::decodeScan() {
64  if (frame.w != static_cast<unsigned>(mRaw->dim.x) ||
65  frame.h != static_cast<unsigned>(mRaw->dim.y)) {
66  ThrowRDE("LJPEG frame does not match EXIF dimensions: (%u; %u) vs (%i; %i)",
67  frame.w, frame.h, mRaw->dim.x, mRaw->dim.y);
68  }
69 
70  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
71 
72  assert(out.height > 0);
73  assert(out.width > 0);
74  assert(out.width % 2 == 0);
75 
76  const auto ht = getHuffmanTables<1>();
77  ht[0]->verifyCodeSymbolsAreValidDiffLenghts();
78 
79  BitPumpMSB32 bitStream(input);
80  // Pixels are packed two at a time, not like LJPEG:
81  // [p1_length_as_huffman][p2_length_as_huffman][p0_diff_with_length][p1_diff_with_length]|NEXT PIXELS
82  for (int row = 0; row < out.height; row++) {
83  int p1 = 0x8000 + pixelBaseOffset;
84  int p2 = 0x8000 + pixelBaseOffset;
85  for (int col = 0; col < out.width; col += 2) {
86  int len1 = ht[0]->decodeCodeValue(bitStream);
87  int len2 = ht[0]->decodeCodeValue(bitStream);
88  p1 += getBits(&bitStream, len1);
89  p2 += getBits(&bitStream, len2);
90  // NOTE: this is rather unusual and weird, but appears to be correct.
91  // clampBits(p, 16) results in completely garbled images.
92  out(row, col) = uint16_t(p1);
93  out(row, col + 1) = uint16_t(p2);
94  }
95  }
96  input.skipBytes(bitStream.getStreamPosition());
97 }
