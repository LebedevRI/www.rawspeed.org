32 inline Endianness getHostEndiannessRuntime() {
33  uint16_t testvar = 0xfeff;
34  uint32_t firstbyte = (reinterpret_cast<uint8_t*>(&testvar))[0];
35  if (firstbyte == 0xff)
36  return Endianness::little;
37  if (firstbyte == 0xfe)
38  return Endianness::big;
39 
40  assert(false);

85 // the float/double versions use two memcpy which guarantee strict aliasing
86 // and are compiled into the same assembly as the popular union trick.
87 inline float getByteSwapped(float f) {
88  uint32_t i;
89  memcpy(&i, &f, sizeof(i));
90  i = getByteSwapped(i);
91  memcpy(&f, &i, sizeof(i));
92  return f;
93 }

102 template <typename T> inline T getByteSwapped(const void* data, bool bswap) {
103  T ret;
104  // all interesting compilers optimize this memcpy into a single move
105  // this is the most effective way to load some bytes without running into
106  // alignment or aliasing issues
107  memcpy(&ret, data, sizeof(T));
108  return bswap ? getByteSwapped(ret) : ret;
109 }

111 // The following functions may be used to get a multi-byte sized tyoe from some
112 // memory location converted to the native byte order of the host.
113 // 'BE' suffix: source byte order is known to be big endian
114 // 'LE' suffix: source byte order is known to be little endian
115 // Note: these functions should be avoided if higher level access from
116 // Buffer/DataBuffer classes is available.
117 
118 template <typename T> inline T getBE(const void* data) {
119  return getByteSwapped<T>(data, getHostEndianness() == Endianness::little);
120 }
121 
122 template <typename T> inline T getLE(const void* data) {
123  return getByteSwapped<T>(data, getHostEndianness() == Endianness::big);
124 }
