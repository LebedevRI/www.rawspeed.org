61 RawImage Rw2Decoder::decodeRawInternal() {
62 
63  const TiffIFD* raw = nullptr;
64  bool isOldPanasonic = ! mRootIFD->hasEntryRecursive(PANASONIC_STRIPOFFSET);
65 
66  if (! isOldPanasonic)
67  raw = mRootIFD->getIFDWithTag(PANASONIC_STRIPOFFSET);
68  else
69  raw = mRootIFD->getIFDWithTag(STRIPOFFSETS);
70 
71  uint32_t height = raw->getEntry(static_cast<TiffTag>(3))->getU16();
72  uint32_t width = raw->getEntry(static_cast<TiffTag>(2))->getU16();
73 
74  if (isOldPanasonic) {
75  if (width == 0 || height == 0 || width > 4330 || height > 2751)
76  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
77 
78  TiffEntry *offsets = raw->getEntry(STRIPOFFSETS);
79 
80  if (offsets->count != 1) {
81  ThrowRDE("Multiple Strips found: %u", offsets->count);
82  }
83  uint32_t offset = offsets->getU32();
84  if (!mFile->isValid(offset))
85  ThrowRDE("Invalid image data offset, cannot decode.");
86 
87  mRaw->dim = iPoint2D(width, height);
88 
89  uint32_t size = mFile->getSize() - offset;
90 
91  UncompressedDecompressor u(
92  ByteStream(DataBuffer(mFile->getSubView(offset), Endianness::little)),
93  mRaw);
94 
95  if (size >= width*height*2) {
96  // It's completely unpacked little-endian
97  mRaw->createData();
98  u.decodeRawUnpacked<12, Endianness::little>(width, height);
99  } else if (size >= width*height*3/2) {
100  // It's a packed format
101  mRaw->createData();
102  u.decode12BitRaw<Endianness::little, false, true>(width, height);
103  } else {
104  uint32_t section_split_offset = 0;
105  PanasonicDecompressorV4 p(
106  mRaw,
107  ByteStream(DataBuffer(mFile->getSubView(offset), Endianness::little)),
108  hints.has("zero_is_not_bad"), section_split_offset);
109  mRaw->createData();
110  p.decompress();
111  }
112  } else {
113  mRaw->dim = iPoint2D(width, height);
114 
115  TiffEntry *offsets = raw->getEntry(PANASONIC_STRIPOFFSET);
116 
117  if (offsets->count != 1) {
118  ThrowRDE("Multiple Strips found: %u", offsets->count);
119  }
120 
121  uint32_t offset = offsets->getU32();
122 
123  ByteStream bs(DataBuffer(mFile->getSubView(offset), Endianness::little));
124 
125  uint16_t bitsPerSample = 12;
126  if (raw->hasEntry(PANASONIC_BITSPERSAMPLE))
127  bitsPerSample = raw->getEntry(PANASONIC_BITSPERSAMPLE)->getU16();
128 
129  switch (uint16_t version = raw->getEntry(PANASONIC_RAWFORMAT)->getU16()) {
130  case 4: {
131  uint32_t section_split_offset = 0x1FF8;
132  PanasonicDecompressorV4 p(mRaw, bs, hints.has("zero_is_not_bad"),
133  section_split_offset);
134  mRaw->createData();
135  p.decompress();
136  return mRaw;
137  }
138  case 5: {
139  PanasonicDecompressorV5 v5(mRaw, bs, bitsPerSample);
140  mRaw->createData();
141  v5.decompress();
142  return mRaw;
143  }
144  case 6: {
145  PanasonicDecompressorV6 v6(mRaw, bs);
146  mRaw->createData();
147  v6.decompress();
148  return mRaw;
149  }
150  default:
151  ThrowRDE("Version %i is unsupported", version);
152  }
153  }
154 
155  return mRaw;
156 }

164 void Rw2Decoder::parseCFA() {
165  if (!mRootIFD->hasEntryRecursive(PANASONIC_CFAPATTERN))
166  ThrowRDE("No PANASONIC_CFAPATTERN entry found!");
167 
168  TiffEntry* CFA = mRootIFD->getEntryRecursive(PANASONIC_CFAPATTERN);
169  if (CFA->type != TiffDataType::TIFF_SHORT || CFA->count != 1) {
170  ThrowRDE("Bad PANASONIC_CFAPATTERN entry (type %u, count %u).", CFA->type,
171  CFA->count);
172  }
173 
174  switch (auto i = CFA->getU16()) {
175  case 1:
176  mRaw->cfa.setCFA(iPoint2D(2, 2), CFA_RED, CFA_GREEN, CFA_GREEN, CFA_BLUE);
177  break;
178  case 2:
179  mRaw->cfa.setCFA(iPoint2D(2, 2), CFA_GREEN, CFA_RED, CFA_BLUE, CFA_GREEN);
180  break;
181  case 3:
182  mRaw->cfa.setCFA(iPoint2D(2, 2), CFA_GREEN, CFA_BLUE, CFA_RED, CFA_GREEN);
183  break;
184  case 4:
185  mRaw->cfa.setCFA(iPoint2D(2, 2), CFA_BLUE, CFA_GREEN, CFA_GREEN, CFA_RED);
186  break;
187  default:
188  ThrowRDE("Unexpected CFA pattern: %u", i);
189  }
190 }

192 void Rw2Decoder::decodeMetaDataInternal(const CameraMetaData* meta) {
193  parseCFA();
194 
195  auto id = mRootIFD->getID();
196  string mode = guessMode();
197  int iso = 0;
198  if (mRootIFD->hasEntryRecursive(PANASONIC_ISO_SPEED))
199  iso = mRootIFD->getEntryRecursive(PANASONIC_ISO_SPEED)->getU32();
200 
201  if (this->checkCameraSupported(meta, id, mode)) {
202  setMetaData(meta, id, mode, iso);
203  } else {
204  mRaw->metadata.mode = mode;
205  writeLog(DEBUG_PRIO_EXTRA, "Mode not found in DB: %s", mode.c_str());
206  setMetaData(meta, id, "", iso);
207  }
208 
209  const TiffIFD* raw = mRootIFD->hasEntryRecursive(PANASONIC_STRIPOFFSET)
210  ? mRootIFD->getIFDWithTag(PANASONIC_STRIPOFFSET)
211  : mRootIFD->getIFDWithTag(STRIPOFFSETS);
212 
213  // Read blacklevels
214  if (raw->hasEntry(static_cast<TiffTag>(0x1c)) &&
215  raw->hasEntry(static_cast<TiffTag>(0x1d)) &&
216  raw->hasEntry(static_cast<TiffTag>(0x1e))) {
217  const auto getBlack = [&raw](TiffTag t) -> int {
218  const auto val = raw->getEntry(t)->getU32();
219  int out;
220  if (__builtin_sadd_overflow(val, 15, &out))
221  ThrowRDE("Integer overflow when calculating black level");
222  return out;
223  };
224 
225  const int blackRed = getBlack(static_cast<TiffTag>(0x1c));
226  const int blackGreen = getBlack(static_cast<TiffTag>(0x1d));
227  const int blackBlue = getBlack(static_cast<TiffTag>(0x1e));
228 
229  for(int i = 0; i < 2; i++) {
230  for(int j = 0; j < 2; j++) {
231  const int k = i + 2 * j;
232  const CFAColor c = mRaw->cfa.getColorAt(i, j);
233  switch (c) {
234  case CFA_RED:
235  mRaw->blackLevelSeparate[k] = blackRed;
236  break;
237  case CFA_GREEN:
238  mRaw->blackLevelSeparate[k] = blackGreen;
239  break;
240  case CFA_BLUE:
241  mRaw->blackLevelSeparate[k] = blackBlue;
242  break;
243  default:
244  ThrowRDE("Unexpected CFA color %s.",
245  ColorFilterArray::colorToString(c).c_str());
246  }
247  }
248  }
249  }
250 
251  // Read WB levels
252  if (raw->hasEntry(static_cast<TiffTag>(0x0024)) &&
253  raw->hasEntry(static_cast<TiffTag>(0x0025)) &&
254  raw->hasEntry(static_cast<TiffTag>(0x0026))) {
255  mRaw->metadata.wbCoeffs[0] = static_cast<float>(
256  raw->getEntry(static_cast<TiffTag>(0x0024))->getU16());
257  mRaw->metadata.wbCoeffs[1] = static_cast<float>(
258  raw->getEntry(static_cast<TiffTag>(0x0025))->getU16());
259  mRaw->metadata.wbCoeffs[2] = static_cast<float>(
260  raw->getEntry(static_cast<TiffTag>(0x0026))->getU16());
261  } else if (raw->hasEntry(static_cast<TiffTag>(0x0011)) &&
262  raw->hasEntry(static_cast<TiffTag>(0x0012))) {
263  mRaw->metadata.wbCoeffs[0] = static_cast<float>(
264  raw->getEntry(static_cast<TiffTag>(0x0011))->getU16());
265  mRaw->metadata.wbCoeffs[1] = 256.0F;
266  mRaw->metadata.wbCoeffs[2] = static_cast<float>(
267  raw->getEntry(static_cast<TiffTag>(0x0012))->getU16());
268  }
269 }

271 std::string Rw2Decoder::guessMode() {
272  float ratio = 3.0F / 2.0F; // Default
273 
274  if (!mRaw->isAllocated())
275  return "";
276 
277  ratio = static_cast<float>(mRaw->dim.x) / static_cast<float>(mRaw->dim.y);
278 
279  float min_diff = fabs(ratio - 16.0F / 9.0F);
280  std::string closest_match = "16:9";
281 
282  float t = fabs(ratio - 3.0F / 2.0F);
283  if (t < min_diff) {
284  closest_match = "3:2";
285  min_diff = t;
286  }
287 
288  t = fabs(ratio - 4.0F / 3.0F);
289  if (t < min_diff) {
290  closest_match = "4:3";
291  min_diff = t;
292  }
293 
294  t = fabs(ratio - 1.0F);
295  if (t < min_diff) {
296  closest_match = "1:1";
297  min_diff = t;
298  }
299  writeLog(DEBUG_PRIO_EXTRA, "Mode guess: '%s'", closest_match.c_str());
300  return closest_match;
301 }
