86 void LJpegDecompressor::decodeScan()
87 {
88  assert(frame.cps > 0);
89 
90  if (predictorMode != 1)
91  ThrowRDE("Unsupported predictor mode: %u", predictorMode);
92 
93  for (uint32_t i = 0; i < frame.cps; i++)
94  if (frame.compInfo[i].superH != 1 || frame.compInfo[i].superV != 1)
95  ThrowRDE("Unsupported subsampling");
96 
97  assert(static_cast<unsigned>(mRaw->dim.x) > offX);
98  if ((mRaw->getCpp() * (mRaw->dim.x - offX)) < frame.cps)
99  ThrowRDE("Got less pixels than the components per sample");
100 
101  // How many output pixels are we expected to produce, as per DNG tiling?
102  const auto tileRequiredWidth = mRaw->getCpp() * w;
103 
104  // How many full pixel blocks do we need to consume for that?
105  const auto blocksToConsume = roundUpDivision(tileRequiredWidth, frame.cps);
106  if (frame.w < blocksToConsume || frame.h < h) {
107  ThrowRDE("LJpeg frame (%u, %u) is smaller than expected (%u, %u)",
108  frame.cps * frame.w, frame.h, tileRequiredWidth, h);
109  }
110 
111  // How many full pixel blocks will we produce?
112  fullBlocks = tileRequiredWidth / frame.cps; // Truncating division!
113  // Do we need to also produce part of a block?
114  trailingPixels = tileRequiredWidth % frame.cps;
115 
116  if (trailingPixels == 0) {
117  switch (frame.cps) {
118  case 1:
119  decodeN<1>();
120  break;
121  case 2:
122  decodeN<2>();
123  break;
124  case 3:
125  decodeN<3>();
126  break;
127  case 4:
128  decodeN<4>();
129  break;
130  default:
131  ThrowRDE("Unsupported number of components: %u", frame.cps);
132  }
133  } else /* trailingPixels != 0 */ {
134  // FIXME: using different function just for one tile likely causes
135  // i-cache misses and whatnot. Need to check how not splitting it into
136  // two different functions affects performance of the normal case.
137  switch (frame.cps) {
138  // Naturally can't happen for CPS=1.
139  case 2:
140  decodeN<2, /*WeirdWidth=*/true>();
141  break;
142  case 3:
143  decodeN<3, /*WeirdWidth=*/true>();
144  break;
145  case 4:
146  decodeN<4, /*WeirdWidth=*/true>();
147  break;
148  default:
149  ThrowRDE("Unsupported number of components: %u", frame.cps);
150  }
151  }
152 }

156 template <int N_COMP, bool WeirdWidth> void LJpegDecompressor::decodeN() {
157  assert(mRaw->getCpp() > 0);
158  assert(N_COMP > 0);
159  assert(N_COMP >= mRaw->getCpp());
160  assert((N_COMP / mRaw->getCpp()) > 0);
161 
162  assert(mRaw->dim.x >= N_COMP);
163  assert((mRaw->getCpp() * (mRaw->dim.x - offX)) >= N_COMP);
164 
165  auto ht = getHuffmanTables<N_COMP>();
166  auto pred = getInitialPredictors<N_COMP>();
167  auto predNext = pred.data();
168 
169  BitPumpJPEG bitStream(input);
170 
171  // A recoded DNG might be split up into tiles of self contained LJpeg blobs.
172  // The tiles at the bottom and the right may extend beyond the dimension of
173  // the raw image buffer. The excessive content has to be ignored.
174 
175  assert(frame.h >= h);
176  assert(frame.cps * frame.w >= mRaw->getCpp() * w);
177 
178  assert(offY + h <= static_cast<unsigned>(mRaw->dim.y));
179  assert(offX + w <= static_cast<unsigned>(mRaw->dim.x));
180 
181  // For y, we can simply stop decoding when we reached the border.
182  for (unsigned y = 0; y < h; ++y) {
183  auto destY = offY + y;
184  auto* dest =
185  reinterpret_cast<uint16_t*>(mRaw->getDataUncropped(offX, destY));
186 
187  copy_n(predNext, N_COMP, pred.data());
188  // the predictor for the next line is the start of this line
189  predNext = dest;
190 
191  unsigned x = 0;
192 
193  // FIXME: predictor may have value outside of the uint16_t.
194  // https://github.com/darktable-org/rawspeed/issues/175
195 
196  // For x, we first process all full pixel blocks within the image buffer ...
197  for (; x < fullBlocks; ++x) {
198  unroll_loop<N_COMP>([&](int i) {
199  pred[i] = uint16_t(pred[i] + ht[i]->decodeDifference(bitStream));
200  *dest++ = pred[i];
201  });
202  }
203 
204  // Sometimes we also need to consume one more block, and produce part of it.
205  if /*constexpr*/ (WeirdWidth) {
206  // FIXME: evaluate i-cache implications due to this being compile-time.
207  static_assert(N_COMP > 1 || !WeirdWidth,
208  "can't want part of 1-pixel-wide block");
209  // Some rather esoteric DNG's have odd dimensions, e.g. width % 2 = 1.
210  // We may end up needing just part of last N_COMP pixels.
211  assert(trailingPixels > 0);
212  assert(trailingPixels < N_COMP);
213  unsigned c = 0;
214  for (; c < trailingPixels; ++c) {
215  pred[c] = uint16_t(pred[c] + ht[c]->decodeDifference(bitStream));
216  *dest++ = pred[c];
217  }
218  // Discard the rest of the block.
219  assert(c < N_COMP);
220  for (; c < N_COMP; ++c) {
221  ht[c]->decodeDifference(bitStream);
222  }
223  ++x; // We did just process one more block.
224  }
225 
226  // ... and discard the rest.
227  for (; x < frame.w; ++x) {
228  unroll_loop<N_COMP>([&](int i) { ht[i]->decodeDifference(bitStream); });
229  }
230  }
231 }
