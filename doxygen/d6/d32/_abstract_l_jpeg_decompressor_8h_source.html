110 /*
111 * The following structure stores basic information about one component.
112 */
113 struct JpegComponentInfo {
114  /*
115  * These values are fixed over the whole image.
116  * They are read from the SOF marker.
117  */
118  uint32_t componentId = ~0U; /* identifier for this component (0..255) */
119 
120  /*
121  * Huffman table selector (0..3). The value may vary
122  * between scans. It is read from the SOS marker.
123  */
124  uint32_t dcTblNo = ~0U;
125  uint32_t superH = ~0U; // Horizontal Supersampling
126  uint32_t superV = ~0U; // Vertical Supersampling
127 };

162  template <int N_COMP>
163  std::array<HuffmanTable*, N_COMP> getHuffmanTables() const {
164  std::array<HuffmanTable*, N_COMP> ht;
165  for (int i = 0; i < N_COMP; ++i) {
166  const unsigned dcTblNo = frame.compInfo[i].dcTblNo;
167  const unsigned dcTbls = huff.size();
168  if (dcTblNo >= dcTbls) {
169  ThrowRDE("Decoding table %u for comp %i does not exist (tables = %u)",
170  dcTblNo, i, dcTbls);
171  }
172  ht[i] = huff[dcTblNo];
173  }
174 
175  return ht;
176  }
177 
178  template <int N_COMP>
179  __attribute__((pure)) std::array<uint16_t, N_COMP>
180  getInitialPredictors() const {
181  std::array<uint16_t, N_COMP> pred;
182  if (frame.prec < (Pt + 1)) {
183  ThrowRDE("Invalid precision (%u) and point transform (%u) combination!",
184  frame.prec, Pt);
185  }
186  pred.fill(1 << (frame.prec - Pt - 1));
187  return pred;
188  }
