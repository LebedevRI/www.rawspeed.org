46 TiffEntry::TiffEntry(TiffIFD* parent_, ByteStream* bs)
47  : parent(parent_), tag(static_cast<TiffTag>(bs->getU16())) {
48  const uint16_t numType = bs->getU16();
49  if (numType > TIFF_OFFSET)
50  ThrowTPE("Error reading TIFF structure. Unknown Type 0x%x encountered.", numType);
51  type = static_cast<TiffDataType>(numType);
52  count = bs->getU32();
53 
54  // check for count << datashift overflow
55  if (count > UINT32_MAX >> datashifts[type])
56  ThrowTPE("integer overflow in size calculation.");
57 
58  uint32_t byte_size = count << datashifts[type];
59  uint32_t data_offset = UINT32_MAX;
60 
61  if (byte_size <= 4) {
62  data_offset = bs->getPosition();
63  data = bs->getSubStream(bs->getPosition(), byte_size);
64  bs->skipBytes(4);
65  } else {
66  data_offset = bs->getU32();
67  if (type == TIFF_OFFSET || isIn(tag, {DNGPRIVATEDATA, MAKERNOTE, MAKERNOTE_ALT, FUJI_RAW_IFD, SUBIFDS, EXIFIFDPOINTER})) {
68  // preserve offset for SUB_IFD/EXIF/MAKER_NOTE data
69 #if 0
70  // limit access to range from 0 to data_offset+byte_size
71  data = ByteStream(bs, data_offset, byte_size, bs.getByteOrder());
72 #else
73  // allow access to whole file, necessary if offsets inside the maker note
74  // point to outside data, which is forbidden due to the TIFF/DNG spec but
75  // may happen none the less (see e.g. "old" ORF files like EX-1, note:
76  // the tags outside of the maker note area are currently not used anyway)
77  data = *bs;
78  data.setPosition(data_offset);
79  data.check(byte_size);
80 #endif
81  } else {
82  data = bs->getSubStream(data_offset, byte_size);
83  }
84  }
85 }

179 float TiffEntry::getFloat(uint32_t index) const {
180  if (!isFloat()) {
181  ThrowTPE("Wrong type 0x%x encountered. Expected Float or something "
182  "convertible on 0x%x",
183  type, tag);
184  }
185 
186  switch (type) {
187  case TIFF_DOUBLE: return data.peek<double>(index);
188  case TIFF_FLOAT: return data.peek<float>(index);
189  case TIFF_LONG:
190  case TIFF_SHORT:
191  return static_cast<float>(getU32(index));
192  case TIFF_SLONG:
193  case TIFF_SSHORT:
194  return static_cast<float>(getI32(index));
195  case TIFF_RATIONAL: {
196  uint32_t a = getU32(index * 2);
197  uint32_t b = getU32(index * 2 + 1);
198  return b != 0 ? static_cast<float>(a) / b : 0.0F;
199  }
200  case TIFF_SRATIONAL: {
201  auto a = static_cast<int>(getU32(index * 2));
202  auto b = static_cast<int>(getU32(index * 2 + 1));
203  return b ? static_cast<float>(a) / b : 0.0F;
204  }
205  default:
206  // unreachable
207  return 0.0F;
208  }
209 }

211 string TiffEntry::getString() const {
212  if (type != TIFF_ASCII && type != TIFF_BYTE)
213  ThrowTPE("Wrong type 0x%x encountered. Expected Ascii or Byte", type);
214 
215  // *NOT* ByteStream::peekString() !
216  const auto bufSize = data.getRemainSize();
217  const auto* buf = data.peekData(bufSize);
218  const auto* s = reinterpret_cast<const char*>(buf);
219  return string(s, strnlen(s, bufSize));
