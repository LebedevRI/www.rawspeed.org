275 // setWithLookUp will set a single pixel by using the lookup table if supplied,
276 // You must supply the destination where the value should be written, and a pointer to
277 // a value that will be used to store a random counter that can be reused between calls.
278 // this needs to be inline to speed up tight decompressor loops
279 inline void RawImageDataU16::setWithLookUp(uint16_t value, uint8_t* dst,
280  uint32_t* random) {
281  auto* dest = reinterpret_cast<uint16_t*>(dst);
282  if (table == nullptr) {
283  *dest = value;
284  return;
285  }
286  if (table->dither) {
287  const auto* t = reinterpret_cast<const uint32_t*>(table->tables.data());
288  uint32_t lookup = t[value];
289  uint32_t base = lookup & 0xffff;
290  uint32_t delta = lookup >> 16;
291  uint32_t r = *random;
292 
293  uint32_t pix = base + ((delta * (r & 2047) + 1024) >> 12);
294  *random = 15700 *(r & 65535) + (r >> 16);
295  *dest = pix;
296  return;
297  }
298  *dest = table->tables[value];
299 }
