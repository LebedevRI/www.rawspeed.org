47

55  }
56 #endif

65  std::string found_camfile(bindir +
66  "/../share/darktable/rawspeed/cameras.xml");

74  found_camfile.c_str());
75  found_camfile =
76  bindir + "/../Resources/share/darktable/rawspeed/cameras.xml";
77  if (stat(found_camfile.c_str(), &statbuf)) {

192  fprintf(stdout, "make: %s\n", r->metadata.make.c_str());

271  }
272 
273  fprintf(stdout, "Image float sum: %lf\n", sum);
274  fprintf(stdout, "Image float avg: %lf\n",

288  }

293  }

299  return 2;

301
