368 template <int iterations>
369 static void UnrollLoopTestIsMonotonicallyPositiveTest() {
370  static_assert(iterations >= 0, "negative iteration count makes no sense.");
371 
372  int invocation = 0;
373 
374  std::vector<int> expected;
375  expected.reserve(iterations);
376  invocation = 0;
377  std::generate_n(std::back_inserter(expected), iterations,
378  [&invocation]() -> int { return invocation++; });
379  if (iterations > 0) {
380  ASSERT_EQ(expected.size(), iterations);
381  ASSERT_EQ(expected.front(), 0);
382  ASSERT_EQ(expected.back(), iterations - 1);
383  }
384 
385  std::vector<int> data;
386  data.reserve(iterations);
387  invocation = 0;
388  unroll_loop<iterations>([&](int i) {
389  ASSERT_GE(invocation, 0);
390  ASSERT_GE(i, 0);
391  ASSERT_LT(invocation, iterations);
392  ASSERT_LT(i, iterations);
393  ASSERT_EQ(i, invocation);
394 
395  data.emplace_back(i);
396  invocation++;
397  });
398 
399  ASSERT_EQ(data.size(), expected.size());
400  ASSERT_EQ(data, expected);
401 }
