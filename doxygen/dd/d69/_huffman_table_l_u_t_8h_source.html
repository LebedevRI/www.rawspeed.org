71 class HuffmanTableLUT final : public HuffmanTableLookup {
72  // The code can be compiled with two different decode lookup table layouts.
73  // The idea is that different CPU architectures may perform better with
74  // one or the other, depending on the relative performance of their arithmetic
75  // core vs their memory access. For an Intel Core i7, the big table is better.
76 #if 1
77  // lookup table containing 3 fields: payload:16|flag:8|len:8
78  // The payload may be the fully decoded diff or the length of the diff.
79  // The len field contains the number of bits, this lookup consumed.
80  // A lookup value of 0 means the code was too big to fit into the table.
81  // The optimal LookupDepth is also likely to depend on the CPU architecture.
82  static constexpr unsigned PayloadShift = 16;
83  static constexpr unsigned FlagMask = 0x100;
84  static constexpr unsigned LenMask = 0xff;
85  static constexpr unsigned LookupDepth = 11;
86  std::vector<int32_t> decodeLookup;
87 #else
88  // lookup table containing 2 fields: payload:4|len:4
89  // the payload is the length of the diff, len is the length of the code
90  static constexpr unsigned LookupDepth = 15;
91  static constexpr unsigned PayloadShift = 4;
92  static constexpr unsigned FlagMask = 0;
93  static constexpr unsigned LenMask = 0x0f;
94  std::vector<uint8_t> decodeLookup;
95 #endif
96 
97 public:
98  void setup(bool fullDecode_, bool fixDNGBug16_) {
99  const std::vector<CodeSymbol> symbols =
100  HuffmanTableLookup::setup(fullDecode_, fixDNGBug16_);
101 
102  // Generate lookup table for fast decoding lookup.
103  // See definition of decodeLookup above
104  decodeLookup.resize(1 << LookupDepth);
105  for (size_t i = 0; i < symbols.size(); i++) {
106  uint8_t code_l = symbols[i].code_len;
107  if (code_l > static_cast<int>(LookupDepth))
108  break;
109 
110  uint16_t ll = symbols[i].code << (LookupDepth - code_l);
111  uint16_t ul = ll | ((1 << (LookupDepth - code_l)) - 1);
112  uint16_t diff_l = codeValues[i];
113  for (uint16_t c = ll; c <= ul; c++) {
114  if (!(c < decodeLookup.size()))
115  ThrowRDE("Corrupt Huffman");
116 
117  if (!FlagMask || !fullDecode || code_l > LookupDepth ||
118  (code_l + diff_l > LookupDepth && diff_l != 16)) {
119  // lookup bit depth is too small to fit both the encoded length
120  // and the final difference value.
121  // -> store only the length and do a normal sign extension later
122  assert(!fullDecode || diff_l > 0);
123  decodeLookup[c] = diff_l << PayloadShift | code_l;
124 
125  if (!fullDecode)
126  decodeLookup[c] |= FlagMask;
127  } else {
128  // Lookup bit depth is sufficient to encode the final value.
129  decodeLookup[c] = FlagMask | code_l;
130  if (diff_l != 16 || fixDNGBug16)
131  decodeLookup[c] += diff_l;
132 
133  if (diff_l) {
134  uint32_t diff;
135  if (diff_l != 16) {
136  diff = extractHighBits(c, code_l + diff_l,
137  /*effectiveBitwidth=*/LookupDepth);
138  diff &= ((1 << diff_l) - 1);
139  } else
140  diff = uint32_t(-32768);
141  decodeLookup[c] |= static_cast<int32_t>(
142  static_cast<uint32_t>(extend(diff, diff_l)) << PayloadShift);
143  }
144  }
145  }
146  }
147  }

158  template <typename BIT_STREAM>
159  inline __attribute__((always_inline)) int
160  decodeDifference(BIT_STREAM& bs) const {
161  static_assert(BitStreamTraits<BIT_STREAM>::canUseWithHuffmanTable,
162  "This BitStream specialization is not marked as usable here");
163  assert(fullDecode);
164  return decode<BIT_STREAM, true>(bs);
165  }
166 
167  // The bool template paraeter is to enable two versions:
168  // one returning only the length of the of diff bits (see Hasselblad),
169  // one to return the fully decoded diff.
170  // All ifs depending on this bool will be optimized out by the compiler
171  template <typename BIT_STREAM, bool FULL_DECODE>
172  inline __attribute__((always_inline)) int decode(BIT_STREAM& bs) const {
173  static_assert(BitStreamTraits<BIT_STREAM>::canUseWithHuffmanTable,
174  "This BitStream specialization is not marked as usable here");
175  assert(FULL_DECODE == fullDecode);
176  bs.fill(32);
177 
178  CodeSymbol partial;
179  partial.code_len = LookupDepth;
180  partial.code = bs.peekBitsNoFill(partial.code_len);
181 
182  assert(partial.code < decodeLookup.size());
183  auto lutEntry = static_cast<unsigned>(decodeLookup[partial.code]);
184  int payload = static_cast<int>(lutEntry) >> PayloadShift;
185  int len = lutEntry & LenMask;
186 
187  // How far did reading of those LookupDepth bits *actually* move us forward?
188  bs.skipBitsNoFill(len);
189 
190  // If the flag bit is set, then we have already skipped all the len bits
191  // we needed to skip, and payload is the answer we were looking for.
192  if (lutEntry & FlagMask)
193  return payload;
194 
195  int codeValue;
196  if (lutEntry) {
197  // If the flag is not set, but the entry is not empty,
198  // the payload is the code value for this symbol.
199  partial.code_len = len;
200  codeValue = payload;
201  assert(!FULL_DECODE || codeValue /*aka diff_l*/ > 0);
202  } else {
203  // No match in the lookup table, because either the code is longer
204  // than LookupDepth or the input is corrupt. Need to read more bits...
205  assert(len == 0);
206  bs.skipBitsNoFill(partial.code_len);
207  std::tie(partial, codeValue) = finishReadingPartialSymbol(bs, partial);
208  }
209 
210  return processSymbol<BIT_STREAM, FULL_DECODE>(bs, partial, codeValue);
211  }
212 };
