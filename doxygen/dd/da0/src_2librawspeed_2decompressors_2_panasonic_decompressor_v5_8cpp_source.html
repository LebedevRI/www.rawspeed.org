62 PanasonicDecompressorV5::PanasonicDecompressorV5(const RawImage& img,
63  const ByteStream& input_,
64  uint32_t bps_)
65  : mRaw(img), bps(bps_) {
66  if (mRaw->getCpp() != 1 || mRaw->getDataType() != TYPE_USHORT16 ||
67  mRaw->getBpp() != sizeof(uint16_t))
68  ThrowRDE("Unexpected component count / data type");
69 
70  const PacketDsc* dsc = nullptr;
71  switch (bps) {
72  case 12:
73  dsc = &TwelveBitPacket;
74  break;
75  case 14:
76  dsc = &FourteenBitPacket;
77  break;
78  default:
79  ThrowRDE("Unsupported bps: %u", bps);
80  }
81 
82  if (!mRaw->dim.hasPositiveArea() || mRaw->dim.x % dsc->pixelsPerPacket != 0) {
83  ThrowRDE("Unexpected image dimensions found: (%i; %i)", mRaw->dim.x,
84  mRaw->dim.y);
85  }
86 
87  // How many pixel packets does the specified pixel count require?
88  assert(mRaw->dim.area() % dsc->pixelsPerPacket == 0);
89  const auto numPackets = mRaw->dim.area() / dsc->pixelsPerPacket;
90  assert(numPackets > 0);
91 
92  // And how many blocks that would be? Last block may not be full, pad it.
93  numBlocks = roundUpDivision(numPackets, PacketsPerBlock);
94  assert(numBlocks > 0);
95 
96  // How many full blocks does the input contain? This is truncating division.
97  const auto haveBlocks = input_.getRemainSize() / BlockSize;
98 
99  // Does the input contain enough blocks?
100  if (haveBlocks < numBlocks)
101  ThrowRDE("Insufficient count of input blocks for a given image");
102 
103  // We only want those blocks we need, no extras.
104  input = input_.peekStream(numBlocks, BlockSize);
105 
106  chopInputIntoBlocks(*dsc);
107 }

109 void PanasonicDecompressorV5::chopInputIntoBlocks(const PacketDsc& dsc) {
110  auto pixelToCoordinate = [width = mRaw->dim.x](unsigned pixel) {
111  return iPoint2D(pixel % width, pixel / width);
112  };
113 
114  assert(numBlocks * BlockSize == input.getRemainSize());
115  blocks.reserve(numBlocks);
116 
117  const auto pixelsPerBlock = dsc.pixelsPerPacket * PacketsPerBlock;
118  assert((numBlocks - 1U) * pixelsPerBlock < mRaw->dim.area());
119  assert(numBlocks * pixelsPerBlock >= mRaw->dim.area());
120 
121  unsigned currPixel = 0;
122  std::generate_n(std::back_inserter(blocks), numBlocks,
123  [input = &input, &currPixel, pixelToCoordinate,
124  pixelsPerBlock]() -> Block {
125  ByteStream bs = input->getStream(BlockSize);
126  iPoint2D beginCoord = pixelToCoordinate(currPixel);
127  currPixel += pixelsPerBlock;
128  iPoint2D endCoord = pixelToCoordinate(currPixel);
129  return {std::move(bs), beginCoord, endCoord};
130  });
131  assert(blocks.size() == numBlocks);
132  assert(currPixel >= mRaw->dim.area());
133  assert(input.getRemainSize() == 0);
134 
135  // Clamp the end coordinate for the last block.
136  blocks.back().endCoord = mRaw->dim;
137  blocks.back().endCoord.y -= 1;
138 }

140 class PanasonicDecompressorV5::ProxyStream {
141  ByteStream block;
142  std::vector<uint8_t> buf;
143  ByteStream input;
144 
145  void parseBlock() {
146  assert(buf.empty());
147  assert(block.getRemainSize() == BlockSize);
148 
149  static_assert(BlockSize > sectionSplitOffset, "");
150 
151  Buffer FirstSection = block.getBuffer(sectionSplitOffset);
152  Buffer SecondSection = block.getBuffer(block.getRemainSize());
153  assert(FirstSection.getSize() < SecondSection.getSize());
154 
155  buf.reserve(BlockSize);
156 
157  // First copy the second section. This makes it the first section.
158  buf.insert(buf.end(), SecondSection.begin(), SecondSection.end());
159  // Now append the original 1'st section right after the new 1'st section.
160  buf.insert(buf.end(), FirstSection.begin(), FirstSection.end());
161 
162  assert(buf.size() == BlockSize);
163  assert(block.getRemainSize() == 0);
164 
165  // And reset the clock.
166  input = ByteStream(
167  DataBuffer(Buffer(buf.data(), buf.size()), Endianness::little));
168  // input.setByteOrder(Endianness::big); // does not seem to matter?!
169  }
170 
171 public:
172  explicit ProxyStream(ByteStream block_) : block(std::move(block_)) {}
173 
174  ByteStream& getStream() {
175  parseBlock();
176  return input;
177  }
178 };

180 template <const PanasonicDecompressorV5::PacketDsc& dsc>
181 inline void PanasonicDecompressorV5::processPixelPacket(BitPumpLSB* bs, int row,
182  int col) const {
183  static_assert(dsc.pixelsPerPacket > 0, "dsc should be compile-time const");
184  static_assert(dsc.bps > 0 && dsc.bps <= 16, "");
185 
186  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
187 
188  assert(bs->getFillLevel() == 0);
189 
190  for (int p = 0; p < dsc.pixelsPerPacket;) {
191  bs->fill();
192  for (; bs->getFillLevel() >= dsc.bps; ++p, ++col)
193  out(row, col) = bs->getBitsNoFill(dsc.bps);
194  }
195  bs->skipBitsNoFill(bs->getFillLevel()); // get rid of padding.
196 }

198 template <const PanasonicDecompressorV5::PacketDsc& dsc>
199 void PanasonicDecompressorV5::processBlock(const Block& block) const {
200  static_assert(dsc.pixelsPerPacket > 0, "dsc should be compile-time const");
201  static_assert(BlockSize % bytesPerPacket == 0, "");
202 
203  ProxyStream proxy(block.bs);
204  BitPumpLSB bs(proxy.getStream());
205 
206  for (int row = block.beginCoord.y; row <= block.endCoord.y; row++) {
207  int col = 0;
208  // First row may not begin at the first column.
209  if (block.beginCoord.y == row)
210  col = block.beginCoord.x;
211 
212  int endx = mRaw->dim.x;
213  // Last row may end before the last column.
214  if (block.endCoord.y == row)
215  endx = block.endCoord.x;
216 
217  assert(col % dsc.pixelsPerPacket == 0);
218  assert(endx % dsc.pixelsPerPacket == 0);
219 
220  for (; col < endx; col += dsc.pixelsPerPacket)
221  processPixelPacket<dsc>(&bs, row, col);
222  }
223 }
