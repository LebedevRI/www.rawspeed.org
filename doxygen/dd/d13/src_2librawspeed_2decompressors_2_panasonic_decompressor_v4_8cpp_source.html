46 PanasonicDecompressorV4::PanasonicDecompressorV4(const RawImage& img,
47  const ByteStream& input_,
48  bool zero_is_not_bad,
49  uint32_t section_split_offset_)
50  : mRaw(img), zero_is_bad(!zero_is_not_bad),
51  section_split_offset(section_split_offset_) {
52  if (mRaw->getCpp() != 1 || mRaw->getDataType() != TYPE_USHORT16 ||
53  mRaw->getBpp() != sizeof(uint16_t))
54  ThrowRDE("Unexpected component count / data type");
55 
56  if (!mRaw->dim.hasPositiveArea() || mRaw->dim.x % PixelsPerPacket != 0) {
57  ThrowRDE("Unexpected image dimensions found: (%i; %i)", mRaw->dim.x,
58  mRaw->dim.y);
59  }
60 
61  if (BlockSize < section_split_offset)
62  ThrowRDE("Bad section_split_offset: %u, less than BlockSize (%u)",
63  section_split_offset, BlockSize);
64 
65  // Naive count of bytes that given pixel count requires.
66  assert(mRaw->dim.area() % PixelsPerPacket == 0);
67  const auto bytesTotal = (mRaw->dim.area() / PixelsPerPacket) * BytesPerPacket;
68  assert(bytesTotal > 0);
69 
70  // If section_split_offset is zero, then that we need to read the normal
71  // amount of bytes. But if it is not, then we need to round up to multiple of
72  // BlockSize, because of splitting&rotation of each BlockSize's slice in half
73  // at section_split_offset bytes.
74  const auto bufSize =
75  section_split_offset == 0 ? bytesTotal : roundUp(bytesTotal, BlockSize);
76 
77  if (bufSize > std::numeric_limits<ByteStream::size_type>::max())
78  ThrowRDE("Raw dimensions require input buffer larger than supported");
79 
80  input = input_.peekStream(bufSize);
81 
82  chopInputIntoBlocks();
83 }

85 void PanasonicDecompressorV4::chopInputIntoBlocks() {
86  auto pixelToCoordinate = [width = mRaw->dim.x](unsigned pixel) {
87  return iPoint2D(pixel % width, pixel / width);
88  };
89 
90  // If section_split_offset == 0, last block may not be full.
91  const auto blocksTotal = roundUpDivision(input.getRemainSize(), BlockSize);
92  assert(blocksTotal > 0);
93  assert(blocksTotal * PixelsPerBlock >= mRaw->dim.area());
94  blocks.reserve(blocksTotal);
95 
96  unsigned currPixel = 0;
97  std::generate_n(std::back_inserter(blocks), blocksTotal,
98  [input = &input, &currPixel, pixelToCoordinate]() -> Block {
99  assert(input->getRemainSize() != 0);
100  const auto blockSize =
101  std::min(input->getRemainSize(), BlockSize);
102  assert(blockSize > 0);
103  assert(blockSize % BytesPerPacket == 0);
104  const auto packets = blockSize / BytesPerPacket;
105  assert(packets > 0);
106  const auto pixels = packets * PixelsPerPacket;
107  assert(pixels > 0);
108 
109  ByteStream bs = input->getStream(blockSize);
110  iPoint2D beginCoord = pixelToCoordinate(currPixel);
111  currPixel += pixels;
112  iPoint2D endCoord = pixelToCoordinate(currPixel);
113  return {std::move(bs), beginCoord, endCoord};
114  });
115  assert(blocks.size() == blocksTotal);
116  assert(currPixel >= mRaw->dim.area());
117  assert(input.getRemainSize() == 0);
118 
119  // Clamp the end coordinate for the last block.
120  blocks.back().endCoord = mRaw->dim;
121  blocks.back().endCoord.y -= 1;
122 }

124 class PanasonicDecompressorV4::ProxyStream {
125  ByteStream block;
126  const uint32_t section_split_offset;
127  std::vector<uint8_t> buf;
128 
129  int vbits = 0;
130 
131  void parseBlock() {
132  assert(buf.empty());
133  assert(block.getRemainSize() <= BlockSize);
134  assert(section_split_offset <= BlockSize);
135 
136  Buffer FirstSection = block.getBuffer(section_split_offset);
137  Buffer SecondSection = block.getBuffer(block.getRemainSize());
138 
139  // get one more byte, so the return statement of getBits does not have
140  // to special case for accessing the last byte
141  buf.reserve(BlockSize + 1UL);
142 
143  // First copy the second section. This makes it the first section.
144  buf.insert(buf.end(), SecondSection.begin(), SecondSection.end());
145  // Now append the original 1'st section right after the new 1'st section.
146  buf.insert(buf.end(), FirstSection.begin(), FirstSection.end());
147 
148  assert(block.getRemainSize() == 0);
149 
150  // get one more byte, so the return statement of getBits does not have
151  // to special case for accessing the last byte
152  buf.emplace_back(0);
153  }
154 
155 public:
156  ProxyStream(ByteStream block_, int section_split_offset_)
157  : block(std::move(block_)), section_split_offset(section_split_offset_) {
158  parseBlock();
159  }
160 
161  uint32_t getBits(int nbits) noexcept {
162  vbits = (vbits - nbits) & 0x1ffff;
163  int byte = vbits >> 3 ^ 0x3ff0;
164  return (buf[byte] | buf[byte + 1UL] << 8) >> (vbits & 7) & ~(-(1 << nbits));
165  }
166 };

168 inline void PanasonicDecompressorV4::processPixelPacket(
169  ProxyStream* bits, int row, int col,
170  std::vector<uint32_t>* zero_pos) const noexcept {
171  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
172 
173  int sh = 0;
174 
175  std::array<int, 2> pred;
176  pred.fill(0);
177 
178  std::array<int, 2> nonz;
179  nonz.fill(0);
180 
181  int u = 0;
182 
183  for (int p = 0; p < PixelsPerPacket; ++p, ++col) {
184  const int c = p & 1;
185 
186  // FIXME: this is actually just `p % 3 == 2`, cleanup after perf is good.
187  if (u == 2) {
188  sh = extractHighBits(4U, bits->getBits(2), /*effectiveBitwidth=*/3);
189  u = -1;
190  }
191 
192  if (nonz[c]) {
193  int j = bits->getBits(8);
194  if (j) {
195  pred[c] -= 0x80 << sh;
196  if (pred[c] < 0 || sh == 4)
197  pred[c] &= ~(-(1 << sh));
198  pred[c] += j << sh;
199  }
200  } else {
201  nonz[c] = bits->getBits(8);
202  if (nonz[c] || p > 11)
203  pred[c] = nonz[c] << 4 | bits->getBits(4);
204  }
205 
206  out(row, col) = pred[c];
207 
208  if (zero_is_bad && 0 == pred[c])
209  zero_pos->push_back((row << 16) | col);
210 
211  u++;
212  }
213 }

215 void PanasonicDecompressorV4::processBlock(
216  const Block& block, std::vector<uint32_t>* zero_pos) const noexcept {
217  ProxyStream bits(block.bs, section_split_offset);
218 
219  for (int row = block.beginCoord.y; row <= block.endCoord.y; row++) {
220  int col = 0;
221  // First row may not begin at the first column.
222  if (block.beginCoord.y == row)
223  col = block.beginCoord.x;
224 
225  int endCol = mRaw->dim.x;
226  // Last row may end before the last column.
227  if (block.endCoord.y == row)
228  endCol = block.endCoord.x;
229 
230  assert(col % PixelsPerPacket == 0);
231  assert(endCol % PixelsPerPacket == 0);
232 
233  for (; col < endCol; col += PixelsPerPacket)
234  processPixelPacket(&bits, row, col, zero_pos);
235  }
236 }

238 void PanasonicDecompressorV4::decompressThread() const noexcept {
239  std::vector<uint32_t> zero_pos;
240 
241  assert(!blocks.empty());
242 
243 #ifdef HAVE_OPENMP
244 #pragma omp for schedule(static)
245 #endif
246  for (auto block = blocks.cbegin(); block < blocks.cend(); ++block)
247  processBlock(*block, &zero_pos);
248 
249  if (zero_is_bad && !zero_pos.empty()) {
250  MutexLocker guard(&mRaw->mBadPixelMutex);
251  mRaw->mBadPixelPositions.insert(mRaw->mBadPixelPositions.end(),
252  zero_pos.begin(), zero_pos.end());
253  }
