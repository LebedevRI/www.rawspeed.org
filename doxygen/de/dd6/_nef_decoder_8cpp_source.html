66 RawImage NefDecoder::decodeRawInternal() {
67  const auto* raw = mRootIFD->getIFDWithTag(CFAPATTERN);
68  auto compression = raw->getEntry(COMPRESSION)->getU32();
69 
70  TiffEntry *offsets = raw->getEntry(STRIPOFFSETS);
71  TiffEntry *counts = raw->getEntry(STRIPBYTECOUNTS);
72 
73  if (mRootIFD->getEntryRecursive(MODEL)->getString() == "NIKON D100 ") { /**Sigh**/
74  if (!mFile->isValid(offsets->getU32()))
75  ThrowRDE("Image data outside of file.");
76  if (!D100IsCompressed(offsets->getU32())) {
77  DecodeD100Uncompressed();
78  return mRaw;
79  }
80  }
81 
82  if (compression == 1 || (hints.has("force_uncompressed")) ||
83  NEFIsUncompressed(raw)) {
84  DecodeUncompressed();
85  return mRaw;
86  }
87 
88  if (NEFIsUncompressedRGB(raw)) {
89  DecodeSNefUncompressed();
90  return mRaw;
91  }
92 
93  if (offsets->count != 1) {
94  ThrowRDE("Multiple Strips found: %u", offsets->count);
95  }
96  if (counts->count != offsets->count) {
97  ThrowRDE(
98  "Byte count number does not match strip size: count:%u, strips:%u ",
99  counts->count, offsets->count);
100  }
101  if (!mFile->isValid(offsets->getU32(), counts->getU32()))
102  ThrowRDE("Invalid strip byte count. File probably truncated.");
103 
104  if (34713 != compression)
105  ThrowRDE("Unsupported compression");
106 
107  uint32_t width = raw->getEntry(IMAGEWIDTH)->getU32();
108  uint32_t height = raw->getEntry(IMAGELENGTH)->getU32();
109  uint32_t bitPerPixel = raw->getEntry(BITSPERSAMPLE)->getU32();
110 
111  mRaw->dim = iPoint2D(width, height);
112 
113  raw = mRootIFD->getIFDWithTag(static_cast<TiffTag>(0x8c));
114 
115  TiffEntry *meta;
116  if (raw->hasEntry(static_cast<TiffTag>(0x96))) {
117  meta = raw->getEntry(static_cast<TiffTag>(0x96));
118  } else {
119  meta = raw->getEntry(static_cast<TiffTag>(0x8c)); // Fall back
120  }
121 
122  ByteStream rawData(
123  DataBuffer(mFile->getSubView(offsets->getU32(), counts->getU32()),
124  Endianness::little));
125 
126  NikonDecompressor n(mRaw, meta->getData(), bitPerPixel);
127  mRaw->createData();
128  n.decompress(rawData, uncorrectedRawValues);
129 
130  return mRaw;
131 }

149 /* At least the D810 has a broken firmware that tags uncompressed images
150  as if they were compressed. For those cases we set uncompressed mode
151  by figuring out that the image is the size of uncompressed packing */
152 bool NefDecoder::NEFIsUncompressed(const TiffIFD* raw) {
153  TiffEntry* counts = raw->getEntry(STRIPBYTECOUNTS);
154  uint32_t width = raw->getEntry(IMAGEWIDTH)->getU32();
155  uint32_t height = raw->getEntry(IMAGELENGTH)->getU32();
156  uint32_t bitPerPixel = raw->getEntry(BITSPERSAMPLE)->getU32();
157 
158  if (!width || !height || !bitPerPixel)
159  return false;
160 
161  const auto avaliableInputBytes = counts->getU32(0);
162  const auto requiredPixels = iPoint2D(width, height).area();
163 
164  // Now, there can be three situations.
165 
166  // We might have not enough input to produce the requested image size.
167  const uint64_t avaliableInputBits = uint64_t(8) * avaliableInputBytes;
168  const auto avaliablePixels = avaliableInputBits / bitPerPixel; // round down!
169  if (avaliablePixels < requiredPixels)
170  return false;
171 
172  // We might have exactly enough input with no padding whatsoever.
173  if (avaliablePixels == requiredPixels)
174  return true;
175 
176  // Or, we might have too much input. And sadly this is the worst case.
177  // We can't just accept this. Some *compressed* NEF's also pass this check :(
178  // Thus, let's accept *some* *small* padding.
179  const auto requiredInputBits = bitPerPixel * requiredPixels;
180  const auto requiredInputBytes = roundUpDivision(requiredInputBits, 8);
181  // While we might have more *pixels* than needed, it does not nessesairly mean
182  // that we have more input *bytes*. We might be off by a few pixels, and with
183  // small image dimensions and bpp, we might still be in the same byte.
184  assert(avaliableInputBytes >= requiredInputBytes);
185  const auto totalPadding = avaliableInputBytes - requiredInputBytes;
186  if (totalPadding % height != 0)
187  return false; // Inconsistent padding makes no sense here.
188  const auto perRowPadding = totalPadding / height;
189  return perRowPadding < 16;
190 }

195 bool NefDecoder::NEFIsUncompressedRGB(const TiffIFD* raw) {
196  uint32_t byteCount = raw->getEntry(STRIPBYTECOUNTS)->getU32(0);
197  uint32_t width = raw->getEntry(IMAGEWIDTH)->getU32();

202

204 }

206 void NefDecoder::DecodeUncompressed() {
207  const auto* raw = getIFDWithLargestImage(CFAPATTERN);
208  TiffEntry *offsets = raw->getEntry(STRIPOFFSETS);
209  TiffEntry *counts = raw->getEntry(STRIPBYTECOUNTS);
210  uint32_t yPerSlice = raw->getEntry(ROWSPERSTRIP)->getU32();
211  uint32_t width = raw->getEntry(IMAGEWIDTH)->getU32();
212  uint32_t height = raw->getEntry(IMAGELENGTH)->getU32();
213  uint32_t bitPerPixel = raw->getEntry(BITSPERSAMPLE)->getU32();
214 
215  mRaw->dim = iPoint2D(width, height);
216 
217  if (width == 0 || height == 0 || width > 8288 || height > 5520)
218  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
219 
220  if (counts->count != offsets->count) {
221  ThrowRDE("Byte count number does not match strip size: "
222  "count:%u, stips:%u ",
223  counts->count, offsets->count);
224  }
225 
226  if (yPerSlice == 0 || yPerSlice > static_cast<uint32_t>(mRaw->dim.y) ||
227  roundUpDivision(mRaw->dim.y, yPerSlice) != counts->count) {
228  ThrowRDE("Invalid y per slice %u or strip count %u (height = %u)",
229  yPerSlice, counts->count, mRaw->dim.y);
230  }
231 
232  vector<NefSlice> slices;
233  slices.reserve(counts->count);
234  uint32_t offY = 0;
235 
236  for (uint32_t s = 0; s < counts->count; s++) {
237  NefSlice slice;
238  slice.offset = offsets->getU32(s);
239  slice.count = counts->getU32(s);
240 
241  if (slice.count < 1)
242  ThrowRDE("Slice %u is empty", s);
243 
244  if (offY + yPerSlice > height)
245  slice.h = height - offY;
246  else
247  slice.h = yPerSlice;
248 
249  offY = min(height, offY + yPerSlice);
250 
251  if (!mFile->isValid(slice.offset, slice.count))
252  ThrowRDE("Slice offset/count invalid");
253 
254  slices.push_back(slice);
255  }
256 
257  if (slices.empty())
258  ThrowRDE("No valid slices found. File probably truncated.");
259 
260  assert(height == offY);
261  assert(slices.size() == counts->count);
262 
263  mRaw->createData();
264  if (bitPerPixel == 14 && width*slices[0].h*2 == slices[0].count)
265  bitPerPixel = 16; // D3 & D810
266 
267  bitPerPixel = hints.get("real_bpp", bitPerPixel);
268 
269  switch (bitPerPixel) {
270  case 12:
271  case 14:
272  case 16:
273  break;
274  default:
275  ThrowRDE("Invalid bpp found: %u", bitPerPixel);
276  }
277 
278  bool bitorder = ! hints.has("msb_override");
279 
280  offY = 0;
281  for (const NefSlice& slice : slices) {
282  ByteStream in(DataBuffer(mFile->getSubView(slice.offset, slice.count),
283  Endianness::little));
284  iPoint2D size(width, slice.h);
285  iPoint2D pos(0, offY);
286 
287  if (hints.has("coolpixmangled")) {
288  UncompressedDecompressor u(in, mRaw);
289  u.readUncompressedRaw(size, pos, width * bitPerPixel / 8, 12,
290  BitOrder_MSB32);
291  } else {
292  if (hints.has("coolpixsplit"))
293  readCoolpixSplitRaw(in, size, pos, width * bitPerPixel / 8);
294  else {
295  UncompressedDecompressor u(in, mRaw);
296  if (in.getSize() % size.y != 0)
297  ThrowRDE("Inconsistent row size");
298  const auto inputPitchBytes = in.getSize() / size.y;
299  u.readUncompressedRaw(size, pos, inputPitchBytes, bitPerPixel,
300  bitorder ? BitOrder_MSB : BitOrder_LSB);
301  }
302  }
303 
304  offY += slice.h;
305  }
306 }

308 void NefDecoder::readCoolpixSplitRaw(ByteStream input, const iPoint2D& size,
309  const iPoint2D& offset, int inputPitch) {
310  const Array2DRef<uint16_t> img(mRaw->getU16DataAsUncroppedArray2DRef());
311 
312  if (size.y % 2 != 0)
313  ThrowRDE("Odd number of rows");
314  if (size.x % 8 != 0)
315  ThrowRDE("Column count isn't multiple of 8");
316  if (inputPitch != ((3 * size.x) / 2))
317  ThrowRDE("Unexpected input pitch");
318 
319  // BitPumpMSB loads exactly 4 bytes at once, and we squeeze 12 bits each time.
320  // We produce 2 pixels per 3 bytes (24 bits). If we want to be smart and to
321  // know where the first input bit for first odd row is, the input slice width
322  // must be a multiple of 8 pixels.
323 
324  if (offset.x > mRaw->dim.x || offset.y > mRaw->dim.y)
325  ThrowRDE("All pixels outside of image");
326  if (offset.x + size.x > mRaw->dim.x || offset.y + size.y > mRaw->dim.y)
327  ThrowRDE("Output is partailly out of image");
328 
329  // The input bytes are laid out in the memory in the following way:
330  // First, all even (0-2-4-) rows, and then all odd (1-3-5-) rows.
331  BitPumpMSB even(input.getStream(size.y / 2, inputPitch));
332  BitPumpMSB odd(input.getStream(size.y / 2, inputPitch));
333  for (int row = offset.y; row < size.y;) {
334  for (int col = offset.x; col < size.x; ++col)
335  img(row, col) = even.getBits(12);
336  ++row;
337  for (int col = offset.x; col < size.x; ++col)
338  img(row, col) = odd.getBits(12);
339  ++row;
340  }
341  assert(even.getRemainingSize() == 0 && odd.getRemainingSize() == 0 &&
342  "Should have run out of input");
343 }

348  uint32_t offset = ifd->getEntry(STRIPOFFSETS)->getU32();

397 string NefDecoder::getMode() {
398  ostringstream mode;
399  const auto* raw = getIFDWithLargestImage(CFAPATTERN);
400  int compression = raw->getEntry(COMPRESSION)->getU32();
401  uint32_t bitPerPixel = raw->getEntry(BITSPERSAMPLE)->getU32();
402 
403  if (NEFIsUncompressedRGB(raw))
404  mode << "sNEF-uncompressed";
405  else {
406  if (1 == compression || NEFIsUncompressed(raw))
407  mode << bitPerPixel << "bit-uncompressed";
408  else
409  mode << bitPerPixel << "bit-compressed";
410  }
411  return mode.str();
412 }

612 // DecodeNikonYUY2 decodes 12 bit data in an YUY2-like pattern (2 Luma, 1 Chroma per 2 pixels).
613 // We un-apply the whitebalance, so output matches lossless.
614 // Note that values are scaled. See comment below on details.
615 // OPTME: It would be trivial to run this multithreaded.
616 void NefDecoder::DecodeNikonSNef(const ByteStream& input) {
617  if (mRaw->dim.x < 6)
618  ThrowIOE("got a %u wide sNEF, aborting", mRaw->dim.x);
619 
620  // We need to read the applied whitebalance, since we should return
621  // data before whitebalance, so we "unapply" it.
622  TiffEntry* wb = mRootIFD->getEntryRecursive(static_cast<TiffTag>(12));
623  if (!wb)
624  ThrowRDE("Unable to locate whitebalance needed for decompression");
625 
626  assert(wb != nullptr);
627  if (wb->count != 4 || wb->type != TIFF_RATIONAL)
628  ThrowRDE("Whitebalance has unknown count or type");
629 
630  float wb_r = wb->getFloat(0);
631  float wb_b = wb->getFloat(1);
632 
633  // ((1024/x)*((1<<16)-1)+(1<<9))<=((1<<31)-1), x>0 gives: (0.0312495)
634  const float lower_limit = 13'421'568.0 / 429'496'627.0;
635  if (wb_r < lower_limit || wb_b < lower_limit || wb_r > 10.0F || wb_b > 10.0F)
636  ThrowRDE("Whitebalance has bad values (%f, %f)", wb_r, wb_b);
637 
638  mRaw->metadata.wbCoeffs[0] = wb_r;
639  mRaw->metadata.wbCoeffs[1] = 1.0F;
640  mRaw->metadata.wbCoeffs[2] = wb_b;
641 
642  auto inv_wb_r = static_cast<int>(1024.0 / wb_r);
643  auto inv_wb_b = static_cast<int>(1024.0 / wb_b);
644 
645  auto curve = gammaCurve(1 / 2.4, 12.92, 1, 4095);
646 
647  // Scale output values to 16 bits.
648  for (int i = 0 ; i < 4096; i++) {
649  curve[i] = clampBits(static_cast<int>(curve[i]) << 2, 16);
650  }
651 
652  curve.resize(4095);
653 
654  RawImageCurveGuard curveHandler(&mRaw, curve, false);
655 
656  uint16_t tmp;
657  auto* tmpch = reinterpret_cast<uint8_t*>(&tmp);
658 
659  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
660  const uint8_t* in = input.peekData(out.width * out.height);
661 
662  for (int row = 0; row < out.height; row++) {
663  uint32_t random = in[0] + (in[1] << 8) + (in[2] << 16);
664  for (int col = 0; col < out.width; col += 6) {
665  uint32_t g1 = in[0];
666  uint32_t g2 = in[1];
667  uint32_t g3 = in[2];
668  uint32_t g4 = in[3];
669  uint32_t g5 = in[4];
670  uint32_t g6 = in[5];
671 
672  in+=6;
673  auto y1 = static_cast<float>(g1 | ((g2 & 0x0f) << 8));
674  auto y2 = static_cast<float>((g2 >> 4) | (g3 << 4));
675  auto cb = static_cast<float>(g4 | ((g5 & 0x0f) << 8));
676  auto cr = static_cast<float>((g5 >> 4) | (g6 << 4));
677 
678  float cb2 = cb;
679  float cr2 = cr;
680  // Interpolate right pixel. We assume the sample is aligned with left pixel.
681  if ((col + 6) < out.width) {
682  g4 = in[3];
683  g5 = in[4];
684  g6 = in[5];
685  cb2 = (static_cast<float>((g4 | ((g5 & 0x0f) << 8))) + cb) * 0.5F;
686  cr2 = (static_cast<float>(((g5 >> 4) | (g6 << 4))) + cr) * 0.5F;
687  }
688 
689  cb -= 2048;
690  cr -= 2048;
691  cb2 -= 2048;
692  cr2 -= 2048;
693 
694  mRaw->setWithLookUp(clampBits(static_cast<int>(y1 + 1.370705 * cr), 12),
695  tmpch, &random);
696  out(row, col) = clampBits((inv_wb_r * tmp + (1 << 9)) >> 10, 15);
697 
698  mRaw->setWithLookUp(
699  clampBits(static_cast<int>(y1 - 0.337633 * cb - 0.698001 * cr), 12),
700  reinterpret_cast<uint8_t*>(&out(row, col + 1)), &random);
701 
702  mRaw->setWithLookUp(clampBits(static_cast<int>(y1 + 1.732446 * cb), 12),
703  tmpch, &random);
704  out(row, col + 2) = clampBits((inv_wb_b * tmp + (1 << 9)) >> 10, 15);
705 
706  mRaw->setWithLookUp(clampBits(static_cast<int>(y2 + 1.370705 * cr2), 12),
707  tmpch, &random);
708  out(row, col + 3) = clampBits((inv_wb_r * tmp + (1 << 9)) >> 10, 15);
709 
710  mRaw->setWithLookUp(
711  clampBits(static_cast<int>(y2 - 0.337633 * cb2 - 0.698001 * cr2), 12),
712  reinterpret_cast<uint8_t*>(&out(row, col + 4)), &random);
713 
714  mRaw->setWithLookUp(clampBits(static_cast<int>(y2 + 1.732446 * cb2), 12),
715  tmpch, &random);
716  out(row, col + 5) = clampBits((inv_wb_b * tmp + (1 << 9)) >> 10, 15);
717  }
718  }

722 #define SQR(x) ((x)*(x))
723 std::vector<uint16_t> NefDecoder::gammaCurve(double pwr, double ts, int mode,
724  int imax) {
725  std::vector<uint16_t> curve(65536);
726 
727  int i;
728  std::array<double, 6> g;
729  std::array<double, 2> bnd = {{}};
730  double r;
731  g[0] = pwr;
732  g[1] = ts;
733  g[2] = g[3] = g[4] = 0;
734  bnd[g[1] >= 1] = 1;
735  if (g[1] && (g[1]-1)*(g[0]-1) <= 0) {
736  for (i=0; i < 48; i++) {
737  g[2] = (bnd[0] + bnd[1])/2;
738  if (g[0])
739  bnd[(pow(g[2] / g[1], -g[0]) - 1) / g[0] - 1 / g[2] > -1] = g[2];
740  else
741  bnd[g[2] / exp(1 - 1 / g[2]) < g[1]] = g[2];
742  }
743  g[3] = g[2] / g[1];
744  if (g[0])
745  g[4] = g[2] * (1 / g[0] - 1);
746  }
747  if (g[0]) {
748  g[5] = 1 / (g[1] * SQR(g[3]) / 2 - g[4] * (1 - g[3]) +
749  (1 - pow(g[3], 1 + g[0])) * (1 + g[4]) / (1 + g[0])) -
750  1;
751  } else {
752  g[5] = 1 / (g[1] * SQR(g[3]) / 2 + 1 - g[2] - g[3] -
753  g[2] * g[3] * (log(g[3]) - 1)) -
754  1;
755  }
756 
757  if (mode == 0)
758  ThrowRDE("Unimplemented mode");
759 
760  mode--;
761 
762  for (i=0; i < 0x10000; i++) {
763  curve[i] = 0xffff;
764  if ((r = static_cast<double>(i) / imax) < 1) {
765  curve[i] = static_cast<uint16_t>(
766  0x10000 *
767  (mode ? (r < g[3] ? r * g[1]
768  : (g[0] ? pow(r, g[0]) * (1 + g[4]) - g[4]
769  : log(r) * g[2] + 1))
770  : (r < g[2] ? r / g[1]
771  : (g[0] ? pow((r + g[4]) / (1 + g[4]), 1 / g[0])
772  : exp((r - 1) / g[2])))));
773  }
774  }
775 
776  assert(curve.size() == 65536);
777 
778  return curve;
779 }
