36 class PanasonicDecompressorV5 final : public AbstractDecompressor {
37  // The RW2 raw image buffer consists of individual blocks,
38  // each one BlockSize bytes in size.
39  static constexpr uint32_t BlockSize = 0x4000;
40 
41  // These blocks themselves comprise of two sections,
42  // split and swapped at section_split_offset:
43  // bytes: [0..sectionSplitOffset-1][sectionSplitOffset..BlockSize-1]
44  // pixels: [a..b][0..a-1]
45  // When reading, these two sections need to be swapped to enable linear
46  // processing..
47  static constexpr uint32_t sectionSplitOffset = 0x1FF8;
48 
49  // The blocks themselves consist of packets with fixed size of bytesPerPacket,
50  // and each packet decodes to pixelsPerPacket pixels, which depends on bps.
51  static constexpr uint32_t bytesPerPacket = 16;
52  static constexpr uint32_t bitsPerPacket = 8 * bytesPerPacket;
53  static_assert(BlockSize % bytesPerPacket == 0, "");
54  static constexpr uint32_t PacketsPerBlock = BlockSize / bytesPerPacket;
55

59  // There are two variants. Which one is to be used depends on image's bps.
60  static const PacketDsc TwelveBitPacket;
61  static const PacketDsc FourteenBitPacket;
