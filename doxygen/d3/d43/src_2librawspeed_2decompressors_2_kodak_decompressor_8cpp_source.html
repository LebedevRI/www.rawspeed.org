63 KodakDecompressor::decodeSegment(const uint32_t bsize) {
64  assert(bsize > 0);
65  assert(bsize % 4 == 0);
66  assert(bsize <= segment_size);
67 
68  segment out;
69  static_assert(out.size() == segment_size, "Wrong segment size");
70 
71 #if __has_feature(address_sanitizer) || defined(__SANITIZE_ADDRESS__)
72  // We are to produce only bsize pixels.
73  __sanitizer_annotate_contiguous_container(out.begin(), out.end(), out.end(),
74  out.begin() + bsize);
75 #endif
76 
77  std::array<uint8_t, 2 * segment_size> blen;
78  uint64_t bitbuf = 0;
79  uint32_t bits = 0;
80 
81  for (uint32_t i = 0; i < bsize; i += 2) {
82  // One byte per two pixels
83  blen[i] = input.peekByte() & 15;
84  blen[i + 1] = input.getByte() >> 4;
85  }
86  if ((bsize & 7) == 4) {
87  bitbuf = (static_cast<uint64_t>(input.getByte())) << 8UL;
88  bitbuf += (static_cast<int>(input.getByte()));
89  bits = 16;
90  }
91  for (uint32_t i = 0; i < bsize; i++) {
92  uint32_t len = blen[i];
93  assert(len < 16);
94 
95  if (bits < len) {
96  for (uint32_t j = 0; j < 32; j += 8) {
97  bitbuf += static_cast<int64_t>(static_cast<int>(input.getByte()))
98  << (bits + (j ^ 8));
99  }
100  bits += 32;
101  }
102 
103  uint32_t diff = static_cast<uint32_t>(bitbuf) &
104  extractHighBits(0xffffU, len, /*effectiveBitwidth=*/16);
105  bitbuf >>= len;
106  bits -= len;
107 
108  out[i] = len != 0 ? HuffmanTable::extend(diff, len) : int(diff);
109  }
110 
111  return out;
112 }

114 void KodakDecompressor::decompress() {
115  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
116 
117  uint32_t random = 0;
118  for (int row = 0; row < out.height; row++) {
119  for (int col = 0; col < out.width;) {
120  const int len = std::min(segment_size, mRaw->dim.x - col);
121 
122  const segment buf = decodeSegment(len);
123 
124  std::array<int, 2> pred;
125  pred.fill(0);
126 
127  for (int i = 0; i < len; ++i, ++col) {
128  pred[i & 1] += buf[i];
129 
130  int value = pred[i & 1];
131  if (!isIntN(value, bps))
132  ThrowRDE("Value out of bounds %d (bps = %i)", value, bps);
133 
134  if (uncorrectedRawValues)
135  out(row, col) = value;
136  else
137  mRaw->setWithLookUp(value, reinterpret_cast<uint8_t*>(&out(row, col)),
138  &random);
139  }
140  }
141  }
142 }
