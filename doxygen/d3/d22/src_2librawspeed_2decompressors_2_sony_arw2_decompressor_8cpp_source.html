38 SonyArw2Decompressor::SonyArw2Decompressor(const RawImage& img,
39  const ByteStream& input_)
40  : mRaw(img) {
41  if (mRaw->getCpp() != 1 || mRaw->getDataType() != TYPE_USHORT16 ||
42  mRaw->getBpp() != sizeof(uint16_t))
43  ThrowRDE("Unexpected component count / data type");
44 
45  const uint32_t w = mRaw->dim.x;
46  const uint32_t h = mRaw->dim.y;
47

50

55 void SonyArw2Decompressor::decompressRow(int row) const {
56  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
57  assert(out.width > 0);
58  assert(out.width % 32 == 0);
59 
60  // Allow compiler to devirtualize the calls below.
61  auto& rawdata = reinterpret_cast<RawImageDataU16&>(*mRaw);
62 
63  ByteStream rowBs = input;
64  rowBs.skipBytes(row * out.width);
65  rowBs = rowBs.peekStream(out.width);
66 
67  BitPumpLSB bits(rowBs);
68 
69  uint32_t random = bits.peekBits(24);
70 
71  // Each loop iteration processes 16 pixels, consuming 128 bits of input.
72  for (int col = 0; col < out.width; col += ((col & 1) != 0) ? 31 : 1) {
73  // 30 bits.
74  int _max = bits.getBits(11);
75  int _min = bits.getBits(11);
76  int _imax = bits.getBits(4);
77  int _imin = bits.getBits(4);
78 
79  // 128-30 = 98 bits remaining, still need to decode 16 pixels...
80  // Each full pixel consumes 7 bits, thus we can only have 14 full pixels.
81  // So we lack 2 pixels. That is where _imin and _imax come into play,
82  // values of those pixels were already specified in _min and _max.
83  // But what that means is, _imin and _imax must not be equal!
84  if (_imax == _imin)
85  ThrowRDE("ARW2 invariant failed, same pixel is both min and max");
86 
87  int sh = 0;
88  while ((sh < 4) && ((0x80 << sh) <= (_max - _min)))
89  sh++;
90 
91  for (int i = 0; i < 16; i++) {
92  int p;
93  if (i == _imax)
94  p = _max;
95  else {
96  if (i == _imin)
97  p = _min;
98  else {
99  p = (bits.getBits(7) << sh) + _min;
100  if (p > 0x7ff)
101  p = 0x7ff;
102  }
103  }
104  rawdata.setWithLookUp(
105  p << 1, reinterpret_cast<uint8_t*>(&out(row, col + i * 2)), &random);
106  }
107  }
108 }

110 void SonyArw2Decompressor::decompressThread() const noexcept {
111  assert(mRaw->dim.x > 0);
112  assert(mRaw->dim.x % 32 == 0);
113  assert(mRaw->dim.y > 0);
114 
115 #ifdef HAVE_OPENMP
116 #pragma omp for schedule(static)
117 #endif
118  for (int y = 0; y < mRaw->dim.y; y++) {
119  try {
120  decompressRow(y);
121  } catch (RawspeedException& err) {
122  // Propagate the exception out of OpenMP magic.
123  mRaw->setError(err.what());
124 #ifdef HAVE_OPENMP
125 #pragma omp cancel for
126 #endif
127  }
128  }
129 }
