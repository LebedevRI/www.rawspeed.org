40 template <const iPoint2D& subSampling, typename version>
41 static inline void BM_Cr2sRawInterpolator(benchmark::State& state) {
42  static const array<int, 3> sraw_coeffs = {{999, 1000, 1001}};
43  static const int hue = 1269;
44 
45  iPoint2D interpolatedDims = areaToRectangle(state.range(0), {3, 2});
46 
47  interpolatedDims.x = roundUp(interpolatedDims.x, 6);
48  if (subSampling.y == 2)
49  interpolatedDims.x = roundUp(interpolatedDims.x, 4);
50 
51  iPoint2D subsampledDim = interpolatedDims;
52  subsampledDim.x /= subSampling.x;
53  subsampledDim.y /= subSampling.y;
54  subsampledDim.x *= 2 + subSampling.x * subSampling.y;
55 
56  RawImage subsampledRaw = RawImage::create(subsampledDim, TYPE_USHORT16, 1);
57  subsampledRaw->metadata.subsampling = subSampling;
58 
59  RawImage mRaw = RawImage::create(interpolatedDims, TYPE_USHORT16, 3);
60  mRaw->metadata.subsampling = subSampling;
61 
62  Cr2sRawInterpolator i(mRaw, subsampledRaw->getU16DataAsUncroppedArray2DRef(),
63  sraw_coeffs, hue);
64 
65  for (auto _ : state)
66  i.interpolate(version::value);
67 
68  state.SetComplexityN(interpolatedDims.area());
69  state.counters.insert(
70  {{"Pixels", benchmark::Counter(
71  state.complexity_length_n(),
72  benchmark::Counter::Flags::kIsIterationInvariantRate)},
73  {"Bytes",
74  benchmark::Counter(3UL * sizeof(uint16_t) * state.complexity_length_n(),
75  benchmark::Counter::Flags::kIsIterationInvariantRate,
76  benchmark::Counter::kIs1024)}});
77 }
