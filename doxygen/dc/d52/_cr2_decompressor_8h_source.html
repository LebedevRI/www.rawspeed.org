41 public:

44  Cr2Slicing(uint16_t numSlices_, uint16_t sliceWidth_,
45  uint16_t lastSliceWidth_)
46  : numSlices(numSlices_), sliceWidth(sliceWidth_),
47  lastSliceWidth(lastSliceWidth_) {
48  if (numSlices < 1)
49  ThrowRDE("Bad slice count: %u", numSlices);
50  }
51 
52  bool empty() const {
53  return 0 == numSlices && 0 == sliceWidth && 0 == lastSliceWidth;
54  }
55 
56  unsigned widthOfSlice(int sliceId) const {
57  assert(sliceId >= 0);
58  assert(sliceId < numSlices);
59  if ((sliceId + 1) == numSlices)
60  return lastSliceWidth;
61  return sliceWidth;
62  }
63 
64  unsigned totalWidth() const {
65  int width = 0;
66  for (auto sliceId = 0; sliceId < numSlices; sliceId++)
67  width += widthOfSlice(sliceId);
68  return width;
69  }
70 };
