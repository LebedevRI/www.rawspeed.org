69 class HuffmanTableLookup : public AbstractHuffmanTable {
70 protected:
71  // private fields calculated from codesPerBits and codeValues
72  // they are index '1' based, so we can directly lookup the value
73  // for code length l without decrementing
74  std::vector<uint32_t> maxCodeOL; // index is length of code
75  std::vector<uint16_t> codeOffsetOL; // index is length of code
76 
77 public:
78  std::vector<CodeSymbol> setup(bool fullDecode_, bool fixDNGBug16_) {
79  AbstractHuffmanTable::setup(fullDecode_, fixDNGBug16_);
80 
81  // Figure C.1: make table of Huffman code length for each symbol
82  // Figure C.2: generate the codes themselves
83  std::vector<CodeSymbol> symbols = generateCodeSymbols();
84  assert(symbols.size() == maxCodesCount());
85 
86  // Figure F.15: generate decoding tables
87  unsigned int maxCodeLength = nCodesPerLength.size() - 1U;
88  codeOffsetOL.resize(maxCodeLength + 1UL, 0xFFFF);
89  maxCodeOL.resize(maxCodeLength + 1UL, 0xFFFFFFFF);
90  for (unsigned int numCodesSoFar = 0, codeLen = 1; codeLen <= maxCodeLength;
91  codeLen++) {
92  if (!nCodesPerLength[codeLen])
93  continue;
94  codeOffsetOL[codeLen] = symbols[numCodesSoFar].code - numCodesSoFar;
95  numCodesSoFar += nCodesPerLength[codeLen];
96  maxCodeOL[codeLen] = symbols[numCodesSoFar - 1].code;
97  }
98 
99  return symbols;
100  }

102  template <typename BIT_STREAM>
103  inline int decodeCodeValue(BIT_STREAM& bs) const {
104  static_assert(BitStreamTraits<BIT_STREAM>::canUseWithHuffmanTable,
105  "This BitStream specialization is not marked as usable here");
106  assert(!fullDecode);

110  template <typename BIT_STREAM>
111  inline int decodeDifference(BIT_STREAM& bs) const {
112  static_assert(BitStreamTraits<BIT_STREAM>::canUseWithHuffmanTable,
113  "This BitStream specialization is not marked as usable here");
114  assert(fullDecode);
115  return decode<BIT_STREAM, true>(bs);
116  }
117 
118 protected:
119  template <typename BIT_STREAM>
120  inline std::pair<CodeSymbol, int /*codeValue*/>
121  finishReadingPartialSymbol(BIT_STREAM& bs, CodeSymbol partial) const {
122  while (partial.code_len < maxCodeOL.size() &&
123  (0xFFFFFFFF == maxCodeOL[partial.code_len] ||
124  partial.code > maxCodeOL[partial.code_len])) {
125  uint32_t temp = bs.getBitsNoFill(1);
126  partial.code = (partial.code << 1) | temp;
127  partial.code_len++;
128  }
129 
130  if (partial.code_len >= maxCodeOL.size() ||
131  (0xFFFFFFFF == maxCodeOL[partial.code_len] ||
132  partial.code > maxCodeOL[partial.code_len]) ||
133  partial.code < codeOffsetOL[partial.code_len])
134  ThrowRDE("bad Huffman code: %u (len: %u)", partial.code,
135  partial.code_len);
136 
137  int codeValue = codeValues[partial.code - codeOffsetOL[partial.code_len]];
138 
139  return {partial, codeValue};
140  }
141 
142  template <typename BIT_STREAM>
143  inline std::pair<CodeSymbol, int /*codeValue*/>
144  readSymbol(BIT_STREAM& bs) const {
145  // Start from completely unknown symbol.
146  CodeSymbol partial;
147  partial.code_len = 0;
148  partial.code = 0;
149 
150  return finishReadingPartialSymbol(bs, partial);
151  }
152 
153 public:
154  // The bool template paraeter is to enable two versions:
155  // one returning only the length of the of diff bits (see Hasselblad),
156  // one to return the fully decoded diff.
157  // All ifs depending on this bool will be optimized out by the compiler
158  template <typename BIT_STREAM, bool FULL_DECODE>
159  inline int decode(BIT_STREAM& bs) const {
160  static_assert(BitStreamTraits<BIT_STREAM>::canUseWithHuffmanTable,
161  "This BitStream specialization is not marked as usable here");
162  assert(FULL_DECODE == fullDecode);
163  bs.fill(32);
164 
165  CodeSymbol symbol;
166  int codeValue;
167  std::tie(symbol, codeValue) = readSymbol(bs);
168 
169  return processSymbol<BIT_STREAM, FULL_DECODE>(bs, symbol, codeValue);
170  }
