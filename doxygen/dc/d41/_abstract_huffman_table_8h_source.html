53  static bool HaveCommonPrefix(const CodeSymbol& symbol,
54  const CodeSymbol& partial) {
55  assert(partial.code_len <= symbol.code_len);
56 
57  const auto s0 = extractHighBits(symbol.code, partial.code_len,

113  static void VerifyCodeSymbols(const std::vector<CodeSymbol>& symbols) {
114 #ifndef NDEBUG
115  // The code symbols are ordered so that all the code values are strictly
116  // increasing and code lengths are not decreasing.
117  const auto symbolSort = [](const CodeSymbol& lhs,
118  const CodeSymbol& rhs) -> bool {
119  return std::less<>()(lhs.code, rhs.code) &&
120  std::less_equal<>()(lhs.code_len, rhs.code_len);
121  };
122 #endif
123  assert(std::adjacent_find(symbols.cbegin(), symbols.cend(),
124  [&symbolSort](const CodeSymbol& lhs,
125  const CodeSymbol& rhs) -> bool {
126  return !symbolSort(lhs, rhs);
127  }) == symbols.cend() &&
128  "all code symbols are globally ordered");
129 
130  // No two symbols should have the same prefix (high bytes)
131  // Only analyze the lower triangular matrix, excluding diagonal
132  for (auto sId = 0UL; sId < symbols.size(); sId++) {
133  for (auto pId = 0UL; pId < sId; pId++)
134  assert(!CodeSymbol::HaveCommonPrefix(symbols[sId], symbols[pId]));
135  }
136  }

138  std::vector<CodeSymbol> generateCodeSymbols() const {
139  std::vector<CodeSymbol> symbols;
140 
141  assert(!nCodesPerLength.empty());
142  assert(maxCodesCount() > 0);
143 
144  const auto maxCodeLength = nCodesPerLength.size() - 1U;
145  assert(codeValues.size() == maxCodesCount());
146 
147  // reserve all the memory. avoids lots of small allocs
148  symbols.reserve(maxCodesCount());
149 
150  // Figure C.1: make table of Huffman code length for each symbol
151  // Figure C.2: generate the codes themselves
152  uint32_t code = 0;
153  for (unsigned int l = 1; l <= maxCodeLength; ++l) {
154  for (unsigned int i = 0; i < nCodesPerLength[l]; ++i) {
155  assert(code <= 0xffff);
156 
157  symbols.emplace_back(code, l);
158  code++;
159  }
160 
161  code <<= 1;
162  }
163 
164  assert(symbols.size() == maxCodesCount());
165  VerifyCodeSymbols(symbols);
166 
167  return symbols;
168  }

176  uint32_t setNCodesPerLength(const Buffer& data) {
177  assert(data.getSize() == 16);
178 
179  nCodesPerLength.resize(17, 0);
180  std::copy(data.begin(), data.end(), &nCodesPerLength[1]);
181  assert(nCodesPerLength[0] == 0);
182 
183  // trim empty entries from the codes per length table on the right
184  while (!nCodesPerLength.empty() && nCodesPerLength.back() == 0)
185  nCodesPerLength.pop_back();
186 
187  if (nCodesPerLength.empty())
188  ThrowRDE("Codes-per-length table is empty");
189 
190  assert(nCodesPerLength.back() > 0);
191 
192  const auto count = maxCodesCount();
193  assert(count > 0);
194 
195  if (count > 162)
196  ThrowRDE("Too big code-values table");
197 
198  // We are at the Root node, len is 1, there are two possible child Nodes
199  unsigned maxCodes = 2;
200 
201  for (auto codeLen = 1UL; codeLen < nCodesPerLength.size(); codeLen++) {
202  // we have codeLen bits. make sure that that code count can actually fit
203  // E.g. for len 1 we could have two codes: 0b0 and 0b1
204  // (but in that case there can be no other codes (with higher lengths))
205  const auto maxCodesInCurrLen = (1U << codeLen);
206  const auto nCodes = nCodesPerLength[codeLen];
207  if (nCodes > maxCodesInCurrLen) {
208  ThrowRDE("Corrupt Huffman. Can never have %u codes in %lu-bit len",
209  nCodes, codeLen);
210  }
211 
212  // Also, check that we actually can have this much leafs for this length
213  if (nCodes > maxCodes) {
214  ThrowRDE(
215  "Corrupt Huffman. Can only fit %u out of %u codes in %lu-bit len",
216  maxCodes, nCodes, codeLen);
217  }
218 
219  // There are nCodes leafs on this level, and those can not be branches
220  maxCodes -= nCodes;
221  // On the next level, rest can be branches, and can have two child Nodes
222  maxCodes *= 2;
223  }
224 
225  return count;
226  }

240  template <typename BIT_STREAM, bool FULL_DECODE>
241  inline int processSymbol(BIT_STREAM& bs, CodeSymbol symbol,
242  int codeValue) const {
243  assert(symbol.code_len >= 0 && symbol.code_len <= 16);
244 
245  // If we were only looking for symbol's code value, then just return it.
246  if (!FULL_DECODE)
247  return codeValue;
248 
249  // Else, treat it as the length of following difference
250  // that we need to read and extend.
251  int diff_l = codeValue;
252  assert(diff_l >= 0 && diff_l <= 16);
253 
254  if (diff_l == 16) {
255  if (fixDNGBug16)
256  bs.skipBitsNoFill(16);
257  return -32768;
258  }
259 
260  assert(symbol.code_len + diff_l <= 32);
261  return diff_l ? extend(bs.getBitsNoFill(diff_l), diff_l) : 0;
262  }
263 
264  // Figure F.12 – Extending the sign bit of a decoded value in V
265  // WARNING: this is *not* your normal 2's complement sign extension!
266  inline static int __attribute__((const)) extend(uint32_t diff, uint32_t len) {
267  assert(len > 0);
268  auto ret = static_cast<int32_t>(diff);
269  if ((diff & (1 << (len - 1))) == 0)
270  ret -= (1 << len) - 1;
271  return ret;
272  }
