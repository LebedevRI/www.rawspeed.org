43 enum struct SamsungV2Decompressor::OptFlags : uint32_t {
44  NONE = 0U, // no flags
45  SKIP = 1U << 0U, // Skip checking if we need differences from previous line
46  MV = 1U << 1U, // Simplify motion vector definition

83 SamsungV2Decompressor::SamsungV2Decompressor(const RawImage& image,
84  const ByteStream& bs,
85  unsigned bits)
86  : AbstractSamsungDecompressor(image) {
87  if (mRaw->getCpp() != 1 || mRaw->getDataType() != TYPE_USHORT16 ||
88  mRaw->getBpp() != sizeof(uint16_t))
89  ThrowRDE("Unexpected component count / data type");
90 
91  switch (bits) {
92  case 12:
93  case 14:
94  break;
95  default:
96  ThrowRDE("Unexpected bit per pixel (%u)", bits);
97  }
98 
99  static constexpr const auto headerSize = 16;
100  bs.check(headerSize);
101 
102  BitPumpMSB32 startpump(bs);
103 
104  // Process the initial metadata bits, we only really use initVal, width and
105  // height (the last two match the TIFF values anyway)
106  startpump.getBits(16); // NLCVersion
107  startpump.getBits(4); // ImgFormat
108  bitDepth = startpump.getBits(4) + 1;
109  if (bitDepth != bits)
110  ThrowRDE("Bit depth mismatch with container, %u vs %u", bitDepth, bits);
111  startpump.getBits(4); // NumBlkInRCUnit
112  startpump.getBits(4); // CompressionRatio
113  width = startpump.getBits(16);
114  height = startpump.getBits(16);
115  startpump.getBits(16); // TileWidth
116  startpump.getBits(4); // reserved
117 
118  // The format includes an optimization code that sets 3 flags to change the
119  // decoding parameters
120  const uint32_t _flags = startpump.getBits(4);
121  if (_flags > static_cast<uint32_t>(OptFlags::ALL))
122  ThrowRDE("Invalid opt flags %x", _flags);
123  optflags = static_cast<OptFlags>(_flags);
124 
125  startpump.getBits(8); // OverlapWidth
126  startpump.getBits(8); // reserved
127  startpump.getBits(8); // Inc
128  startpump.getBits(2); // reserved
129  initVal = startpump.getBits(14);
130 
131  assert(startpump.getInputPosition() == headerSize);
132 
133  if (width == 0 || height == 0 || width % 16 != 0 || width > 6496 ||
134  height > 4336)
135  ThrowRDE("Unexpected image dimensions found: (%i; %i)", width, height);
136 
137  if (width != mRaw->dim.x || height != mRaw->dim.y)
138  ThrowRDE("EXIF image dimensions do not match dimensions from raw header");
139 
140  data = bs.getSubStream(startpump.getInputPosition(),
141  startpump.getRemainingSize());
142 }

144 // The format is relatively straightforward. Each line gets encoded as a set
145 // of differences from pixels from another line. Pixels are grouped in blocks
146 // of 16 (8 green, 8 red or blue). Each block is encoded in three sections.
147 // First 1 or 4 bits to specify which reference pixels to use, then a section
148 // that specifies for each pixel the number of bits in the difference, then
149 // the actual difference bits
150 
151 inline __attribute__((always_inline)) std::array<uint16_t, 16>
152 SamsungV2Decompressor::prepareBaselineValues(BitPumpMSB32* pump, int row,
153  int col) {
154  const Array2DRef<uint16_t> img(mRaw->getU16DataAsUncroppedArray2DRef());
155 
156  std::array<uint16_t, 16> baseline;
157 
158  if (!(optflags & OptFlags::QP) && (col % 64) == 0) {
159  // Change scale every four 16-pixel blocks.
160  static constexpr std::array<int32_t, 3> scalevals = {{0, -2, 2}};
161  uint32_t i = pump->getBits(2);
162  scale = i < 3 ? scale + scalevals[i] : pump->getBits(12);
163  }
164 
165  // First we figure out which reference pixels mode we're in
166  if (optflags & OptFlags::MV)
167  motion = pump->getBits(1) ? 3 : 7;
168  else if (!pump->getBits(1))
169  motion = pump->getBits(3);
170 
171  if ((row == 0 || row == 1) && (motion != 7))
172  ThrowRDE("At start of image and motion isn't 7. File corrupted?");
173 
174  if (motion == 7) {
175  // The base case.
176  // If we're at the left edge we just start at the initial value.
177  if (col == 0) {
178  baseline.fill(initVal);
179  return baseline;
180  }
181  // Else just set all pixels to the previous ones on the same line.
182  std::array<uint16_t, 2> prev;
183  for (int i = 0; i < 2; i++)
184  prev[i] = img(row, col + i - 2);
185  for (int i = 0; i < 16; i++)
186  baseline[i] = prev[i & 1];
187  return baseline;
188  }
189 
190  // The complex case, we now need to actually lookup one or two lines above
191  if (row < 2)
192  ThrowRDE("Got a previous line lookup on first two lines. File corrupted?");
193 
194  static constexpr std::array<int32_t, 7> motionOffset = {-4, -2, -2, 0,
195  0, 2, 4};
196  static constexpr std::array<int32_t, 7> motionDoAverage = {0, 0, 1, 0,
197  1, 0, 0};
198 
199  int32_t slideOffset = motionOffset[motion];
200  int32_t doAverage = motionDoAverage[motion];
201 
202  for (int i = 0; i < 16; i++) {
203  int refRow = row;
204  int refCol = col + i + slideOffset;
205 
206  if ((row + i) & 1) { // Red or blue pixels use same color two lines up
207  refRow -= 2;
208  } else { // Green pixel N uses Green pixel N from row above
209  refRow -= 1;
210  refCol += (i & 1) ? -1 : 1; // (top left or top right)
211  }
212 
213  if (refCol < 0)
214  ThrowRDE("Bad motion %u at the beginning of the row", motion);
215  if ((refCol >= width) || (doAverage && (refCol + 2 >= width)))
216  ThrowRDE("Bad motion %u at the end of the row", motion);
217 
218  // In some cases we use as reference interpolation of this pixel and
219  // the next
220  if (doAverage) {
221  baseline[i] = (img(refRow, refCol) + img(refRow, refCol + 2) + 1) >> 1;
222  } else
223  baseline[i] = img(refRow, refCol);
224  }
225 
226  return baseline;
227 }

229 inline __attribute__((always_inline)) std::array<uint32_t, 4>
230 SamsungV2Decompressor::decodeDiffLengths(BitPumpMSB32* pump, int row) {
231  if (!(optflags & OptFlags::SKIP) && pump->getBits(1))
232  return {};
233 
234  std::array<uint32_t, 4> diffBits;
235 
236  // Figure out how many difference bits we have to read for each pixel
237  std::array<uint32_t, 4> flags;
238  for (unsigned int& flag : flags)
239  flag = pump->getBits(2);
240 
241  for (int i = 0; i < 4; i++) {
242  // The color is 0-Green 1-Blue 2-Red
243  uint32_t colornum = (row % 2 != 0) ? i >> 1 : ((i >> 1) + 2) % 3;
244 
245  assert(flags[i] <= 3);
246  switch (flags[i]) {
247  case 0:
248  diffBits[i] = diffBitsMode[colornum][0];
249  break;
250  case 1:
251  diffBits[i] = diffBitsMode[colornum][0] + 1;
252  break;
253  case 2:
254  if (diffBitsMode[colornum][0] == 0)
255  ThrowRDE("Difference bits underflow. File corrupted?");
256  diffBits[i] = diffBitsMode[colornum][0] - 1;
257  break;
258  case 3:
259  diffBits[i] = pump->getBits(4);
260  break;
261  default:
262  __builtin_unreachable();
263  }
264 
265  diffBitsMode[colornum][0] = diffBitsMode[colornum][1];
266  diffBitsMode[colornum][1] = diffBits[i];
267 
268  if (diffBits[i] > bitDepth + 1)
269  ThrowRDE("Too many difference bits (%u). File corrupted?", diffBits[i]);
270  assert(diffBits[i] <= 15 && "So any difference fits within uint16_t");
271  }
272 
273  return diffBits;
274 }

276 inline __attribute__((always_inline)) std::array<int, 16>
277 SamsungV2Decompressor::decodeDifferences(BitPumpMSB32* pump, int row) {
278  // Figure out how many difference bits we have to read for each pixel
279  const std::array<uint32_t, 4> diffBits = decodeDiffLengths(pump, row);
280 
281  // Actually read the differences. We know these fit into 15-bit ints.
282  std::array<int16_t, 16> diffs;
283  for (int i = 0; i < 16; i++) {
284  uint32_t len = diffBits[i >> 2];
285  int16_t diff = getDiff(pump, len);
286  diffs[i] = diff;
287  }
288 
289  // Reshuffle the differences, while they still are only 16-bit.
290  std::array<int16_t, 16> shuffled;
291  for (int i = 0; i < 16; i++) {
292  int p;
293  // The differences are stored interlaced:
294  // 0 2 4 6 8 10 12 14 1 3 5 7 9 11 13 15
295  if (row % 2)
296  p = ((i % 8) << 1) - (i >> 3) + 1;
297  else
298  p = ((i % 8) << 1) + (i >> 3);
299 
300  shuffled[p] = diffs[i];
301  }
302 
303  // And finally widen and scale the differences.
304  std::array<int, 16> scaled;
305  for (int i = 0; i < 16; i++) {
306  int scaledDiff = int(shuffled[i]) * (scale * 2 + 1) + scale;
307  scaled[i] = scaledDiff;
308  }
309 
310  return scaled;
311 }

313 inline __attribute__((always_inline)) void
314 SamsungV2Decompressor::processBlock(BitPumpMSB32* pump, int row, int col) {
315  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
316 
317  const std::array<uint16_t, 16> baseline =
318  prepareBaselineValues(pump, row, col);
319 
320  // Figure out how many difference bits we have to read for each pixel
321  const std::array<int, 16> diffs = decodeDifferences(pump, row);
322 
323  // Actually apply the differences and write them to the pixels
324  for (int i = 0; i < 16; ++i, ++col)
325  out(row, col) = clampBits(baseline[i] + diffs[i], bitDepth);
326 }

328 void SamsungV2Decompressor::decompressRow(int row) {
329  // Align pump to 16byte boundary
330  const auto line_offset = data.getPosition();
331  if ((line_offset & 0xf) != 0)
332  data.skipBytes(16 - (line_offset & 0xf));
333 
334  BitPumpMSB32 pump(data);
335 
336  // Initialize the motion and diff modes at the start of the line
337  motion = 7;
338  // By default we are not scaling values at all
339  scale = 0;
340 
341  for (auto& i : diffBitsMode)
342  i[0] = i[1] = (row == 0 || row == 1) ? 7 : 4;
343 
344  assert(width >= 16);
345  assert(width % 16 == 0);
346  for (int col = 0; col < width; col += 16)
347  processBlock(&pump, row, col);
348 
349  data.skipBytes(pump.getStreamPosition());
350 }
