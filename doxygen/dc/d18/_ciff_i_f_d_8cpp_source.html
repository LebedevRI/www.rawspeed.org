44 void CiffIFD::parseIFDEntry(NORangesSet<Buffer>* valueDatas,
45  const ByteStream* valueData,
46  ByteStream* dirEntries) {
47  assert(valueDatas);
48  assert(valueData);
49  assert(dirEntries);
50 
51  ByteStream dirEntry = dirEntries->getStream(10); // Entry is 10 bytes.
52 
53  auto t = std::make_unique<CiffEntry>(valueDatas, valueData, dirEntry);
54 
55  switch (t->type) {
56  case CIFF_SUB1:
57  case CIFF_SUB2: {
58  add(std::make_unique<CiffIFD>(this, t->data));
59  break;
60  }
61 
62  default:
63  // Will we ever look for this entry?
64  if (!isIn(t->tag, CiffTagsWeCareAbout))
65  return;
66  add(move(t));
67  }
68 }

70 CiffIFD::CiffIFD(CiffIFD* const parent_) : parent(parent_) {
71  recursivelyCheckSubIFDs(1);

73  // we are still here. However, due to the way we add parsed sub-IFD's (lazy),
74  // we need to count this IFD right *NOW*, not when adding it at the end.
75  recursivelyIncrementSubIFDCount();

78 CiffIFD::CiffIFD(CiffIFD* const parent_, ByteStream directory)
79  : CiffIFD(parent_) {
80  if (directory.getSize() < 4)
81  ThrowCPE("CIFF directory is too short.");
82 
83  directory.setPosition(directory.getSize() - 4);
84  const uint32_t valueDataSize = directory.getU32();
85 
86  // The Recursion. Directory entries store data here. May contain IFDs.
87  directory.setPosition(0);
88  const ByteStream valueData(directory.getStream(valueDataSize));
89 
90  // count of the Directory entries in this IFD
91  const uint16_t entryCount = directory.getU16();
92 
93  // each entry is 10 bytes
94  ByteStream dirEntries(directory.getStream(entryCount, 10));
95 
96  // IFDData might still contain OtherData until the valueDataSize at the end.
97  // But we do not care about that.
98 
99  // Each IFD has it's own valueData area.
100  // In that area, no two entries may overlap.
101  NORangesSet<Buffer> valueDatas;
102 
103  for (uint32_t i = 0; i < entryCount; i++)
104  parseIFDEntry(&valueDatas, &valueData, &dirEntries);
105 
106  assert(valueDatas.size() <= entryCount);
107  assert(mEntry.size() <= CiffTagsWeCareAbout.size());
108  assert(mSubIFD.size() == decltype(mSubIFD)::size_type(subIFDCount));
109  assert(subIFDCount <= subIFDCountRecursive);
110  assert(mEntry.size() + mSubIFD.size() <= entryCount);
111 }

113 void CiffIFD::recursivelyIncrementSubIFDCount() {
114  CiffIFD* p = this->parent;
115  if (!p)
116  return;
117 
118  p->subIFDCount++;
119 
120  for (; p != nullptr; p = p->parent)
121  p->subIFDCountRecursive++;
122 }

124 void CiffIFD::checkSubIFDs(int headroom) const {
125  int count = headroom + subIFDCount;
126  if (!headroom)
127  assert(count <= CiffIFD::Limits::SubIFDCount);
128  else if (count > CiffIFD::Limits::SubIFDCount)
129  ThrowCPE("TIFF IFD has %u SubIFDs", count);
130 
131  count = headroom + subIFDCountRecursive;
132  if (!headroom)
133  assert(count <= CiffIFD::Limits::RecursiveSubIFDCount);
134  else if (count > CiffIFD::Limits::RecursiveSubIFDCount)
135  ThrowCPE("TIFF IFD file has %u SubIFDs (recursively)", count);
136 }

138 void CiffIFD::recursivelyCheckSubIFDs(int headroom) const {
139  int depth = 0;
140  for (const CiffIFD* p = this; p != nullptr;) {
141  if (!headroom)
142  assert(depth <= CiffIFD::Limits::Depth);
143  else if (depth > CiffIFD::Limits::Depth)
144  ThrowCPE("CiffIFD cascading overflow, found %u level IFD", depth);
145 
146  p->checkSubIFDs(headroom);
147 
148  // And step up
149  p = p->parent;
150  depth++;
151  }
152 }

169 template <typename Lambda>
170 std::vector<const CiffIFD*> CiffIFD::getIFDsWithTagIf(CiffTag tag,
171  const Lambda& f) const {
172  assert(isIn(tag, CiffTagsWeCareAbout));
173 
174  std::vector<const CiffIFD*> matchingIFDs;
175 
176  const auto found = mEntry.find(tag);
177  if (found != mEntry.end()) {
178  const auto* const entry = found->second.get();
179  if (f(entry))
180  matchingIFDs.push_back(this);
181  }
182 
183  for (const auto& i : mSubIFD) {
184  const auto t = i->getIFDsWithTagIf(tag, f);
185  matchingIFDs.insert(matchingIFDs.end(), t.begin(), t.end());
186  }
187 
188  return matchingIFDs;
189 }
