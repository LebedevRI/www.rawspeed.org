61 RawImage ArwDecoder::decodeSRF(const TiffIFD* raw) {
62  raw = mRootIFD->getIFDWithTag(IMAGEWIDTH);
63 
64  uint32_t width = raw->getEntry(IMAGEWIDTH)->getU32();
65  uint32_t height = raw->getEntry(IMAGELENGTH)->getU32();
66 
67  if (width == 0 || height == 0 || width > 3360 || height > 2460)
68  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
69 
70  uint32_t len = width * height * 2;
71 
72  // Constants taken from dcraw
73  uint32_t off = 862144;
74  uint32_t key_off = 200896;
75  uint32_t head_off = 164600;
76 
77  // Replicate the dcraw contortions to get the "decryption" key
78  const uint8_t* keyData = mFile->getData(key_off, 1);
79  uint32_t offset = (*keyData) * 4;
80  keyData = mFile->getData(key_off + offset, 4);
81  uint32_t key = getU32BE(keyData);
82  static const size_t head_size = 40;
83  const uint8_t* head_orig = mFile->getData(head_off, head_size);
84  vector<uint8_t> head(head_size);
85  SonyDecrypt(reinterpret_cast<const uint32_t*>(head_orig),
86  reinterpret_cast<uint32_t*>(&head[0]), 10, key);
87  for (int i = 26; i > 22; i--)
88  key = key << 8 | head[i - 1];
89 
90  // "Decrypt" the whole image buffer
91  const auto* image_data = mFile->getData(off, len);
92  auto image_decoded = Buffer::Create(len);
93  SonyDecrypt(reinterpret_cast<const uint32_t*>(image_data),
94  reinterpret_cast<uint32_t*>(image_decoded.get()), len / 4, key);
95 
96  Buffer di(move(image_decoded), len);
97 
98  // And now decode as a normal 16bit raw
99  mRaw->dim = iPoint2D(width, height);
100  mRaw->createData();
101 
102  UncompressedDecompressor u(
103  ByteStream(DataBuffer(di.getSubView(0, len), Endianness::little)), mRaw);
104  u.decodeRawUnpacked<16, Endianness::big>(width, height);
105 
106  return mRaw;
107 }

109 RawImage ArwDecoder::decodeRawInternal() {
110  const TiffIFD* raw = nullptr;
111  vector<const TiffIFD*> data = mRootIFD->getIFDsWithTag(STRIPOFFSETS);
112 
113  if (data.empty()) {
114  TiffEntry *model = mRootIFD->getEntryRecursive(MODEL);
115 
116  if (model && model->getString() == "DSLR-A100") {
117  // We've caught the elusive A100 in the wild, a transitional format
118  // between the simple sanity of the MRW custom format and the wordly
119  // wonderfullness of the Tiff-based ARW format, let's shoot from the hip
120  raw = mRootIFD->getIFDWithTag(SUBIFDS);
121  uint32_t off = raw->getEntry(SUBIFDS)->getU32();
122  uint32_t width = 3881;
123  uint32_t height = 2608;
124 
125  mRaw->dim = iPoint2D(width, height);
126 
127  ByteStream input(DataBuffer(mFile->getSubView(off), Endianness::little));
128  SonyArw1Decompressor a(mRaw);
129  mRaw->createData();
130  a.decompress(input);
131 
132  return mRaw;
133  }
134 
135  if (hints.has("srf_format"))
136  return decodeSRF(raw);
137 
138  ThrowRDE("No image data found");
139  }
140 
141  raw = data[0];
142  int compression = raw->getEntry(COMPRESSION)->getU32();
143  if (1 == compression) {
144  DecodeUncompressed(raw);
145  return mRaw;
146  }
147 
148  if (32767 != compression)
149  ThrowRDE("Unsupported compression %i", compression);
150 
151  TiffEntry *offsets = raw->getEntry(STRIPOFFSETS);
152  TiffEntry *counts = raw->getEntry(STRIPBYTECOUNTS);
153 
154  if (offsets->count != 1) {
155  ThrowRDE("Multiple Strips found: %u", offsets->count);
156  }
157  if (counts->count != offsets->count) {
158  ThrowRDE(
159  "Byte count number does not match strip size: count:%u, strips:%u ",
160  counts->count, offsets->count);
161  }
162  uint32_t width = raw->getEntry(IMAGEWIDTH)->getU32();
163  uint32_t height = raw->getEntry(IMAGELENGTH)->getU32();
164  uint32_t bitPerPixel = raw->getEntry(BITSPERSAMPLE)->getU32();
165 
166  switch (bitPerPixel) {
167  case 8:
168  case 12:
169  case 14:
170  break;
171  default:
172  ThrowRDE("Unexpected bits per pixel: %u", bitPerPixel);
173  }
174 
175  // Sony E-550 marks compressed 8bpp ARW with 12 bit per pixel
176  // this makes the compression detect it as a ARW v1.
177  // This camera has however another MAKER entry, so we MAY be able
178  // to detect it this way in the future.
179  data = mRootIFD->getIFDsWithTag(MAKE);
180  if (data.size() > 1) {
181  for (auto &i : data) {
182  string make = i->getEntry(MAKE)->getString();
183  /* Check for maker "SONY" without spaces */
184  if (make == "SONY")
185  bitPerPixel = 8;
186  }
187  }
188 
189  if (width == 0 || height == 0 || height % 2 != 0 || width > 9600 ||
190  height > 6376)
191  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
192 
193  bool arw1 = uint64_t(counts->getU32()) * 8 != width * height * bitPerPixel;
194  if (arw1)
195  height += 8;
196 
197  mRaw->dim = iPoint2D(width, height);
198 
199  std::vector<uint16_t> curve(0x4001);
200  TiffEntry *c = raw->getEntry(SONY_CURVE);
201  std::array<uint32_t, 6> sony_curve = {{0, 0, 0, 0, 0, 4095}};
202 
203  for (uint32_t i = 0; i < 4; i++)
204  sony_curve[i+1] = (c->getU16(i) >> 2) & 0xfff;
205 
206  for (uint32_t i = 0; i < 0x4001; i++)
207  curve[i] = i;
208 
209  for (uint32_t i = 0; i < 5; i++)
210  for (uint32_t j = sony_curve[i] + 1; j <= sony_curve[i + 1]; j++)
211  curve[j] = curve[j-1] + (1 << i);
212 
213  RawImageCurveGuard curveHandler(&mRaw, curve, uncorrectedRawValues);
214 
215  uint32_t c2 = counts->getU32();
216  uint32_t off = offsets->getU32();
217 
218  if (!mFile->isValid(off))
219  ThrowRDE("Data offset after EOF, file probably truncated");
220 
221  if (!mFile->isValid(off, c2))
222  c2 = mFile->getSize() - off;
223 
224  ByteStream input(DataBuffer(mFile->getSubView(off, c2), Endianness::little));
225 
226  if (arw1) {
227  SonyArw1Decompressor a(mRaw);
228  mRaw->createData();
229  a.decompress(input);
230  } else
231  DecodeARW2(input, width, height, bitPerPixel);
232 
233  return mRaw;
234 }

236 void ArwDecoder::DecodeUncompressed(const TiffIFD* raw) {
237  uint32_t width = raw->getEntry(IMAGEWIDTH)->getU32();
238  uint32_t height = raw->getEntry(IMAGELENGTH)->getU32();
239  uint32_t off = raw->getEntry(STRIPOFFSETS)->getU32();
240  uint32_t c2 = raw->getEntry(STRIPBYTECOUNTS)->getU32();
241 
242  mRaw->dim = iPoint2D(width, height);
243 
244  if (width == 0 || height == 0 || width > 9600 || height > 6376)
245  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
246 
247  if (c2 == 0)
248  ThrowRDE("Strip is empty, nothing to decode!");
249 
250  const Buffer buf(mFile->getSubView(off, c2));
251 
252  mRaw->createData();
253 
254  UncompressedDecompressor u(ByteStream(DataBuffer(buf, Endianness::little)),
255  mRaw);
256 
257  if (hints.has("sr2_format"))
258  u.decodeRawUnpacked<14, Endianness::big>(width, height);
259  else
260  u.decodeRawUnpacked<16, Endianness::little>(width, height);
261 }

263 void ArwDecoder::DecodeARW2(const ByteStream& input, uint32_t w, uint32_t h,
264  uint32_t bpp) {
265 
266  if (bpp == 8) {
267  SonyArw2Decompressor a2(mRaw, input);
268  mRaw->createData();
269  a2.decompress();
270  return;
271  } // End bpp = 8
272 
273  if (bpp == 12) {
274  mRaw->createData();
275  UncompressedDecompressor u(
276  ByteStream(DataBuffer(input, Endianness::little)), mRaw);
277  u.decode12BitRaw<Endianness::little>(w, h);
278 
279  // Shift scales, since black and white are the same as compressed precision
280  mShiftDownScale = 2;
281  return;
282  }
283  ThrowRDE("Unsupported bit depth");
284 }

286 void ArwDecoder::ParseA100WB() {
287  if (!mRootIFD->hasEntryRecursive(DNGPRIVATEDATA))
288  return;
289 
290  // only contains the offset, not the length!
291  TiffEntry* priv = mRootIFD->getEntryRecursive(DNGPRIVATEDATA);
292  ByteStream bs = priv->getData();
293  bs.setByteOrder(Endianness::little);
294  const uint32_t off = bs.getU32();
295 
296  bs = ByteStream(DataBuffer(mFile->getSubView(off), Endianness::little));
297 
298  // MRW style, see MrwDecoder
299 
300  bs.setByteOrder(Endianness::big);
301  uint32_t tag = bs.getU32();
302  if (0x4D5249 != tag) // MRI
303  ThrowRDE("Can not parse DNGPRIVATEDATA, invalid tag (0x%x).", tag);
304 
305  bs.setByteOrder(Endianness::little);
306  uint32_t len = bs.getU32();
307 
308  bs = bs.getSubStream(bs.getPosition(), len);
309 
310  while (bs.getRemainSize() > 0) {
311  bs.setByteOrder(Endianness::big);
312  tag = bs.getU32();
313  bs.setByteOrder(Endianness::little);
314  len = bs.getU32();
315  bs.check(len);
316  if (!len)
317  ThrowRDE("Found entry of zero length, corrupt.");
318 
319  if (0x574247 != tag) { // WBG
320  // not the tag we are interested in, skip
321  bs.skipBytes(len);
322  continue;
323  }
324 
325  bs.skipBytes(4);
326 
327  bs.setByteOrder(Endianness::little);
328  std::array<uint16_t, 4> tmp;
329  for (auto& coeff : tmp)
330  coeff = bs.getU16();
331 
332  mRaw->metadata.wbCoeffs[0] = static_cast<float>(tmp[0]);
333  mRaw->metadata.wbCoeffs[1] = static_cast<float>(tmp[1]);
334  mRaw->metadata.wbCoeffs[2] = static_cast<float>(tmp[3]);
335 
336  // only need this one block, no need to process any further
337  break;
338  }
339 }

369 void ArwDecoder::SonyDecrypt(const uint32_t* ibuf, uint32_t* obuf, uint32_t len,
370  uint32_t key) {
371  if (0 == len)
372  return;
373 
374  std::array<uint32_t, 128> pad;
375 
376  // Initialize the decryption pad from the key
377  for (int p=0; p < 4; p++)
378  pad[p] = key = uint32_t(key * 48828125UL + 1UL);
379  pad[3] = pad[3] << 1 | (pad[0]^pad[2]) >> 31;
380  for (int p=4; p < 127; p++)
381  pad[p] = (pad[p-4]^pad[p-2]) << 1 | (pad[p-3]^pad[p-1]) >> 31;
382  for (int p=0; p < 127; p++)
383  pad[p] = getU32BE(&pad[p]);
384 
385  int p = 127;
386  // Decrypt the buffer in place using the pad
387  for (; len > 0; len--) {
388  pad[p & 127] = pad[(p+1) & 127] ^ pad[(p+1+64) & 127];
389 
390  uint32_t pv;
391  memcpy(&pv, &(pad[p & 127]), sizeof(uint32_t));
392 
393  uint32_t bv;
394  memcpy(&bv, ibuf, sizeof(uint32_t));
395 
396  bv ^= pv;
397 
398  memcpy(obuf, &bv, sizeof(uint32_t));
399 
400  ibuf++;
401  obuf++;
402  p++;
403  }

406 void ArwDecoder::GetWB() {
407  // Set the whitebalance for all the modern ARW formats (everything after A100)
408  if (mRootIFD->hasEntryRecursive(DNGPRIVATEDATA)) {
409  NORangesSet<Buffer> ifds_undecoded;
410 
411  TiffEntry *priv = mRootIFD->getEntryRecursive(DNGPRIVATEDATA);
412  TiffRootIFD makerNoteIFD(nullptr, &ifds_undecoded, priv->getRootIfdData(),
413  priv->getU32());
414 
415  TiffEntry *sony_offset = makerNoteIFD.getEntryRecursive(SONY_OFFSET);
416  TiffEntry *sony_length = makerNoteIFD.getEntryRecursive(SONY_LENGTH);
417  TiffEntry *sony_key = makerNoteIFD.getEntryRecursive(SONY_KEY);
418  if(!sony_offset || !sony_length || !sony_key || sony_key->count != 4)
419  ThrowRDE("couldn't find the correct metadata for WB decoding");
420 
421  assert(sony_offset != nullptr);
422  uint32_t off = sony_offset->getU32();
423 
424  assert(sony_length != nullptr);
425  // The Decryption is done in blocks of 4 bytes.
426  uint32_t len = roundDown(sony_length->getU32(), 4);
427 
428  assert(sony_key != nullptr);
429  uint32_t key = getU32LE(sony_key->getData(4));
430 
431  // "Decrypt" IFD
432  const auto& ifd_crypt = priv->getRootIfdData();
433  const auto EncryptedBuffer = ifd_crypt.getSubView(off, len);
434  // We do have to prepend 'off' padding, because TIFF uses absolute offsets.
435  const auto DecryptedBufferSize = off + EncryptedBuffer.getSize();
436  auto DecryptedBuffer = Buffer::Create(DecryptedBufferSize);
437 
438  SonyDecrypt(reinterpret_cast<const uint32_t*>(EncryptedBuffer.begin()),
439  reinterpret_cast<uint32_t*>(DecryptedBuffer.get() + off),
440  len / 4, key);
441 
442  NORangesSet<Buffer> ifds_decoded;
443  Buffer decIFD(std::move(DecryptedBuffer), DecryptedBufferSize);
444  const Buffer Padding(decIFD.getSubView(0, off));
445  // The Decrypted Root Ifd can not point to preceding padding buffer.
446  ifds_decoded.emplace(Padding);
447 
448  DataBuffer dbIDD(decIFD, priv->getRootIfdData().getByteOrder());
449  TiffRootIFD encryptedIFD(nullptr, &ifds_decoded, dbIDD, off);
450 
451  if (encryptedIFD.hasEntry(SONYGRBGLEVELS)){
452  TiffEntry *wb = encryptedIFD.getEntry(SONYGRBGLEVELS);
453  if (wb->count != 4)
454  ThrowRDE("WB has %d entries instead of 4", wb->count);
455  mRaw->metadata.wbCoeffs[0] = wb->getFloat(1);
456  mRaw->metadata.wbCoeffs[1] = wb->getFloat(0);
457  mRaw->metadata.wbCoeffs[2] = wb->getFloat(2);
458  } else if (encryptedIFD.hasEntry(SONYRGGBLEVELS)){
459  TiffEntry *wb = encryptedIFD.getEntry(SONYRGGBLEVELS);
460  if (wb->count != 4)
461  ThrowRDE("WB has %d entries instead of 4", wb->count);
462  mRaw->metadata.wbCoeffs[0] = wb->getFloat(0);
463  mRaw->metadata.wbCoeffs[1] = wb->getFloat(1);
464  mRaw->metadata.wbCoeffs[2] = wb->getFloat(3);
465  }
466  }
467 }
