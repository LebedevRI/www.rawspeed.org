74  // TIFF IFD are tree-like structure, with branches.
75  // A branch (IFD) can have branches (IFDs) of it's own.
76  // We must be careful to weed-out all the degenerative cases that
77  // can be produced e.g. via fuzzing, or other means.
78  struct Limits final {
79  // How many layers of IFD's can there be?
80  // All RPU samples (as of 2018-02-11) are ok with 4.
81  // However, let's be on the safe side, and pad it by one.
82  static constexpr int Depth = 4 + 1;
83 
84  // How many sub-IFD's can this IFD have?
85  // NOTE: only for the given IFD, *NOT* recursively including all sub-IFD's!
86  // All RPU samples (as of 2018-02-11) are ok with 5.
87  // However, let's be on the safe side, and double it.
88  static constexpr int SubIFDCount = 5 * 2;
89 
90  // How many sub-IFD's can this IFD have, recursively?
91  // All RPU samples (as of 2018-02-11) are ok with 14.
92  // However, let's be on the safe side, and double it.
93  static constexpr int RecursiveSubIFDCount = 14 * 2;
94  };

142 inline Endianness getTiffByteOrder(const ByteStream& bs, uint32_t pos,
143  const char* context = "") {
144  if (bs.hasPatternAt("II", 2, pos))
145  return Endianness::little;
146  if (bs.hasPatternAt("MM", 2, pos))
147  return Endianness::big;
148 
149  ThrowTPE("Failed to parse TIFF endianness information in %s.", context);
150 }
