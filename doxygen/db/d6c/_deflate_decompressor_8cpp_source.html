39 static inline void decodeFPDeltaRow(unsigned char* src, unsigned char* dst,
40  size_t tileWidth, size_t realTileWidth,
41  unsigned int bytesps, int factor) {
42  // DecodeDeltaBytes
43  for (size_t col = factor; col < realTileWidth * bytesps; ++col) {
44  // Yes, this is correct, and is symmetrical with EncodeDeltaBytes in
45  // hdrmerge, and they both combined are lossless.
46  // This is indeed working in modulo-2^n arighmetics.
47  src[col] = static_cast<unsigned char>(src[col] + src[col - factor]);
48  }
49  // Reorder bytes into the image
50  // 16 and 32-bit versions depend on local architecture, 24-bit does not
51  if (bytesps == 3) {
52  for (size_t col = 0; col < tileWidth; ++col) {
53  dst[col * 3] = src[col];
54  dst[col * 3 + 1] = src[col + realTileWidth];
55  dst[col * 3 + 2] = src[col + realTileWidth * 2];
56  }
57  } else {
58  if (getHostEndianness() == Endianness::little) {
59  for (size_t col = 0; col < tileWidth; ++col) {
60  for (size_t byte = 0; byte < bytesps; ++byte)
61  dst[col * bytesps + byte] =
62  src[col + realTileWidth * (bytesps - byte - 1)];
63  }
64  } else {
65  for (size_t col = 0; col < tileWidth; ++col) {
66  for (size_t byte = 0; byte < bytesps; ++byte)
67  dst[col * bytesps + byte] = src[col + realTileWidth * byte];
68  }
69  }
70  }
71 }

73 static inline void expandFP16(unsigned char* dst, int width) {
74  auto* dst16 = reinterpret_cast<uint16_t*>(dst);
75  auto* dst32 = reinterpret_cast<uint32_t*>(dst);
76 
77  for (int x = width - 1; x >= 0; x--)
78  dst32[x] = fp16ToFloat(dst16[x]);
79 }

81 static inline void expandFP24(unsigned char* dst, int width) {
82  auto* dst32 = reinterpret_cast<uint32_t*>(dst);
83  dst += (width - 1) * 3;
84  for (int x = width - 1; x >= 0; x--) {
85  dst32[x] = fp24ToFloat((dst[0] << 16) | (dst[1] << 8) | dst[2]);
86  dst -= 3;
87  }
88 }
