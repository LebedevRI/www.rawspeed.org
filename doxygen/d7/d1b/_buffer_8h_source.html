36 /*************************************************************************
37  * This is the buffer abstraction.
38  *
39  * It allows access to some piece of memory, typically a whole or part
40  * of a raw file. The underlying memory may be owned by the buffer or not.
41  * It supports move operations to properly deal with ownership transfer.
42  * It intentionally supports only read/const access to the underlying memory.
43  *
44  *************************************************************************/
45 class Buffer
46 {
47 public:
48  using size_type = uint32_t;
49 
50 protected:
51  const uint8_t* data = nullptr;
52  size_type size = 0;
53  bool isOwner = false;
54 
55 public:
56  // allocates the databuffer, and returns owning non-const pointer.
57  static std::unique_ptr<uint8_t, decltype(&alignedFree)>
58  Create(size_type size) {
59  if (!size)
60  ThrowIOE("Trying to allocate 0 bytes sized buffer.");
61 
62  std::unique_ptr<uint8_t, decltype(&alignedFree)> data(
63  alignedMalloc<uint8_t, 16>(roundUp(size, 16)),
64  &alignedFree);
65  if (!data)
66  ThrowIOE("Failed to allocate %uz bytes memory buffer.", size);
67 
68  assert(!ASan::RegionIsPoisoned(data.get(), size));
69 
70  return data;
71  }
72 
73  // constructs an empty buffer
74  Buffer() = default;
75 
76  // creates buffer from owning unique_ptr
77  Buffer(std::unique_ptr<uint8_t, decltype(&alignedFree)> data_,
78  size_type size_)
79  : size(size_) {
80  if (!size)
81  ThrowIOE("Buffer has zero size?");
82 
83  if (data_.get_deleter() != &alignedFree)
84  ThrowIOE("Wrong deleter. Expected rawspeed::alignedFree()");
85 
86  data = data_.release();
87  if (!data)
88  ThrowIOE("Memory buffer is nonexistent");
89 
90  assert(!ASan::RegionIsPoisoned(data, size));
91 
92  isOwner = true;
93  }
94 
95  // Data already allocated
96  explicit Buffer(const uint8_t* data_, size_type size_)
97  : data(data_), size(size_) {
98  assert(!ASan::RegionIsPoisoned(data, size));
99  }
100 
101  // creates a (non-owning) copy / view of rhs
102  Buffer(const Buffer& rhs) : data(rhs.data), size(rhs.size) {
103  assert(!ASan::RegionIsPoisoned(data, size));
104  }
105 
106  // Move data and ownership from rhs to this
107  Buffer(Buffer&& rhs) noexcept
108  : data(rhs.data), size(rhs.size), isOwner(rhs.isOwner) {
109  assert(!ASan::RegionIsPoisoned(data, size));
110  rhs.isOwner = false;
111  }
112 
113  // Frees memory if owned
114  ~Buffer() {
115  if (isOwner) {
116  alignedFreeConstPtr(data);
117  }
118  }
119 
120  Buffer& operator=(Buffer&& rhs) noexcept {
121  if (this == &rhs) {
122  assert(!ASan::RegionIsPoisoned(data, size));
123  return *this;
124  }
125 
126  if (isOwner)
127  alignedFreeConstPtr(data);
128 
129  data = rhs.data;
130  size = rhs.size;
131  isOwner = rhs.isOwner;
132 
133  assert(!ASan::RegionIsPoisoned(data, size));
134 
135  rhs.isOwner = false;
136 
137  return *this;
138  }
139 
140  Buffer& operator=(const Buffer& rhs) {
141  if (this == &rhs) {
142  assert(!ASan::RegionIsPoisoned(data, size));
143  return *this;
144  }
145 
146  Buffer unOwningTmp(rhs.data, rhs.size);
147  *this = std::move(unOwningTmp);
148  assert(!isOwner);
149  assert(!ASan::RegionIsPoisoned(data, size));
150 
151  return *this;
152  }

161  Buffer getSubView(size_type offset) const {
162  if (!isValid(0, offset))

164 
165  size_type newSize = size - offset;

169  // get pointer to memory at 'offset', make sure at least 'count' bytes are accessible
170  const uint8_t* getData(size_type offset, size_type count) const {
171  if (!isValid(offset, count))
172  ThrowIOE("Buffer overflow: image file may be truncated");
173 
174  assert(data);
175  assert(!ASan::RegionIsPoisoned(data + offset, count));
176 
177  return data + offset;
178  }
179 
180  // convenience getter for single bytes
181  uint8_t operator[](size_type offset) const { return *getData(offset, 1); }
182 
183  // std begin/end iterators to allow for range loop
184  const uint8_t* begin() const {
185  assert(data);
186  assert(!ASan::RegionIsPoisoned(data, 0));
187  return data;
188  }
189  const uint8_t* end() const {
190  assert(data);
191  assert(!ASan::RegionIsPoisoned(data, size));
192  return data + size;
193  }

195  // get memory of type T from byte offset 'offset + sizeof(T)*index' and swap byte order if required
196  template<typename T> inline T get(bool inNativeByteOrder, size_type offset, size_type index = 0) const {
197  return getByteSwapped<T>(
198  getData(offset + index * static_cast<size_type>(sizeof(T)),
199  static_cast<size_type>(sizeof(T))),
200  !inNativeByteOrder);
201  }
202 
203  inline size_type getSize() const {
204  assert(!ASan::RegionIsPoisoned(data, size));
205  return size;
206  }
207 
208  inline bool isValid(size_type offset, size_type count = 1) const {
209  return static_cast<uint64_t>(offset) + count <= static_cast<uint64_t>(size);
210  }

213 /*
214  * DataBuffer is a simple extension to Buffer. It knows about the byte order
215  * of its contents and can therefore provide save access to larger than
216  * byte sized data, like int, float, etc.
217  */
218 class DataBuffer : public Buffer {
219  // FIXME: default should be Endianness::unknown !
220 
221  Endianness endianness = Endianness::little;
222 
223 public:
224  DataBuffer() = default;
225 
226  explicit DataBuffer(const Buffer& data_, Endianness endianness_)
227  : Buffer(data_), endianness(endianness_) {}
228 
229  // get memory of type T from byte offset 'offset + sizeof(T)*index' and swap
230  // byte order if required
231  template <typename T>
232  inline T get(size_type offset, size_type index = 0) const {
233  assert(Endianness::unknown != endianness);
234  assert(Endianness::little == endianness || Endianness::big == endianness);
235 
236  return Buffer::get<T>(getHostEndianness() == endianness, offset, index);
237  }
238 
239  inline Endianness getByteOrder() const { return endianness; }
240 
241  inline Endianness setByteOrder(Endianness endianness_) {
242  std::swap(endianness, endianness_);
243  return endianness_;
244  }
245 };
