38 // simple 64-bit wide cache implementation that acts like a FiFo.
39 // There are two variants:
40 // * L->R: new bits are pushed in on the left and pulled out on the right
41 // * L<-R: new bits are pushed in on the right and pulled out on the left
42 // Each BitStream specialization uses one of the two.
43 
44 struct BitStreamCacheBase
45 {
46  uint64_t cache = 0; // the actual bits stored in the cache
47  unsigned int fillLevel = 0; // bits left in cache
48 
49  static constexpr unsigned Size = bitwidth<decltype(cache)>();
50 
51  // how many bits could be requested to be filled
52  static constexpr unsigned MaxGetBits = bitwidth<uint32_t>();
53 
54  // maximal number of bytes the implementation may read.
55  // NOTE: this is not the same as MaxGetBits/8 !!!
56  static constexpr unsigned MaxProcessBytes = 8;
57 };
58 
59 struct BitStreamCacheLeftInRightOut : BitStreamCacheBase
60 {
61  inline void push(uint64_t bits, uint32_t count) noexcept {
62  assert(count + fillLevel <= bitwidth(cache));
63  cache |= bits << fillLevel;
64  fillLevel += count;
65  }
66 
67  inline uint32_t peek(uint32_t count) const noexcept {
68  return cache & ((1U << count) - 1U);
69  }
70 
71  inline void skip(uint32_t count) noexcept {
72  cache >>= count;
73  fillLevel -= count;
74  }
75 };
76 
77 struct BitStreamCacheRightInLeftOut : BitStreamCacheBase
78 {
79  inline void push(uint64_t bits, uint32_t count) noexcept {
80  assert(count + fillLevel <= bitwidth(cache));
81  assert(count < bitwidth(cache));
82  cache = cache << count | bits;
83  fillLevel += count;
84  }
85 
86  inline uint32_t peek(uint32_t count) const noexcept {
87  return extractHighBits(cache, count, /*effectiveBitwidth=*/fillLevel) &
88  ((1U << count) - 1U);
89  }
90 
91  inline void skip(uint32_t count) noexcept { fillLevel -= count; }
92 };

99 class BitStream final : public ByteStream {

102  // A temporary intermediate buffer that may be used by fill() method either
103  // in debug build to enforce lack of out-of-bounds reads, or when we are
104  // nearing the end of the input buffer and can not just read MaxProcessBytes
105  // from it, but have to read as much as we can and fill rest with zeros.
106  std::array<uint8_t, BitStreamCacheBase::MaxProcessBytes> tmp = {};
107 
108  // this method hase to be implemented in the concrete BitStream template
109  // specializations. It will return the number of bytes processed. It needs
110  // to process up to BitStreamCacheBase::MaxProcessBytes bytes of input.
111  size_type fillCache(const uint8_t* input, size_type bufferSize,
112  size_type* bufPos);
113 
114 public:
115  BitStream() = default;
116 
117  explicit BitStream(const ByteStream& s)
118  : ByteStream(s.getSubStream(s.getPosition(), s.getRemainSize())) {
119  setByteOrder(Endianness::unknown);
120  }
121 
122 private:
123  inline const uint8_t* getInput() {
124  assert(data);
125 
126 #if !defined(DEBUG)
127  // Do we have MaxProcessBytes or more bytes left in the input buffer?
128  // If so, then we can just read from said buffer.
129  if (pos + BitStreamCacheBase::MaxProcessBytes <= size)
130  return data + pos;
131 #endif
132 
133  // We have to use intermediate buffer, either because the input is running
134  // out of bytes, or because we want to enforce bounds checking.
135 
136  // Note that in order to keep all fill-level invariants we must allow to
137  // over-read past-the-end a bit.
138  if (pos > size + BitStreamCacheBase::MaxProcessBytes)
139  ThrowIOE("Buffer overflow read in BitStream");
140 
141  tmp.fill(0);
142 
143  // How many bytes are left in input buffer?
144  // Since pos can be past-the-end we need to carefully handle overflow.
145  Buffer::size_type bytesRemaining = (pos < size) ? size - pos : 0;
146  // And if we are not at the end of the input, we may have more than we need.
147  bytesRemaining =
148  std::min(BitStreamCacheBase::MaxProcessBytes, bytesRemaining);
149 
150  memcpy(tmp.data(), data + pos, bytesRemaining);
151  return tmp.data();
152  }
153 
154 public:
155  inline void fill(uint32_t nbits = Cache::MaxGetBits) {
156  assert(data);
157  assert(nbits <= Cache::MaxGetBits);
158 
159  if (cache.fillLevel >= nbits)
160  return;
161 
162  pos += fillCache(getInput(), size, &pos);
163  }
164 
165  // these methods might be specialized by implementations that support it
166  inline size_type getBufferPosition() const {
167  return pos - (cache.fillLevel >> 3);
168  }
169 
170  inline size_type getFillLevel() const { return cache.fillLevel; }
171 
172  // rewinds to the beginning of the buffer.
173  void resetBufferPosition() {
174  pos = 0;
175  cache.fillLevel = 0;
176  cache.cache = 0;
177  }
178 
179  void setBufferPosition(size_type newPos);
180 
181  inline uint32_t __attribute__((pure)) peekBitsNoFill(uint32_t nbits) {
182  assert(nbits != 0);
183  assert(nbits < Cache::MaxGetBits);
184  assert(nbits <= cache.fillLevel);
185  return cache.peek(nbits);
186  }
187 
188  inline void skipBitsNoFill(uint32_t nbits) {
189  assert(nbits <= Cache::MaxGetBits);
190  assert(nbits <= cache.fillLevel);
191  cache.skip(nbits);
192  }
193 
194  inline uint32_t getBitsNoFill(uint32_t nbits) {
195  uint32_t ret = peekBitsNoFill(nbits);
196  skipBitsNoFill(nbits);
197  return ret;
198  }
199 
200  inline uint32_t peekBits(uint32_t nbits) {
201  fill(nbits);
202  return peekBitsNoFill(nbits);
203  }
204 
205  inline uint32_t getBits(uint32_t nbits) {
206  fill(nbits);
207  return getBitsNoFill(nbits);
208  }
209 
210  // This may be used to skip arbitrarily large number of *bytes*,
211  // not limited by the fill level.
212  inline void skipBytes(uint32_t nbytes) {
213  uint32_t remainingBitsToSkip = 8 * nbytes;
214  for (; remainingBitsToSkip >= Cache::MaxGetBits;
215  remainingBitsToSkip -= Cache::MaxGetBits) {
216  fill(Cache::MaxGetBits);
217  skipBitsNoFill(Cache::MaxGetBits);
218  }
219  if (remainingBitsToSkip > 0) {
220  fill(remainingBitsToSkip);
221  skipBitsNoFill(remainingBitsToSkip);
222  }
223  }
224 };
