38 // simple 64-bit wide cache implementation that acts like a FiFo.
39 // There are two variants:
40 // * L->R: new bits are pushed in on the left and pulled out on the right
41 // * L<-R: new bits are pushed in on the right and pulled out on the left
42 // Each BitStream specialization uses one of the two.
43 
44 struct BitStreamCacheBase
45 {
46  uint64_t cache = 0; // the actual bits stored in the cache
47  unsigned int fillLevel = 0; // bits left in cache
48 
49  static constexpr unsigned Size = bitwidth<decltype(cache)>();
50 
51  // how many bits could be requested to be filled
52  static constexpr unsigned MaxGetBits = bitwidth<uint32_t>();
53 
54  // maximal number of bytes the implementation may read.
55  // NOTE: this is not the same as MaxGetBits/8 !!!
56  static constexpr unsigned MaxProcessBytes = 8;
57 };

59 struct BitStreamCacheLeftInRightOut : BitStreamCacheBase
60 {
61  inline void push(uint64_t bits, uint32_t count) noexcept {
62  assert(count + fillLevel <= bitwidth(cache));
63  cache |= bits << fillLevel;
64  fillLevel += count;
65  }
66 
67  inline uint32_t peek(uint32_t count) const noexcept {
68  return cache & ((1U << count) - 1U);
69  }
70 
71  inline void skip(uint32_t count) noexcept {
72  cache >>= count;
73  fillLevel -= count;
74  }
75 };
76 
77 struct BitStreamCacheRightInLeftOut : BitStreamCacheBase
78 {
79  inline void push(uint64_t bits, uint32_t count) noexcept {
80  assert(count + fillLevel <= Size);
81  assert(count != 0);
82  // If the maximal size of the cache is BitStreamCacheBase::Size, and we
83  // have fillLevel [high] bits set, how many empty [low] bits do we have?
84  const uint32_t vacantBits = BitStreamCacheBase::Size - fillLevel;
85  // If we just directly 'or' these low bits into the cache right now,
86  // how many unfilled bits of a gap will there be in the middle of a cache?
87  const uint32_t emptyBitsGap = vacantBits - count;
88  // So just shift the new bits so that there is no gap in the middle.
89  cache |= bits << emptyBitsGap;
90  fillLevel += count;
91  }
92 
93  inline uint32_t peek(uint32_t count) const noexcept {
94  return extractHighBits(cache, count, /*effectiveBitwidth=*/BitStreamCacheBase::Size);
95  }
96 
97  inline void skip(uint32_t count) noexcept {
98  fillLevel -= count;
99  cache <<= count;
100  }
101 };

103 struct BitStreamReplenisherBase {
104  using size_type = uint32_t;
105 
106  const uint8_t* data;
107  size_type size;
108  unsigned pos = 0;
109 
110  BitStreamReplenisherBase() = default;
111 
112  explicit BitStreamReplenisherBase(const Buffer& input)
113  : data(input.getData(0, input.getSize())), size(input.getSize()) {}
114 
115  // A temporary intermediate buffer that may be used by fill() method either
116  // in debug build to enforce lack of out-of-bounds reads, or when we are
117  // nearing the end of the input buffer and can not just read MaxProcessBytes
118  // from it, but have to read as much as we can and fill rest with zeros.
119  std::array<uint8_t, BitStreamCacheBase::MaxProcessBytes> tmp = {};
120 };

122 struct BitStreamForwardSequentialReplenisher final : BitStreamReplenisherBase {
123  BitStreamForwardSequentialReplenisher() = default;
124 
125  explicit BitStreamForwardSequentialReplenisher(const Buffer& input)
126  : BitStreamReplenisherBase(input) {}
127 
128  inline size_type getPos() const { return pos; }
129  inline size_type getRemainingSize() const { return size - getPos(); }
130  inline void markNumBytesAsConsumed(size_type numBytes) { pos += numBytes; }
131 
132  inline const uint8_t* getInput() {
133 #if !defined(DEBUG)
134  // Do we have MaxProcessBytes or more bytes left in the input buffer?
135  // If so, then we can just read from said buffer.
136  if (pos + BitStreamCacheBase::MaxProcessBytes <= size)
137  return data + pos;
138 #endif
139 
140  // We have to use intermediate buffer, either because the input is running
141  // out of bytes, or because we want to enforce bounds checking.
142 
143  // Note that in order to keep all fill-level invariants we must allow to
144  // over-read past-the-end a bit.
145  if (pos > size + BitStreamCacheBase::MaxProcessBytes)
146  ThrowIOE("Buffer overflow read in BitStream");
147 
148  tmp.fill(0);
149 
150  // How many bytes are left in input buffer?
151  // Since pos can be past-the-end we need to carefully handle overflow.
152  size_type bytesRemaining = (pos < size) ? size - pos : 0;
153  // And if we are not at the end of the input, we may have more than we need.
154  bytesRemaining =
155  std::min(BitStreamCacheBase::MaxProcessBytes, bytesRemaining);
156 
157  memcpy(tmp.data(), data + pos, bytesRemaining);
158  return tmp.data();
159  }
160 };

166 template <typename Tag, typename Cache,
167  typename Replenisher = BitStreamForwardSequentialReplenisher>
168 class BitStream final {
169  Cache cache;
170 
171  Replenisher replenisher;
172 
173  using size_type = uint32_t;
174 
175  // this method hase to be implemented in the concrete BitStream template
176  // specializations. It will return the number of bytes processed. It needs
177  // to process up to BitStreamCacheBase::MaxProcessBytes bytes of input.
178  size_type fillCache(const uint8_t* input);
179 
180 public:
181  BitStream() = default;
182 
183  explicit BitStream(const Buffer& buf) : replenisher(buf) {}
184 
185  explicit BitStream(const ByteStream& s)
186  : BitStream(s.getSubView(s.getPosition(), s.getRemainSize())) {}

188  inline void fill(uint32_t nbits = Cache::MaxGetBits) {
189  assert(nbits <= Cache::MaxGetBits);
190 
191  if (cache.fillLevel >= nbits)
192  return;
193 
194  replenisher.markNumBytesAsConsumed(fillCache(replenisher.getInput()));
195  }
196 
197  // these methods might be specialized by implementations that support it
198  inline size_type getInputPosition() const { return replenisher.getPos(); }
199 
200  // these methods might be specialized by implementations that support it
201  inline size_type getStreamPosition() const {
202  return getInputPosition() - (cache.fillLevel >> 3);
203  }
204 
205  inline size_type getRemainingSize() const {
206  return replenisher.getRemainingSize();
207  }
208 
209  inline size_type getFillLevel() const { return cache.fillLevel; }

211  inline uint32_t __attribute__((pure)) peekBitsNoFill(uint32_t nbits) {
212  assert(nbits != 0);
213  assert(nbits < Cache::MaxGetBits);
214  assert(nbits <= cache.fillLevel);
215  return cache.peek(nbits);
216  }
217 
218  inline void skipBitsNoFill(uint32_t nbits) {
219  assert(nbits <= Cache::MaxGetBits);
220  assert(nbits <= cache.fillLevel);
221  cache.skip(nbits);
222  }
223 
224  inline uint32_t getBitsNoFill(uint32_t nbits) {
225  uint32_t ret = peekBitsNoFill(nbits);
226  skipBitsNoFill(nbits);
227  return ret;
228  }
229 
230  inline uint32_t peekBits(uint32_t nbits) {
231  fill(nbits);
232  return peekBitsNoFill(nbits);
233  }
234 
235  inline uint32_t getBits(uint32_t nbits) {
236  fill(nbits);
237  return getBitsNoFill(nbits);
238  }
239 
240  // This may be used to skip arbitrarily large number of *bytes*,
241  // not limited by the fill level.
242  inline void skipBytes(uint32_t nbytes) {
243  uint32_t remainingBitsToSkip = 8 * nbytes;
244  for (; remainingBitsToSkip >= Cache::MaxGetBits;
245  remainingBitsToSkip -= Cache::MaxGetBits) {
246  fill(Cache::MaxGetBits);
247  skipBitsNoFill(Cache::MaxGetBits);
248  }
249  if (remainingBitsToSkip > 0) {
250  fill(remainingBitsToSkip);
251  skipBitsNoFill(remainingBitsToSkip);
252  }
253  }
254 };
