203 auto GenOnesLE = [](int zerosToOutput,
204  int zerosOutputted) -> std::array<uint32_t, 29> {
205  std::array<uint32_t, 29> v;
206  uint32_t bits = 0;
207  int currBit = -1;
208  for (auto& value : v) {
209  if (zerosToOutput == zerosOutputted) {
210  assert(currBit < 32);
211  bits |= 0b1 << currBit;
212  zerosToOutput++;
213  zerosOutputted = 0;
214  }
215  value = bits;
216  zerosOutputted++;
217  currBit++;
218  }
219  return v;
220 };
221 auto GenOnesBE = [](int zerosToOutput,
222  int zerosOutputted) -> std::array<uint32_t, 29> {
223  std::array<uint32_t, 29> v;
224  uint32_t bits = 0;
225  for (auto& value : v) {
226  if (zerosToOutput == zerosOutputted) {
227  bits |= 0b1;
228  zerosToOutput++;
229  zerosOutputted = 0;
230  }
231  value = bits;
232  zerosOutputted++;
233  bits <<= 1;
234  }
235  return v;
236 };
