58 void SamsungV0Decompressor::computeStripes(ByteStream bso, ByteStream bsr) {
59  const uint32_t height = mRaw->dim.y;
60 
61  std::vector<uint32_t> offsets;
62  offsets.reserve(1 + height);
63  for (uint32_t y = 0; y < height; y++)
64  offsets.emplace_back(bso.getU32());
65  offsets.emplace_back(bsr.getSize());
66 
67  stripes.reserve(height);
68 
69  auto offset_iterator = std::begin(offsets);
70  bsr.skipBytes(*offset_iterator);
71 
72  auto next_offset_iterator = std::next(offset_iterator);
73  while (next_offset_iterator < std::end(offsets)) {
74  if (*offset_iterator >= *next_offset_iterator)
75  ThrowRDE("Line offsets are out of sequence or slice is empty.");
76 
77  const auto size = *next_offset_iterator - *offset_iterator;
78  assert(size > 0);
79 
80  stripes.emplace_back(bsr.getStream(size));
81 
82  std::advance(offset_iterator, 1);
83  std::advance(next_offset_iterator, 1);
84  }
85 
86  assert(stripes.size() == height);
87 }

89 void SamsungV0Decompressor::decompress() const {
90  for (int row = 0; row < mRaw->dim.y; row++)
91  decompressStrip(row, stripes[row]);
92 
93  // Swap red and blue pixels to get the final CFA pattern
94  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
95  for (int row = 0; row < out.height - 1; row += 2) {
96  for (int col = 0; col < out.width - 1; col += 2)
97  std::swap(out(row, col + 1), out(row + 1, col));
98  }

107 void SamsungV0Decompressor::decompressStrip(int row,
108  const ByteStream& bs) const {
109  const Array2DRef<uint16_t> out(mRaw->getU16DataAsUncroppedArray2DRef());
110  assert(out.width > 0);
111 
112  BitPumpMSB32 bits(bs);
113 
114  std::array<int, 4> len;
115  for (int& i : len)
116  i = row < 2 ? 7 : 4;
117 
118  // Image is arranged in groups of 16 pixels horizontally
119  for (int col = 0; col < out.width; col += 16) {
120  bits.fill();
121  bool dir = !!bits.getBitsNoFill(1);
122 
123  std::array<int, 4> op;
124  for (int& i : op)
125  i = bits.getBitsNoFill(2);
126 
127  for (int i = 0; i < 4; i++) {
128  assert(op[i] >= 0 && op[i] <= 3);
129 
130  switch (op[i]) {
131  case 3:
132  len[i] = bits.getBits(4);
133  break;
134  case 2:
135  len[i]--;
136  break;
137  case 1:
138  len[i]++;
139  break;
140  default:
141  // FIXME: it can be zero too.
142  break;
143  }
144 
145  if (len[i] < 0)
146  ThrowRDE("Bit length less than 0.");
147  if (len[i] > 16)
148  ThrowRDE("Bit Length more than 16.");
149  }
150 
151  if (dir) {
152  // Upward prediction
153 
154  if (row < 2)
155  ThrowRDE("Upward prediction for the first two rows. Raw corrupt");
156 
157  if (col + 16 >= out.width)
158  ThrowRDE("Upward prediction for the last block of pixels. Raw corrupt");
159 
160  // First we decode even pixels
161  for (int c = 0; c < 16; c += 2) {
162  int b = len[c >> 3];
163  int32_t adj = calcAdj(&bits, b);
164 
165  out(row, col + c) = adj + out(row - 1, col + c);
166  }
167 
168  // Now we decode odd pixels
169  // Why on earth upward prediction only looks up 1 line above
170  // is beyond me, it will hurt compression a deal.
171  for (int c = 1; c < 16; c += 2) {
172  int b = len[2 | (c >> 3)];
173  int32_t adj = calcAdj(&bits, b);
174 
175  out(row, col + c) = adj + out(row - 2, col + c);
176  }
177  } else {
178  // Left to right prediction
179  // First we decode even pixels
180  int pred_left = col != 0 ? out(row, col - 2) : 128;
181  for (int c = 0; c < 16; c += 2) {
182  int b = len[c >> 3];
183  int32_t adj = calcAdj(&bits, b);
184 
185  if (col + c < out.width)
186  out(row, col + c) = adj + pred_left;
187  }
188 
189  // Now we decode odd pixels
190  pred_left = col != 0 ? out(row, col - 1) : 128;
191  for (int c = 1; c < 16; c += 2) {
192  int b = len[2 | (c >> 3)];
193  int32_t adj = calcAdj(&bits, b);
194 
195  if (col + c < out.width)
196  out(row, col + c) = adj + pred_left;
197  }
198  }
199  }
200 }
