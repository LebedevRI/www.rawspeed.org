35 // The length of the sha1 digest (160-bit, 40 hexadecimal chars).
36 constexpr auto Sha1CheckSumLength = 40;
37 // The separator after the digest and before filename.
38 // Should be either " " or " b".
39 constexpr auto CheckSumSeparatorWidth = 2;

41 ChecksumFileEntry ParseChecksumFileLine(const std::string& Line,
42  const std::string& RootDir) {
43  ChecksumFileEntry Entry;
44 
45  // We are just assuming that the checksum file is correct and valid.
46  // It is up to user to validate it first (via actually running `sha1sum -c`).
47 
48  static constexpr auto Offset = Sha1CheckSumLength + CheckSumSeparatorWidth;
49 
50  if (Line.size() <= Offset)
51  ThrowRSE("Malformed checksum line: \"%s\"", Line.c_str());
52 
53  Entry.RelFileName = Line.substr(Offset);
54  assert(!Entry.RelFileName.empty());
55  assert(Entry.RelFileName.back() != '\n');
56 
57  Entry.FullFileName = RootDir + "/" + Entry.RelFileName;
58 
59  return Entry;
60 }

64 std::vector<ChecksumFileEntry>
65 ParseChecksumFileContent(const std::string& ChecksumFileContent,
66  const std::string& RootDir) {
67  std::vector<ChecksumFileEntry> Listing;
68 
69  const std::vector<std::string> Lines = splitString(ChecksumFileContent, '\n');
70 
71  Listing.reserve(Lines.size());
72 
73  for (const auto& Line : Lines) {
74  assert(!Line.empty());
75  Listing.emplace_back(ParseChecksumFileLine(Line, RootDir));
76  }
77 
78  return Listing;
79 }
