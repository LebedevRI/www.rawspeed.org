64 class DngOpcodes::FixBadPixelsConstant final : public DngOpcodes::DngOpcode {
65  uint32_t value;
66 
67 public:
68  explicit FixBadPixelsConstant(const RawImage& ri, ByteStream* bs)
69  : value(bs->getU32()) {
70  bs->getU32(); // Bayer Phase not used
71  }
72 
73  void setup(const RawImage& ri) override {
74  // These limitations are present within the DNG SDK as well.
75  if (ri->getDataType() != TYPE_USHORT16)
76  ThrowRDE("Only 16 bit images supported");
77 
78  if (ri->getCpp() > 1)
79  ThrowRDE("Only 1 component images supported");
80  }
81 
82  void apply(const RawImage& ri) override {
83  MutexLocker guard(&ri->mBadPixelMutex);
84  iPoint2D crop = ri->getCropOffset();
85  uint32_t offset = crop.x | (crop.y << 16);
86  for (auto y = 0; y < ri->dim.y; ++y) {
87  auto* src = reinterpret_cast<uint16_t*>(ri->getData(0, y));
88  for (auto x = 0; x < ri->dim.x; ++x) {
89  if (src[x] == value)
90  ri->mBadPixelPositions.push_back(offset + (y << 16 | x));
91  }
92  }
93  }
94 };

98 class DngOpcodes::ROIOpcode : public DngOpcodes::DngOpcode {
99  iRectangle2D roi;
100 
101 protected:
102  explicit ROIOpcode(const RawImage& ri, ByteStream* bs, bool minusOne) {
103  const iRectangle2D fullImage =
104  minusOne ? iRectangle2D(0, 0, ri->dim.x - 1, ri->dim.y - 1)
105  : iRectangle2D(0, 0, ri->dim.x, ri->dim.y);
106 
107  uint32_t top = bs->getU32();
108  uint32_t left = bs->getU32();
109  uint32_t bottom = bs->getU32();
110  uint32_t right = bs->getU32();
111 
112  const iPoint2D topLeft(left, top);
113  const iPoint2D bottomRight(right, bottom);
114 
115  if (!(fullImage.isPointInsideInclusive(topLeft) &&
116  fullImage.isPointInsideInclusive(bottomRight) &&
117  bottomRight >= topLeft)) {
118  ThrowRDE("Rectangle (%u, %u, %u, %u) not inside image (%u, %u, %u, %u).",
119  topLeft.x, topLeft.y, bottomRight.x, bottomRight.y,
120  fullImage.getTopLeft().x, fullImage.getTopLeft().y,
121  fullImage.getBottomRight().x, fullImage.getBottomRight().y);
122  }
123 
124  roi.setTopLeft(topLeft);
125  roi.setBottomRightAbsolute(bottomRight);
126  assert(roi.isThisInside(fullImage));
127  }
128 
129  const iRectangle2D& __attribute__((pure)) getRoi() const { return roi; }
130 };

151 class DngOpcodes::FixBadPixelsList final : public DngOpcodes::DngOpcode {
152  std::vector<uint32_t> badPixels;
153 
154 public:
155  explicit FixBadPixelsList(const RawImage& ri, ByteStream* bs) {
156  const iRectangle2D fullImage(0, 0, ri->getUncroppedDim().x - 1,
157  ri->getUncroppedDim().y - 1);
158 
159  bs->getU32(); // Skip phase - we don't care
160  auto badPointCount = bs->getU32();
161  auto badRectCount = bs->getU32();
162 
163  // first, check that we indeed have much enough data
164  const auto origPos = bs->getPosition();
165  bs->skipBytes(badPointCount, 2 * 4);
166  bs->skipBytes(badRectCount, 4 * 4);
167  bs->setPosition(origPos);
168 
169  // Read points
170  badPixels.reserve(badPixels.size() + badPointCount);
171  for (auto i = 0U; i < badPointCount; ++i) {
172  auto y = bs->getU32();
173  auto x = bs->getU32();
174 
175  const iPoint2D badPoint(x, y);
176  if (!fullImage.isPointInsideInclusive(badPoint))
177  ThrowRDE("Bad point not inside image.");
178 
179  badPixels.emplace_back(y << 16 | x);
180  }
181 
182  // Read rects
183  for (auto i = 0U; i < badRectCount; ++i) {
184  const DummyROIOpcode dummy(ri, bs);
185 
186  const iRectangle2D badRect = dummy.getRoi();
187  assert(badRect.isThisInside(fullImage));
188 
189  auto area = (1 + badRect.getHeight()) * (1 + badRect.getWidth());
190  badPixels.reserve(badPixels.size() + area);
191  for (auto y = badRect.getTop(); y <= badRect.getBottom(); ++y) {
192  for (auto x = badRect.getLeft(); x <= badRect.getRight(); ++x) {
193  badPixels.emplace_back(y << 16 | x);
194  }
195  }
196  }
197  }
198 
199  void apply(const RawImage& ri) override {
200  MutexLocker guard(&ri->mBadPixelMutex);
201  ri->mBadPixelPositions.insert(ri->mBadPixelPositions.begin(),
202  badPixels.begin(), badPixels.end());
203  }
204 };

218 class DngOpcodes::PixelOpcode : public ROIOpcode {
219  uint32_t firstPlane;
220  uint32_t planes;
221  uint32_t rowPitch;
222  uint32_t colPitch;
223 
224 protected:
225  explicit PixelOpcode(const RawImage& ri, ByteStream* bs)
226  : ROIOpcode(ri, bs, false), firstPlane(bs->getU32()),
227  planes(bs->getU32()) {
228 
229  if (planes == 0 || firstPlane > ri->getCpp() || planes > ri->getCpp() ||
230  firstPlane + planes > ri->getCpp()) {
231  ThrowRDE("Bad plane params (first %u, num %u), got planes = %u",
232  firstPlane, planes, ri->getCpp());
233  }
234 
235  rowPitch = bs->getU32();
236  colPitch = bs->getU32();
237 
238  const iRectangle2D& ROI = getRoi();
239 
240  if (rowPitch < 1 || rowPitch > static_cast<uint32_t>(ROI.getHeight()) ||
241  colPitch < 1 || colPitch > static_cast<uint32_t>(ROI.getWidth()))
242  ThrowRDE("Invalid pitch");
243  }
244 
245  // traverses the current ROI and applies the operation OP to each pixel,
246  // i.e. each pixel value v is replaced by op(x, y, v), where x/y are the
247  // coordinates of the pixel value v.
248  template <typename T, typename OP> void applyOP(const RawImage& ri, OP op) {
249  int cpp = ri->getCpp();
250  const iRectangle2D& ROI = getRoi();
251  for (auto y = ROI.getTop(); y < ROI.getBottom(); y += rowPitch) {
252  auto* src = reinterpret_cast<T*>(ri->getData(0, y));
253  // Add offset, so this is always first plane
254  src += firstPlane;
255  // FIXME: is op() really supposed to receive global image coordinates,
256  // and not [0..ROI.getHeight()-1][0..ROI.getWidth()-1] ?
257  for (auto x = ROI.getLeft(); x < ROI.getRight(); x += colPitch) {
258  for (auto p = 0U; p < planes; ++p)
259  src[x * cpp + p] = op(x, y, src[x * cpp + p]);
260  }
261  }
262  }
263 };

289 public:
290  explicit TableMap(const RawImage& ri, ByteStream* bs) : LookupOpcode(ri, bs) {
291  auto count = bs->getU32();
292 
293  if (count == 0 || count > 65536)
294  ThrowRDE("Invalid size of lookup table");

306 class DngOpcodes::PolynomialMap final : public LookupOpcode {
307 public:
308  explicit PolynomialMap(const RawImage& ri, ByteStream* bs)
309  : LookupOpcode(ri, bs) {
310  vector<double> polynomial;
311 
312  const auto polynomial_size = bs->getU32() + 1UL;
313  bs->check(8UL * polynomial_size);
314  if (polynomial_size > 9)
315  ThrowRDE("A polynomial with more than 8 degrees not allowed");
316 
317  polynomial.reserve(polynomial_size);
318  std::generate_n(std::back_inserter(polynomial), polynomial_size,
319  [&bs]() { return bs->get<double>(); });
320 
321  // Create lookup
322  lookup.resize(65536);
323  for (auto i = 0UL; i < lookup.size(); ++i) {
324  double val = polynomial[0];
325  for (auto j = 1UL; j < polynomial.size(); ++j)
326  val += polynomial[j] * pow(i / 65536.0, j);
327  lookup[i] = (clampBits(static_cast<int>(val * 65535.5), 16));
328  }
329  }
330 };

348 template <typename S>
349 class DngOpcodes::DeltaRowOrCol : public DeltaRowOrColBase {
350 public:
351  void setup(const RawImage& ri) override {
352  PixelOpcode::setup(ri);
353 
354  // If we are working on a float image, no need to convert to int
355  if (ri->getDataType() != TYPE_USHORT16)
356  return;
357 
358  deltaI.reserve(deltaF.size());
359  for (const auto f : deltaF) {
360  if (!valueIsOk(f))
361  ThrowRDE("Got float %f which is unacceptable.", f);
362  deltaI.emplace_back(static_cast<int>(f2iScale * f));
363  }
364  }
365 
366 protected:
367  const float f2iScale;
368  vector<float> deltaF;
369  vector<int> deltaI;
370 
371  // only meaningful for uint16_t images!
372  virtual bool valueIsOk(float value) = 0;
373 
374  DeltaRowOrCol(const RawImage& ri, ByteStream* bs, float f2iScale_)
375  : DeltaRowOrColBase(ri, bs), f2iScale(f2iScale_) {
376  const auto deltaF_count = bs->getU32();
377  bs->check(deltaF_count, 4);
378 
379  // See PixelOpcode::applyOP(). We will access deltaF/deltaI up to (excl.)
380  // either ROI.getRight() or ROI.getBottom() index. Thus, we need to have
381  // either ROI.getRight() or ROI.getBottom() elements in there.
382  // FIXME: i guess not strictly true with pitch != 1.
383  const auto expectedSize =
384  S::select(getRoi().getRight(), getRoi().getBottom());
385  if (expectedSize != deltaF_count) {
386  ThrowRDE("Got unexpected number of elements (%u), expected %u.",
387  expectedSize, deltaF_count);
388  }
389 
390  deltaF.reserve(deltaF_count);
391  std::generate_n(std::back_inserter(deltaF), deltaF_count, [&bs]() {
392  const auto F = bs->get<float>();
393  if (!std::isfinite(F))
394  ThrowRDE("Got bad float %f.", F);
395  return F;
396  });
397  }
398 };

402 template <typename S>
403 class DngOpcodes::OffsetPerRowOrCol final : public DeltaRowOrCol<S> {
404  // We have pixel value in range of [0..65535]. We apply some offset X.
405  // For this to generate a value within the same range , the offset X needs
406  // to have an absolute value of 65535. Since the offset is multiplied
407  // by f2iScale before applying, we need to divide by f2iScale here.
408  const double absLimit;
409 
410  bool valueIsOk(float value) final { return std::abs(value) <= absLimit; }
411 
412 public:
413  explicit OffsetPerRowOrCol(const RawImage& ri, ByteStream* bs)
414  : DeltaRowOrCol<S>(ri, bs, 65535.0F),
415  absLimit(double(std::numeric_limits<uint16_t>::max()) /
416  this->f2iScale) {}
417 
418  void apply(const RawImage& ri) override {
419  if (ri->getDataType() == TYPE_USHORT16) {
420  this->template applyOP<uint16_t>(
421  ri, [this](uint32_t x, uint32_t y, uint16_t v) {
422  return clampBits(this->deltaI[S::select(x, y)] + v, 16);
423  });
424  } else {
425  this->template applyOP<float>(ri,
426  [this](uint32_t x, uint32_t y, float v) {
427  return this->deltaF[S::select(x, y)] + v;
428  });
429  }
430  }
431 };

433 template <typename S>
434 class DngOpcodes::ScalePerRowOrCol final : public DeltaRowOrCol<S> {
435  // We have pixel value in range of [0..65535]. We scale by float X.
436  // For this to generate a value within the same range, the scale X needs
437  // to be in the range [0..65535]. However, we are operating with 32-bit
438  // signed integer space, so the new value can not be larger than 2^31,
439  // else we'd have signed integer overflow. Since the offset is multiplied
440  // by f2iScale before applying, we need to divide by f2iScale here.
441  static constexpr const double minLimit = 0.0;
442  static constexpr int rounding = 512;
443  const double maxLimit;
444 
445  bool valueIsOk(float value) final {
446  return value >= minLimit && value <= maxLimit;
447  }
448 
449 public:
450  explicit ScalePerRowOrCol(const RawImage& ri, ByteStream* bs)
451  : DeltaRowOrCol<S>(ri, bs, 1024.0F),
452  maxLimit((double(std::numeric_limits<int>::max() - rounding) /
453  double(std::numeric_limits<uint16_t>::max())) /
454  this->f2iScale) {}
455 
456  void apply(const RawImage& ri) override {
457  if (ri->getDataType() == TYPE_USHORT16) {
458  this->template applyOP<uint16_t>(ri, [this](uint32_t x, uint32_t y,
459  uint16_t v) {
460  return clampBits((this->deltaI[S::select(x, y)] * v + 512) >> 10, 16);
461  });
462  } else {
463  this->template applyOP<float>(ri,
464  [this](uint32_t x, uint32_t y, float v) {
465  return this->deltaF[S::select(x, y)] * v;
466  });
467  }
468  }
469 };

473 DngOpcodes::DngOpcodes(const RawImage& ri, TiffEntry* entry) {
474  ByteStream bs = entry->getData();
475 
476  // DNG opcodes are always stored in big-endian byte order.
477  bs.setByteOrder(Endianness::big);
478 
479  const auto opcode_count = bs.getU32();
480  auto origPos = bs.getPosition();
481 
482  // validate opcode count. we either have to do this, or we can't preallocate
483  for (auto i = 0U; i < opcode_count; i++) {
484  bs.skipBytes(4); // code
485  bs.skipBytes(4); // version
486  bs.skipBytes(4); // flags
487  const auto opcode_size = bs.getU32();
488  bs.skipBytes(opcode_size);
489  }
490 
491  bs.setPosition(origPos);
492 
493  // okay, we may indeed have that many opcodes in here. now let's reserve
494  opcodes.reserve(opcode_count);
495 
496  for (auto i = 0U; i < opcode_count; i++) {
497  auto code = bs.getU32();
498  bs.skipBytes(4); // ignore version
499 #ifdef DEBUG
500  bs.skipBytes(4); // ignore flags
501 #else
502  auto flags = bs.getU32();
503 #endif
504  const auto opcode_size = bs.getU32();
505  ByteStream opcode_bs = bs.getStream(opcode_size);
506 
507  const char* opName = nullptr;
508  constructor_t opConstructor = nullptr;
509  try {
510  std::tie(opName, opConstructor) = Map.at(code);
511  } catch (std::out_of_range&) {
512  ThrowRDE("Unknown unhandled Opcode: %d", code);
513  }
514 
515  if (opConstructor != nullptr)
516  opcodes.emplace_back(opConstructor(ri, &opcode_bs));
517  else {
518 #ifndef DEBUG
519  // Throw Error if not marked as optional
520  if (!(flags & 1))
521 #endif
522  ThrowRDE("Unsupported Opcode: %d (%s)", code, opName);
523  }
524 
525  if (opcode_bs.getRemainSize() != 0)
526  ThrowRDE("Inconsistent length of opcode");
527  }
528 
529 #ifdef DEBUG
530  assert(opcodes.size() == opcode_count);
531 #endif
532 }
