61 RawImage Cr2Decoder::decodeOldFormat() {
62  uint32_t offset = 0;
63  if (mRootIFD->getEntryRecursive(CANON_RAW_DATA_OFFSET))
64  offset = mRootIFD->getEntryRecursive(CANON_RAW_DATA_OFFSET)->getU32();
65  else {
66  // D2000 is oh so special...
67  const auto* ifd = mRootIFD->getIFDWithTag(CFAPATTERN);
68  if (! ifd->hasEntry(STRIPOFFSETS))
69  ThrowRDE("Couldn't find offset");
70 
71  offset = ifd->getEntry(STRIPOFFSETS)->getU32();
72  }
73 
74  ByteStream b(DataBuffer(mFile->getSubView(offset), Endianness::big));
75  b.skipBytes(41);
76  int height = b.getU16();
77  int width = b.getU16();
78 
79  // some old models (1D/1DS/D2000C) encode two lines as one
80  // see: FIX_CANON_HALF_HEIGHT_DOUBLE_WIDTH
81  if (width > 2*height) {
82  height *= 2;
83  width /= 2;
84  }
85  width *= 2; // components
86 
87  mRaw->dim = {width, height};
88 
89  const ByteStream bs(
90  DataBuffer(mFile->getSubView(offset), Endianness::little));
91 
92  Cr2Decompressor l(bs, mRaw);
93  mRaw->createData();
94 
95  Cr2Slicing slicing(/*numSlices=*/1, /*sliceWidth=don't care*/ 0,
96  /*lastSliceWidth=*/width);
97  l.decode(slicing);
98 
99  // deal with D2000 GrayResponseCurve
100  TiffEntry* curve = mRootIFD->getEntryRecursive(static_cast<TiffTag>(0x123));
101  if (curve && curve->type == TIFF_SHORT && curve->count == 4096) {
102  auto table = curve->getU16Array(curve->count);
103  RawImageCurveGuard curveHandler(&mRaw, table, uncorrectedRawValues);
104 
105  // Apply table
106  if (!uncorrectedRawValues)
107  mRaw->sixteenBitLookup();
108  }
109 
110  return mRaw;
111 }

115 RawImage Cr2Decoder::decodeNewFormat() {
116  TiffEntry* sensorInfoE = mRootIFD->getEntryRecursive(CANON_SENSOR_INFO);
117  if (!sensorInfoE)
118  ThrowTPE("failed to get SensorInfo from MakerNote");
119 
120  assert(sensorInfoE != nullptr);
121 
122  if (isSubSampled() != (getSubSampling() != iPoint2D{1, 1}))
123  ThrowTPE("Subsampling sanity check failed");
124 
125  mRaw->dim = {sensorInfoE->getU16(1), sensorInfoE->getU16(2)};
126  mRaw->setCpp(1);
127  mRaw->isCFA = !isSubSampled();
128 
129  if (isSubSampled()) {
130  iPoint2D& subSampling = mRaw->metadata.subsampling;
131  subSampling = getSubSampling();
132  if (!(subSampling.x > 1 || subSampling.y > 1))
133  ThrowRDE("RAW is expected to be subsampled, but it's not");
134 
135  if (mRaw->dim.x % subSampling.x != 0)
136  ThrowRDE("Raw width is not a multiple of horizontal subsampling factor");
137  mRaw->dim.x /= subSampling.x;
138 
139  if (mRaw->dim.y % subSampling.y != 0)
140  ThrowRDE("Raw height is not a multiple of vertical subsampling factor");
141  mRaw->dim.y /= subSampling.y;
142 
143  mRaw->dim.x *= 2 + subSampling.x * subSampling.y;
144  }
145 
146  TiffIFD* raw = mRootIFD->getSubIFDs()[3].get();
147 
148  Cr2Slicing slicing;
149  // there are four cases:
150  // * there is a tag with three components,
151  // $ last two components are non-zero: all fine then.
152  // $ first two components are zero, last component is non-zero
153  // we let Cr2Decompressor guess it (it'll throw if fails)
154  // $ else the image is considered corrupt.
155  // * there is a tag with not three components, the image is considered
156  // corrupt. $ there is no tag, we let Cr2Decompressor guess it (it'll throw if
157  // fails)
158  TiffEntry* cr2SliceEntry = raw->getEntryRecursive(CANONCR2SLICE);
159  if (cr2SliceEntry) {
160  if (cr2SliceEntry->count != 3) {
161  ThrowRDE("Found RawImageSegmentation tag with %d elements, should be 3.",
162  cr2SliceEntry->count);
163  }
164 
165  if (cr2SliceEntry->getU16(1) != 0 && cr2SliceEntry->getU16(2) != 0) {
166  // first component can be either zero or non-zero, don't care
167  slicing = Cr2Slicing(/*numSlices=*/1 + cr2SliceEntry->getU16(0),
168  /*sliceWidth=*/cr2SliceEntry->getU16(1),
169  /*lastSliceWidth=*/cr2SliceEntry->getU16(2));
170  } else if (cr2SliceEntry->getU16(0) == 0 && cr2SliceEntry->getU16(1) == 0 &&
171  cr2SliceEntry->getU16(2) != 0) {
172  // PowerShot G16, PowerShot S120, let Cr2Decompressor guess.
173  } else {
174  ThrowRDE("Strange RawImageSegmentation tag: (%d, %d, %d), image corrupt.",
175  cr2SliceEntry->getU16(0), cr2SliceEntry->getU16(1),
176  cr2SliceEntry->getU16(2));
177  }
178  } // EOS 20D, EOS-1D Mark II, let Cr2Decompressor guess.
179 
180  const uint32_t offset = raw->getEntry(STRIPOFFSETS)->getU32();
181  const uint32_t count = raw->getEntry(STRIPBYTECOUNTS)->getU32();
182 
183  const ByteStream bs(
184  DataBuffer(mFile->getSubView(offset, count), Endianness::little));
185 
186  Cr2Decompressor d(bs, mRaw);
187  mRaw->createData();
188  d.decode(slicing);
189 
190  assert(getSubSampling() == mRaw->metadata.subsampling);
191 
192  if (mRaw->metadata.subsampling.x > 1 || mRaw->metadata.subsampling.y > 1)
193  sRawInterpolate();
194 
195  return mRaw;
196 }

216 void Cr2Decoder::decodeMetaDataInternal(const CameraMetaData* meta) {
217  int iso = 0;
218  mRaw->cfa.setCFA(iPoint2D(2,2), CFA_RED, CFA_GREEN, CFA_GREEN, CFA_BLUE);
219 
220  string mode;
221 
222  if (mRaw->metadata.subsampling.y == 2 && mRaw->metadata.subsampling.x == 2)
223  mode = "sRaw1";
224 
225  if (mRaw->metadata.subsampling.y == 1 && mRaw->metadata.subsampling.x == 2)
226  mode = "sRaw2";
227 
228  if (mRootIFD->hasEntryRecursive(ISOSPEEDRATINGS))
229  iso = mRootIFD->getEntryRecursive(ISOSPEEDRATINGS)->getU32();
230 
231  // Fetch the white balance
232  try{
233  if (mRootIFD->hasEntryRecursive(CANONCOLORDATA)) {
234  TiffEntry *wb = mRootIFD->getEntryRecursive(CANONCOLORDATA);
235  // this entry is a big table, and different cameras store used WB in
236  // different parts, so find the offset, default is the most common one
237  int offset = hints.get("wb_offset", 126);
238 
239  offset /= 2;
240  mRaw->metadata.wbCoeffs[0] = static_cast<float>(wb->getU16(offset + 0));
241  mRaw->metadata.wbCoeffs[1] = static_cast<float>(wb->getU16(offset + 1));
242  mRaw->metadata.wbCoeffs[2] = static_cast<float>(wb->getU16(offset + 3));
243  } else {
244  if (mRootIFD->hasEntryRecursive(CANONSHOTINFO) &&
245  mRootIFD->hasEntryRecursive(CANONPOWERSHOTG9WB)) {
246  TiffEntry *shot_info = mRootIFD->getEntryRecursive(CANONSHOTINFO);
247  TiffEntry *g9_wb = mRootIFD->getEntryRecursive(CANONPOWERSHOTG9WB);
248 
249  uint16_t wb_index = shot_info->getU16(7);
250  int wb_offset = (wb_index < 18) ? "012347800000005896"[wb_index]-'0' : 0;
251  wb_offset = wb_offset*8 + 2;
252 
253  mRaw->metadata.wbCoeffs[0] =
254  static_cast<float>(g9_wb->getU32(wb_offset + 1));
255  mRaw->metadata.wbCoeffs[1] =
256  (static_cast<float>(g9_wb->getU32(wb_offset + 0)) +
257  static_cast<float>(g9_wb->getU32(wb_offset + 3))) /
258  2.0F;
259  mRaw->metadata.wbCoeffs[2] =
260  static_cast<float>(g9_wb->getU32(wb_offset + 2));
261  } else if (mRootIFD->hasEntryRecursive(static_cast<TiffTag>(0xa4))) {
262  // WB for the old 1D and 1DS
263  TiffEntry* wb = mRootIFD->getEntryRecursive(static_cast<TiffTag>(0xa4));
264  if (wb->count >= 3) {
265  mRaw->metadata.wbCoeffs[0] = wb->getFloat(0);
266  mRaw->metadata.wbCoeffs[1] = wb->getFloat(1);
267  mRaw->metadata.wbCoeffs[2] = wb->getFloat(2);
268  }
269  }
270  }
271  } catch (RawspeedException& e) {
272  mRaw->setError(e.what());
273  // We caught an exception reading WB, just ignore it
274  }
275  setMetaData(meta, mode, iso);
276 }

286 iPoint2D Cr2Decoder::getSubSampling() const {
287  TiffEntry* CCS = mRootIFD->getEntryRecursive(CANON_CAMERA_SETTINGS);
288  if (!CCS)
289  ThrowRDE("CanonCameraSettings entry not found.");
290 
291  if (CCS->type != TIFF_SHORT)
292  ThrowRDE("Unexpected CanonCameraSettings entry type encountered ");
293 
294  if (CCS->count < 47)
295  return {1, 1};
296 
297  switch (uint16_t qual = CCS->getU16(46)) {
298  case 0:
299  return {1, 1};
300  case 1:
301  return {2, 2};
302  case 2:
303  return {2, 1};
304  default:
305  ThrowRDE("Unexpected SRAWQuality value found: %u", qual);
306  }
307 }

309 int Cr2Decoder::getHue() {
310  if (hints.has("old_sraw_hue"))
311  return (mRaw->metadata.subsampling.y * mRaw->metadata.subsampling.x);
312 
313  if (!mRootIFD->hasEntryRecursive(static_cast<TiffTag>(0x10))) {
314  return 0;
315  }
316  uint32_t model_id =
317  mRootIFD->getEntryRecursive(static_cast<TiffTag>(0x10))->getU32();
318  if (model_id >= 0x80000281 || model_id == 0x80000218 || (hints.has("force_new_sraw_hue")))
319  return ((mRaw->metadata.subsampling.y * mRaw->metadata.subsampling.x) - 1) >> 1;
320 
321  return (mRaw->metadata.subsampling.y * mRaw->metadata.subsampling.x);
322 }

324 // Interpolate and convert sRaw data.
325 void Cr2Decoder::sRawInterpolate() {
326  TiffEntry* wb = mRootIFD->getEntryRecursive(CANONCOLORDATA);
327  if (!wb)
328  ThrowRDE("Unable to locate WB info.");
329 
330  // Offset to sRaw coefficients used to reconstruct uncorrected RGB data.
331  uint32_t offset = 78;
332 
333  std::array<int, 3> sraw_coeffs;
334 
335  assert(wb != nullptr);
336  sraw_coeffs[0] = wb->getU16(offset + 0);
337  sraw_coeffs[1] =
338  (wb->getU16(offset + 1) + wb->getU16(offset + 2) + 1) >> 1;
339  sraw_coeffs[2] = wb->getU16(offset + 3);
340 
341  if (hints.has("invert_sraw_wb")) {
342  sraw_coeffs[0] = static_cast<int>(
343  1024.0F / (static_cast<float>(sraw_coeffs[0]) / 1024.0F));
344  sraw_coeffs[2] = static_cast<int>(
345  1024.0F / (static_cast<float>(sraw_coeffs[2]) / 1024.0F));
346  }
347 
348  mRaw->checkMemIsInitialized();
349  RawImage subsampledRaw = mRaw;
350  int hue = getHue();
351 
352  iPoint2D interpolatedDims = {
353  subsampledRaw->metadata.subsampling.x *
354  (subsampledRaw->dim.x /
355  (2 + subsampledRaw->metadata.subsampling.x *
356  subsampledRaw->metadata.subsampling.y)),
357  subsampledRaw->metadata.subsampling.y * subsampledRaw->dim.y};
358 
359  mRaw = RawImage::create(interpolatedDims, TYPE_USHORT16, 3);
360  mRaw->metadata.subsampling = subsampledRaw->metadata.subsampling;
361  mRaw->isCFA = false;
362 
363  Cr2sRawInterpolator i(mRaw, subsampledRaw->getU16DataAsUncroppedArray2DRef(),
364  sraw_coeffs, hue);
365 
366  /* Determine sRaw coefficients */
367  bool isOldSraw = hints.has("sraw_40d");
368  bool isNewSraw = hints.has("sraw_new");
369 
370  int version;
371  if (isOldSraw)
372  version = 0;
373  else {
374  if (isNewSraw) {
375  version = 2;
376  } else {
377  version = 1;
378  }
379  }
380 
381  i.interpolate(version);
382 }
