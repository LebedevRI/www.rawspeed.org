36 // This is a Natural Cubic Spline. The second derivative at curve ends are zero.
37 // See https://en.wikipedia.org/wiki/Spline_(mathematics)
38 // section "Algorithm for computing natural cubic splines"
39 
40 template <typename T = uint16_t,
41  typename = std::enable_if_t<std::is_arithmetic<T>::value>>
42 class Spline final {
43 public:
44  using value_type = T;
45 
46  // These are the constant factors for each segment of the curve.
47  // Each segment i will have the formula:
48  // f(x) = a[i] + b[i]*(x - x[i]) + c[i]*(x - x[i])^2 + d[i]*(x - x[i])^3
49  struct Segment {
50  double a;
51  double b;
52  double c;
53  double d;
54  };
55 
56 private:
57  int num_coords;
58  int num_segments;
59 
60  std::vector<int> xCp;
61  std::vector<Segment> segments;
62 
63  void prepare() {
64  // Extra values used during computation
65  std::vector<double> h(num_segments);
66  std::vector<double> alpha(num_segments);
67  std::vector<double> mu(num_coords);
68  std::vector<double> z(num_coords);
69 
70  for (int i = 0; i < num_segments; i++)
71  h[i] = xCp[i + 1] - xCp[i];
72 
73  for (int i = 1; i < num_segments; i++) {
74  Segment& sp = segments[i - 1];
75  Segment& s = segments[i];
76  Segment& sn = segments[i + 1];
77 
78  alpha[i] = (3. / h[i]) * (sn.a - s.a) - (3. / h[i - 1]) * (s.a - sp.a);
79  }
80 
81  mu[0] = z[0] = 0;
82 
83  for (int i = 1; i < num_segments; i++) {
84  const double l = 2 * (xCp[i + 1] - xCp[i - 1]) - (h[i - 1] * mu[i - 1]);
85  mu[i] = h[i] / l;
86  z[i] = (alpha[i] - h[i - 1] * z[i - 1]) / l;
87  }
88 
89  z.back() = segments.back().c = 0;
90 
91  for (int i = num_segments - 1; i >= 0; i--) {
92  Segment& s = segments[i];
93  Segment& sn = segments[i + 1];
94 
95  s.c = z[i] - mu[i] * sn.c;
96  s.b = (sn.a - s.a) / h[i] - h[i] * (sn.c + 2 * s.c) / 3.;
97  s.d = (sn.c - s.c) / (3. * h[i]);
98  }
99 
100  // The last segment is nonsensical, and was only used to temporarily store
101  // the a and c to simplify calculations, so drop that 'segment' now
102  segments.pop_back();
103 
104  assert(static_cast<typename decltype(segments)::size_type>(num_segments) ==
105  segments.size());
106  }
107 
108 public:
109  explicit Spline(const std::vector<iPoint2D>& control_points) {
110  assert(control_points.size() >= 2 &&
111  "Need at least two points to interpolate between");
112 
113  // Expect the X coords of the curve to start/end at the extreme values
114  assert(control_points.front().x == 0);
115  assert(control_points.back().x == 65535);
116 
117  assert(std::adjacent_find(
118  control_points.cbegin(), control_points.cend(),
119  [](const iPoint2D& lhs, const iPoint2D& rhs) -> bool {
120  return std::greater_equal<>()(lhs.x, rhs.x);
121  }) == control_points.cend() &&
122  "The X coordinates must all be strictly increasing");
123 
124 #ifndef NDEBUG
125  if (!std::is_floating_point<value_type>::value) {
126  // The Y coords must be limited to the range of value_type
127  std::for_each(control_points.cbegin(), control_points.cend(),
128  [](const iPoint2D& p) -> void {
129  assert(p.y >= std::numeric_limits<value_type>::min());
130  assert(p.y <= std::numeric_limits<value_type>::max());
131  });
132  }
133 #endif
134 
135  num_coords = control_points.size();
136  num_segments = num_coords - 1;
137 
138  xCp.resize(num_coords);
139  segments.resize(num_coords);
140  for (int i = 0; i < num_coords; i++) {
141  xCp[i] = control_points[i].x;
142  segments[i].a = control_points[i].y;
143  }
144 
145  prepare();
146  }
147 
148  std::vector<Segment> getSegments() const { return segments; }
149 
150  std::vector<value_type> calculateCurve() const {
151  std::vector<value_type> curve(65536);
152 
153  for (int i = 0; i < num_segments; i++) {
154  const Segment& s = segments[i];
155 
156  for (int x = xCp[i]; x <= xCp[i + 1]; x++) {
157  double diff = x - xCp[i];
158  double diff_2 = diff * diff;
159  double diff_3 = diff * diff * diff;
160 
161  double interpolated = s.a + s.b * diff + s.c * diff_2 + s.d * diff_3;
162 
163  if (!std::is_floating_point<value_type>::value) {
164  interpolated = std::max(
165  interpolated, double(std::numeric_limits<value_type>::min()));
166  interpolated = std::min(
167  interpolated, double(std::numeric_limits<value_type>::max()));
168  }
169 
170  curve[x] = interpolated;
171  }
172  }
173 
174  return curve;
175  }
176 };
