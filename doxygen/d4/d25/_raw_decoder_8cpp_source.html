57 void RawDecoder::decodeUncompressed(const TiffIFD *rawIFD, BitOrder order) {
58  TiffEntry *offsets = rawIFD->getEntry(STRIPOFFSETS);
59  TiffEntry *counts = rawIFD->getEntry(STRIPBYTECOUNTS);
60  uint32_t yPerSlice = rawIFD->getEntry(ROWSPERSTRIP)->getU32();
61  uint32_t width = rawIFD->getEntry(IMAGEWIDTH)->getU32();
62  uint32_t height = rawIFD->getEntry(IMAGELENGTH)->getU32();
63  uint32_t bitPerPixel = rawIFD->getEntry(BITSPERSAMPLE)->getU32();
64 
65  if (width == 0 || height == 0 || width > 5632 || height > 3720)
66  ThrowRDE("Unexpected image dimensions found: (%u; %u)", width, height);
67 
68  mRaw->dim = iPoint2D(width, height);
69 
70  if (counts->count != offsets->count) {
71  ThrowRDE("Byte count number does not match strip size: "
72  "count:%u, stips:%u ",
73  counts->count, offsets->count);
74  }
75 
76  if (yPerSlice == 0 || yPerSlice > static_cast<uint32_t>(mRaw->dim.y) ||
77  roundUpDivision(mRaw->dim.y, yPerSlice) != counts->count) {
78  ThrowRDE("Invalid y per slice %u or strip count %u (height = %u)",
79  yPerSlice, counts->count, mRaw->dim.y);
80  }
81 
82  switch (bitPerPixel) {
83  case 12:
84  case 14:
85  break;
86  default:
87  ThrowRDE("Unexpected bits per pixel: %u.", bitPerPixel);
88  }
89 
90  vector<RawSlice> slices;
91  slices.reserve(counts->count);
92  uint32_t offY = 0;
93 
94  for (uint32_t s = 0; s < counts->count; s++) {
95  RawSlice slice;
96  slice.offset = offsets->getU32(s);
97  slice.count = counts->getU32(s);
98 
99  if (slice.count < 1)
100  ThrowRDE("Slice %u is empty", s);
101 
102  if (offY + yPerSlice > height)
103  slice.h = height - offY;
104  else
105  slice.h = yPerSlice;
106 
107  offY += yPerSlice;
108 
109  if (!mFile->isValid(slice.offset, slice.count))
110  ThrowRDE("Slice offset/count invalid");
111 
112  slices.push_back(slice);
113  }
114 
115  if (slices.empty())
116  ThrowRDE("No valid slices found. File probably truncated.");
117 
118  assert(height <= offY);
119  assert(slices.size() == counts->count);
120 
121  mRaw->createData();
122 
123  // Default white level is (2 ** BitsPerSample) - 1
124  mRaw->whitePoint = (1UL << bitPerPixel) - 1UL;
125 
126  offY = 0;
127  for (const RawSlice& slice : slices) {
128  UncompressedDecompressor u(
129  ByteStream(DataBuffer(mFile->getSubView(slice.offset, slice.count),
130  Endianness::little)),
131  mRaw);
132  iPoint2D size(width, slice.h);
133  iPoint2D pos(0, offY);
134  bitPerPixel = static_cast<int>(
135  static_cast<uint64_t>(static_cast<uint64_t>(slice.count) * 8U) /
136  (slice.h * width));
137  const auto inputPitch = width * bitPerPixel / 8;
138  if (!inputPitch)
139  ThrowRDE("Bad input pitch. Can not decode anything.");
140 
141  u.readUncompressedRaw(size, pos, inputPitch, bitPerPixel, order);
142 
143  offY += slice.h;
144  }
145 }

159 bool RawDecoder::checkCameraSupported(const CameraMetaData* meta,
160  const string& make, const string& model,
161  const string& mode) {
162  mRaw->metadata.make = make;
163  mRaw->metadata.model = model;
164  const Camera* cam = meta->getCamera(make, model, mode);
165  if (!cam) {
166  askForSamples(meta, make, model, mode);
167 
168  if (failOnUnknown)
169  ThrowRDE("Camera '%s' '%s', mode '%s' not supported, and not allowed to guess. Sorry.", make.c_str(), model.c_str(), mode.c_str());
170 
171  // Assume the camera can be decoded, but return false, so decoders can see that we are unsure.
172  return false;
173  }
174 
175  if (!cam->supported)
176  ThrowRDE("Camera not supported (explicit). Sorry.");
177 
178  if (cam->decoderVersion > getDecoderVersion())
179  ThrowRDE("Camera not supported in this version. Update RawSpeed for support.");
180 
181  hints = cam->hints;
182  return true;
183 }

185 void RawDecoder::setMetaData(const CameraMetaData* meta, const string& make,
186  const string& model, const string& mode,
187  int iso_speed) {
188  mRaw->metadata.isoSpeed = iso_speed;
189  const Camera* cam = meta->getCamera(make, model, mode);
190  if (!cam) {
191  askForSamples(meta, make, model, mode);
192 
193  if (failOnUnknown)
194  ThrowRDE("Camera '%s' '%s', mode '%s' not supported, and not allowed to guess. Sorry.", make.c_str(), model.c_str(), mode.c_str());
195 
196  return;
197  }
198 
199  // Only override CFA with the data from cameras.xml if it actually contained
200  // the CFA.
201  if (cam->cfa.getSize().area() > 0)
202  mRaw->cfa = cam->cfa;
203 
204  mRaw->metadata.canonical_make = cam->canonical_make;
205  mRaw->metadata.canonical_model = cam->canonical_model;
206  mRaw->metadata.canonical_alias = cam->canonical_alias;
207  mRaw->metadata.canonical_id = cam->canonical_id;
208  mRaw->metadata.make = make;
209  mRaw->metadata.model = model;
210  mRaw->metadata.mode = mode;
211 
212  if (applyCrop) {
213  iPoint2D new_size = cam->cropSize;
214 
215  // If crop size is negative, use relative cropping
216  if (new_size.x <= 0)
217  new_size.x = mRaw->dim.x - cam->cropPos.x + new_size.x;
218 
219  if (new_size.y <= 0)
220  new_size.y = mRaw->dim.y - cam->cropPos.y + new_size.y;
221 
222  mRaw->subFrame(iRectangle2D(cam->cropPos, new_size));
223  }
224 
225  const CameraSensorInfo *sensor = cam->getSensorInfo(iso_speed);
226  mRaw->blackLevel = sensor->mBlackLevel;
227  mRaw->whitePoint = sensor->mWhiteLevel;
228  mRaw->blackAreas = cam->blackAreas;
229  if (mRaw->blackAreas.empty() && !sensor->mBlackLevelSeparate.empty()) {
230  auto cfaArea = mRaw->cfa.getSize().area();
231  if (mRaw->isCFA && cfaArea <= sensor->mBlackLevelSeparate.size()) {
232  for (auto i = 0UL; i < cfaArea; i++) {
233  mRaw->blackLevelSeparate[i] = sensor->mBlackLevelSeparate[i];
234  }
235  } else if (!mRaw->isCFA && mRaw->getCpp() <= sensor->mBlackLevelSeparate.size()) {
236  for (uint32_t i = 0; i < mRaw->getCpp(); i++) {
237  mRaw->blackLevelSeparate[i] = sensor->mBlackLevelSeparate[i];
238  }
239  }
240  }
241 
242  // Allow overriding individual blacklevels. Values are in CFA order
243  // (the same order as the in the CFA tag)
244  // A hint could be:
245  // <Hint name="override_cfa_black" value="10,20,30,20"/>
246  string cfa_black = hints.get("override_cfa_black", string());
247  if (!cfa_black.empty()) {
248  vector<string> v = splitString(cfa_black, ',');
249  if (v.size() != 4) {
250  mRaw->setError("Expected 4 values '10,20,30,20' as values for override_cfa_black hint.");
251  } else {
252  for (int i = 0; i < 4; i++) {
253  mRaw->blackLevelSeparate[i] = stoi(v[i]);
254  }
255  }
256  }
